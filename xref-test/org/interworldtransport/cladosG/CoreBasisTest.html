32  	class testCaching {
33  		@Test
34  		public void testCachePrefill() throws GeneratorRangeException {
35  			GCache.INSTANCE.clearBases();
36  			for (byte k = 0; k < 11; k++)
37  				GBuilder.createBasis(k);
38  			assertTrue(GCache.INSTANCE.getBasisListSize() == 11);
39  			for (byte k = 0; k < 11; k++)
40  				GCache.INSTANCE.removeBasis(k);
41  			assertTrue(GCache.INSTANCE.getBasisListSize() == 0);
42  		}
43  		@Test
44  		public void testCachedBasis() throws GeneratorRangeException {
45  			GCache.INSTANCE.clearBases();
46  			Basis tB1 = GBuilder.createBasis((byte) 3);	//Builder cached it
47  			assertTrue(GCache.INSTANCE.getBasisListSize() == 1); 	
48  			Basis tB2 = GBuilder.createBasis((byte) 3); 	//Building another like it
49  			assertTrue(tB1 == tB2);		//Builder noticed identical size and returned first one instead.
50  			GBuilder.createBasis(Generator.EA);					//Builder cached it
51  			assertTrue(GCache.INSTANCE.getBasisListSize() == 2);	//Two now
52  			GCache.INSTANCE.removeBasis((byte) 3);				//Get rid of first one
53  			assertTrue(GCache.INSTANCE.getBasisListSize() == 1); 
54  			Optional<Basis> get10 = GCache.INSTANCE.findBasis((byte) 10);
55  			assertTrue(get10.isPresent()); //Earlier removal got rid of the correct one.
56  			assertTrue(GCache.INSTANCE.removeBasis((byte) 3));	//Get rid of first one again doesn't error.
57  		}
58  	}
