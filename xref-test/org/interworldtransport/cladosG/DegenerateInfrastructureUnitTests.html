72       * GProduct behavior depends on generator signatures, so special testing around degenerate cases is 
73       * needed in the BladeDuet. The issue is that degenerate generators square to zero so a pair of them 
74       * in the BladeDuet before simplification must default the returned Blade to a scalar of zero size. 
75       * There is no 'size' in blades, though, so the sign is slightly repurposed to include ZERO for this case.

103         /**
104          * This test checks two mostly filled blades to ensure their products resolve to scalars when a degenerate 
105          * generator is involved in both operands and doesn't with a Euclidian signature. This one covers the 
106          * typical PGA signature
107          */
108         @Test
109         public void testStaticPGASimplify() {
110                 firstB = new Blade((byte) 4, g);
111                 secondB = new Blade((byte) 4, i);
112 
113                 out = BladeDuet.simplify(firstB, firstB, pgasig);
114                 assertTrue(Blade.isScalar(out));
115                 assertTrue(out.sign() == 0);
116 
117                 out = BladeDuet.simplify(firstB, secondB, pgasig);
118                 assertTrue(Blade.isScalar(out));
119                 assertTrue(out.sign() == 0);
120 
121                 out = BladeDuet.simplify(firstB, secondB, euclidsig4);
122                 assertTrue(CanonicalBlade.isNBlade(out, (byte) 1));
123                 assertTrue(out.sign() == -1);               //The shared trivector squares to -1
124         }

209         @Test
210         public void testSigns() {
211                 try {
212                     GProduct tGP = new GProduct(pSig301);
213                     assertTrue(tGP.getACommuteSign(1, 2) == 1);     //They anticommute
214                     assertTrue(tGP.getCommuteSign(1, 9) == 1);      //They commute
215                     assertTrue(tGP.getCommuteSign(2, 6) == 1);      //They commute
216                     assertFalse(tGP.getACommuteSign(1, 12) == 1);   //They multiply to zero
217                     assertTrue(tGP.getCommuteSign(1, 11) == 1);     //They multiply to zero
218                     assertFalse(tGP.getCommuteSign(1, 3) == 1);     //They anticommute
219 
220                     assertTrue(tGP.getSign(1, 2) == 1);             //Should be positive on row 1
221                     assertTrue(tGP.getSign(2, 1) == -1);            //Should be neg to get anticommute
222                     assertTrue(tGP.getSign(15, 15) == 0);           //+1 for all degenerate cases.
223                     assertTrue(tGP.getResult(15, 15) == 0);         //PScalar squares to 0.
224                     assertTrue(tGP.signature().length() == 4);
225                 } catch (BadSignatureException esig) {
226                     ;
227                 }
228         }

230         @Test
231         public void test101() throws BadSignatureException {
232                 GProduct tGP = new GProduct(pSig101);
233                 assertTrue(tGP.signature().equals("0+"));
234                 assertTrue(tGP.getGradeCount() == 3);
235                 assertTrue(tGP.getBladeCount() == (1 << 2));
236                 int tS = (1 << 2);
237                 int tSum = tS * (tS + 1) / 2;
238                 for (int k = 0; k < tGP.getBladeCount(); k++) {
239                     int[] tSpot = tGP.getResult(k);
240                     int tSumP = 0;
241                     int tScalarCount = 0;
242                     for (int j = 0; j < tSpot.length; j++) {
243                         tSumP += Math.abs(tSpot[j]);
244                         if (Math.abs(tSpot[j]) == 0)
245                             tScalarCount++;
246                     }
247                     assertTrue(tSum >= tSumP);
248                     assertTrue(tScalarCount == 0 | tScalarCount == tGP.getBladeCount()/2);
249                 }
250         }
