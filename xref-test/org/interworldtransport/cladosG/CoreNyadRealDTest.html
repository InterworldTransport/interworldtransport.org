793 		@BeforeEach
794 		void setUp() throws CladosNyadException, CladosMonadException {
795 			thing1 = GBuilder.createNyadUsingMonad(motion, "");
796 			thing1.append(property);
797 			((RealD) property.getWeights().getScalar()).setReal(2.0f);		//Now property is scalar = 2.
798 			((RealD) motion.getWeights().getScalar()).setReal(1.0f);			//Now motion is scalar = 1.
799 			newMotion = GBuilder.copyOfMonad(motion, "CopyMotion");
800 			newMotion2 = GBuilder.copyOfMonad(motion, "Copy2Motion");
801 			newProperty = GBuilder.copyOfMonad(property, "CopyProperty");
802 			newProperty2 = GBuilder.copyOfMonad(property, "Copy2Property");
803 		}
804 
805 		@Test
806 		void testSymmetricCompression() throws CladosNyadException {
807 			assertTrue(thing1.getMonadAt(0) == motion);
808 			assertTrue(thing1.getMonadAt(1) == property);
809 	
810 			thing1.compressSymm(property, motion);									//motion gets property's algebra and cardinal
811 			assertTrue(thing1.getMonadAt(0) == property);
812 	
813 			assertTrue(thing1.arity() == 1);									
814 			assertTrue(thing1.algrity() == 1);									//(It happens with only one monad)
815 			assertTrue(((RealD) thing1.getMonadAt(0).getWeights().getScalar()).getReal() == 2.0f);
816 			
817 			thing1.append(newMotion).pop(newMotion);								//Back to a juxtaposition
818 			thing1.compressSymm(newMotion, property);								//property gets newMotion's algebra and cardinal
819 			assertTrue(thing1.arity() == 1);									
820 			assertTrue(thing1.algrity() == 1);									//(It happens with only one monad)
821 			assertTrue(((RealD) thing1.getMonadAt(0).getWeights().getScalar()).getReal() == 2.0f);
822 
823 			thing1.append(newProperty);
824 			assertDoesNotThrow(() -> thing1.compressSymm(0, 1));	//property winds up with motion's algebra and cardinal
825 			assertTrue(thing1.arity() == 1);									//Test caused mutation of thing1.
826 																					//So doing it again should causes out of bounds
827 			assertThrows(IndexOutOfBoundsException.class, () -> thing1.compressSymm(0, 1));
828 
829 			thing1.append(newProperty2);											//Back to juxtaposition with newMotion
830 			assertThrows(CladosNyadException.class, () -> thing1.compressSymm(newMotion, newMotion2));
831 																					//Should bark because newMotion2 not in list
832 			assertThrows(CladosNyadException.class, () -> thing1.compressSymm(newMotion2, newMotion));
