20  public class ShirokovConjugationTest {
21      final Cardinal speed = FBuilder.createCardinal("c");
22      final String footName = "here";
23  	final String aName = "movement";
24  	final String mNameU = "velocity";
25  	final String sigD = "-+++";
26      final RealF by1 = (RealF) FBuilder.REALF.createONE(speed);
27  	final RealF by2 = (RealF) FBuilder.REALF.createONE(speed).scale(BY2_F);
28      final Foot here = GBuilder.createFootLike(footName, speed);
29  
30      Blade time, spaceX, planeTX;
31      Monad motion, reflect, boost;	
32  	Nyad thing1;
33  
34      /*
35  	 * The 'use' monad is in the same nyad as the keep monad.
36  	 * Test both direct reference and indexed reference of the monads.
37  	 */
38  	@Test
39  	void testSandwichInside() throws CladosException, CladosNyadException, BadSignatureException {
40          Algebra physical = GBuilder.createAlgebraWithFoot(here, aName, sigD);                        //A motion algebra
41          Scale<RealF> coeff = GBuilder.createScale(CladosField.REALF, physical.getBasis(), speed);    //ZEROES to start
42          motion = GBuilder.createMonadWithAlgebra(coeff, physical, mNameU);
43          reflect = GBuilder.copyOfMonad(motion,"Reflector");
44  		boost = GBuilder.copyOfMonad(motion, "Booster");
45  
46          time = motion.getAlgebra().getBasis().getSingleBlade(motion.getAlgebra().getGradeRange((byte) 1)[0]);
47          motion.getWeights().getMap().put(time, by1);	                                            //motion is time-like 1-blade
48  		motion.setGradeKey();
49  
50          spaceX = motion.getAlgebra().getBasis().getSingleBlade(motion.getAlgebra().getGradeRange((byte) 1)[0]+1);
51  		reflect.getWeights().getMap().put(spaceX, by1);	//reflect is space-like 1-blade
52  		reflect.setGradeKey();
53  
54          planeTX = motion.getAlgebra().getGP().getResult(spaceX, time);
55  		boost.getWeights().setScalar(by2);
56  		boost.getWeights().getMap().put(planeTX, RealF.copyOf(by2));
57  		boost.setGradeKey();
58  
59          thing1 = GBuilder.createNyadUsingMonad(motion, "testNyad");
60          thing1.append(reflect);
61          thing1.append(boost);
62  
63          assertTrue(thing1.getMonadAt(0) == motion);
64          assertTrue(thing1.getMonadAt(1) == reflect);
65          assertTrue(thing1.getMonadAt(2) == boost);
66  		assertTrue(((RealF) motion.getWeights().getMap().get(time)).getReal() > 0 );
67          assertTrue(((RealF) reflect.getWeights().getMap().get(spaceX)).getReal() > 0 );
68          assertTrue(((RealF) boost.getWeights().getScalar()).getReal() > 0 );
69          assertTrue(((RealF) boost.getWeights().getMap().get(planeTX)).getReal() > 0 );
70  
71  		boost.getWeights().conjugateShirokov(2);											//Shouldn't switch sign on the scalar or vector.
72  
73  		assertTrue(((RealF) boost.getWeights().getScalar()).getReal() > 0 );
74          assertTrue(((RealF) boost.getWeights().getMap().get(planeTX)).getReal() < 0 );
75  	}
