202 		public void testAddingGenerators() {
203 			Blade testThis = Blade.createBlade(tB4);
204 			assertFalse(testThis == tB4);
205 			assertTrue(testThis.rank() == 2);
206 			testThis.add(Generator.E2);
207 			assertTrue(testThis.key() == tB4.key());	//Because E2 was already present
208 			testThis.add(Generator.E4);
209 			assertFalse(testThis.key() == tB4.key());	//Because E4 was NOT present
210 			assertTrue(testThis.rank() == 3);
211 			assertDoesNotThrow(() -> testThis.add(Generator.EC));
212 			assertTrue(testThis.rank() == 3);
213 
214 			assertDoesNotThrow(() -> Blade.augmentBlade(testThis, Generator.E5));
215 			Blade testThis2 = Blade.augmentBlade(testThis, Generator.E5);
216 			assertTrue(testThis2.rank() == 4); //Augment adds room for one more generator. The next one.
217 			assertTrue(testThis2.maxGenerator() == 5);
218 
219 			Blade testThis3 = Blade.augmentBlade(testThis2, Generator.EC);
220 			assertTrue(testThis3.maxGenerator() == 12);
221 			assertFalse(testThis3.rank() == 6); //Augment increases rank by one.
222 
223 			Blade testThis4 = Blade.augmentBlade(testThis3, Generator.EF);
224 			assertTrue(testThis4.maxGenerator() == 15);
