25  package org.interworldtransport.cladosF;
26  
27  /**
28   * This class is a simple string holder that names a division field in a manner
29   * similar to the concept of 'units' in a physical model. Examples... "metres",
30   * "litres", "amperes". There is no requirement one use a particular standard
31   * for unit systems, though. They are free strings.
32   * <br><br>
33   * Cardinals are meant to provide all DivFields instances with a single object
34   * they may share that names them in order to speed field comparisons in
35   * TypeMatch methods. A class using fields for calculations would declare only
36   * one of these and then share the reference among all the objects that may be
37   * operated upon by the two primary opertaions of the field. ( +, * )
38   * <br><br>
39   * This may seem like a waste of time, but it is useful when a class must be
40   * prepared to use different kinds of fields without knowing in advance which
41   * one will be created. It is most important when an object exists that uses
42   * DivFields with unknown pedigrees. The Cardinal can be checked to discover
43   * intended uses and avoid mixing apples and oranges.
44   * <br><br>
45   * One consequence of this approach is that two division fields might use
46   * different Cardinal objects of the same name. The TypeMatch method in a
47   * ProtoN will state that they are different because object equality is tested
48   * instead of string content equality. This allows an application writer to keep
49   * two distinct systems of numbers apart in their application even though the
50   * fields are internally identical. This is useful when objects in one algebra
51   * might be scaled different than objects in another. If one never intends to
52   * use this feature, though, it is easy to avoid. Simply re-use a Cardinal.
53   * <br><br>
54   * Yes. This is soft typing to an OOP developer. The point is that it allows for
55   * differences between objects that can't assume 'scale' means the same thing to
56   * both of them. Since 'scale' and 'multiply' are NOT the same concepts, this
57   * distinction is needed in some scenarios.
58   * <br><br>
59   * 
60   * @version 2.0
61   * @author Dr Alfred W Differ
62   */
63  public final class Cardinal {
64  	/**
65  	 * When one does not wish to use Cardinals much, the best bypass is to use this
66  	 * factory method to create a minimal cardinal. The unit value will be set to
67  	 * the name of the CladosField offered
68  	 * <br>
69  	 * @param pT CladosField Name the element of the enumeration will be re-used as
70  	 *           the unit value within the Cardinal.
71  	 * @return Cardinal
72  	 */
73  	public static final Cardinal generate(CladosField pT) {
74  		return new Cardinal(pT);
75  	}
76  
77  	/**
78  	 * When one does wish to use Cardinals, the best approach is to use this factory
79  	 * method to create a typical cardinal. The unit value is set to the string.
80  	 * <br>
81  	 * @param pT String The Cardinal's unit value will be set to this string.
82  	 * @return Cardinal
83  	 */
84  	public static final Cardinal generate(String pT) {
85  		return new Cardinal(pT);
86  	}
87  
88  	private final String unit;
89  
90  	/*
91  	 * This private constructor re-uses the CladosField enumeration as a basic unit.
92  	 * This constructor exists so objects using CladosF classes have a basic unit that
93  	 * doesn't have to mean anything. The other option they could have is to use a null 
94  	 * Cardinal, but then type matching likely fails. Anyone using the CladosField 
95  	 * enumeration likely prefers not to use Cardinals at all.
96  	 */
97  	private Cardinal(CladosField pT) {
98  		unit = pT.name();
99  		FCache.INSTANCE.appendCardinal(this);
100 	}
101 
102 	/*
103 	 * This private constructor accepts a string as the name of some 'units' associated 
104 	 * with a CladosF number class. The string may be any legal java string.
105 	 */
106 	private Cardinal(String pT) {
107 		unit = pT;
108 		FCache.INSTANCE.appendCardinal(this);
109 	}
110 
111 	@Override
112 	public boolean equals(Object obj) {
113 		if (this == obj)										return true;
114 		if (obj == null)										return false;
115 		if (getClass() != obj.getClass())						return false;
116 		if (!this.getUnit().equals(((Cardinal)obj).getUnit()))	return false;
117 																return true;
118 	}
119 
120 	/**
121 	 * Simple gettor method for the name of the 'unit' represented by this Cardinal.
122 	 * <br>
123 	 * @return String This string names the 'unit type' represented by the Cardinal
124 	 */
125 	public String getUnit() {
126 		return unit;
127 	}
128 
129 	@Override
130 	public int hashCode() {
131 		final int prime = 31;
132 		int result = 1;
133 		result = prime * result + ((unit == null) ? 0 : unit.hashCode());
134 		return result;
135 	}
136 
137 	/**
138 	 * Similar to a toString() method, but it focuses upon an XML style output.
139 	 * @param pCard Cardinal to be exported as XML.
140 	 * @param indent String to assist with human readability of XML output.
141 	 * <br>
142 	 * @return String XML compatible sub-unit for code relying on export-able Cardinals.
143 	 */
144 	public final static String toXMLString(Cardinal pCard, String indent) {
145 		if (indent == null)			indent ="";
146 		return (indent + "<Cardinal unit=\"" + pCard.unit + "\" />\n");
147 	}
148 }
