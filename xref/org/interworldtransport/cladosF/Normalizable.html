25  package org.interworldtransport.cladosF;
26  
27  /**
28   * This interface implements the part of the concept of a Division Field from
29   * mathematics. Specifically, if is the portion that says the field supports the
30   * notion of 'length' or 'magnitude' when several of them are used as scaling
31   * factors in vector spaces.
32   * <br><br>
33   * For real numbers this is just the idea of distance from a number line origin.
34   * In a vector space where several are involved, one uses a root-mean-square
35   * method to combine them. Think Pythagorus.
36   * <br><br>
37   * For complex numbers, it is the radius in a polar representation. When several
38   * are involved, it is Pythagorus again, but the squares are formed with numbers
39   * and their conjugates, but computation is made easier by ignoring the
40   * imaginary and just squaring the components.
41   * <br><br>
42   * @version 2.0
43   * @author Dr Alfred W Differ
44   */
45  public interface Normalizable {
46  	/**
47  	 * This is the square root of the SQ Modulus. It is smarter to calculate
48  	 * SQModulus first.
49  	 * <br>
50  	 * @return Number
51  	 */
52  	public abstract Number modulus();
53  
54  	/**
55  	 * This function delivers the sum of the squares of the numeric values. Many
56  	 * times it is the modulus squared that is actually needed so it makes sense to
57  	 * calculate this before the modulus itself.
58  	 * <br>
59  	 * @return Number
60  	 */
61  	public abstract Number sqModulus();
62  }
