25  package org.interworldtransport.cladosF;
26  
27  import java.util.ArrayList;
28  import java.util.Optional;
29  import java.util.Set;
30  
31  /**
32   * Any classes within CladosF which would benefit from a supporting cache make
33   * use of this singleton enumeration as a 'builder'. Nothing fancy here
34   * otherwise. Just simple create, append, find, and remove capabilities backed
35   * by ArrayLists of cached objects.
36   * <br><br>
37   * At present, the only thing in CladosF being cached is Cardinals. cladosG.Algebra
38   * has its protonumber which is little more than a Cardinal. cladosG.Foot has its
39   * list of cardinals too. So there are a number of places where cardinals get attached.
40   * The point of this cache, therefore, is to provide a global list to the modeler
41   * who wants to ensure object references match perfectly when they should. That is
42   * accomplished by setting the protonumber's Cardinal to one in this list BY REFERENCE
43   * instead of by string.
44   * <br><br>
45   * Be aware that the list of cached objects is inheriently mutable. Alteration of an
46   * object here alters it everywhere it is referenced. ALSO... because of this cache, 
47   * it is unlikely that removal of object references elsewhere (set to null) will 
48   * cause the allocated memory to be garbage collected. Any linked here in the cache
49   * will persist while the cache does or the reference to it in the cache does.
50   * <br><br>
51   * @version 2.0
52   * @author Dr Alfred W Differ
53   */
54  public enum FCache {
55  	/**
56  	 * There is an implicit private constructor for this singleton, but we won't
57  	 * override it.
58  	 * <br>
59  	 * This ensures only one of these is active at a time. All FCache references
60  	 * must refer back to this one cache.
61  	 */
62  	INSTANCE;
63  
64  	/**
65  	 * Heads Up! This ArrayList ensures this 'enumeration' is mutable. It's a cache,
66  	 * so this should surprise no one. It is supposed to be instantiated once and
67  	 * then used to keep track of Cardinals in use.
68  	 */
69  	private ArrayList<Cardinal> listOfCardinals = new ArrayList<Cardinal>(1);
70  
71  	/**
72  	 * Method appends offered Cardinal to cache IF one by that name is not already
73  	 * present. If it IS, nothing is done and the method silently returns.
74  	 * <br>
75  	 * @param pIn Cardinal to be appended to the cache IF it isn't already present.
76  	 */
77  	public void appendCardinal(Cardinal pIn) {
78  		Optional<Cardinal> test = findCardinal(pIn.getUnit());
79  		if (test.isEmpty())
80  			listOfCardinals.add(pIn);
81  	}
82  
83  	/**
84  	 * Method appends offered Cardinals to cache IF not already present. If any ARE,
85  	 * nothing is done and the method silently loops through remaining Cardinals
86  	 * inserting ones that aren't.
87  	 * <br>
88  	 * @param pIn Cardinal Set to be appended to the cache IF they aren't already present.
89  	 */
90  	public void appendCardinal(Set<Cardinal> pIn) {
91  		pIn.stream().forEach(pC -> {
92  			Optional<Cardinal> test = findCardinal(pC.getUnit());
93  			if (test.isEmpty())
94  				listOfCardinals.add(pC);
95  		});
96  	}
97  
98  	/**
99  	 * This method returns an Optional of Cardinal using the string name offered for
100 	 * the search. If found, the optional will be engaged. If not, it will be
101 	 * disengaged. IF by some chance there are two cardinals in the cache by the
102 	 * same name (which should NOT happen) the first one found will be returned.
103 	 * <br>
104 	 * @param pName String name of a Cardinal to be found in the cache
105 	 * @return Optional of Cardinal matching the name offered.
106 	 */
107 	public Optional<Cardinal> findCardinal(String pName) {
108 		return listOfCardinals.stream().filter(x -> x.getUnit().equals(pName)).findFirst();
109 	}
110 
111 	/**
112 	 * This method clears the Cardinal cache.
113 	 */
114 	public void clearCardinals() {
115 		listOfCardinals.clear();
116 	}
117 
118 	/**
119 	 * This is a brute force way of retrieving Cardinals in the cache. It should
120 	 * rarely be used.
121 	 * <br>
122 	 * @param pLoc integer index to be used when returning the listed Cardinal
123 	 * @return Cardinal at pLoc index will be returned.
124 	 */
125 	public Cardinal getCardinal(int pLoc) {
126 		if (listOfCardinals.size() < 1 | listOfCardinals.size() < pLoc+1)
127 			return null;
128 		return listOfCardinals.get(pLoc);
129 	}
130 
131 	/**
132 	 * This method reports the size of the Cardinal cache.
133 	 * <br>
134 	 * @return int size of the Cardinal cache.
135 	 */
136 	public int getCardinalListSize() {
137 		return listOfCardinals.size();
138 	}
139 
140 	/**
141 	 * This is just a pass-thru method to test whether the cache of Cardinals
142 	 * is empty.
143 	 * <br> 
144 	 * @return boolean True if the cache of Cardinals is empty. False otherwise.
145 	 */
146 	public boolean isEmpty() {
147 		return listOfCardinals.isEmpty();
148 	}
149 
150 	/**
151 	 * This method supports the removal of a Cardinal from the cache.
152 	 * <br>
153 	 * @param pCard Cardinal to be removed
154 	 * @return boolean True if Cardinal found and removed. False if not found or
155 	 *         removal fails.
156 	 */
157 	public boolean removeCardinal(Cardinal pCard) {
158 		return listOfCardinals.remove(pCard);
159 	}
160 }
