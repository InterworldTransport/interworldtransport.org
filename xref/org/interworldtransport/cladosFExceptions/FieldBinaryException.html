25  package org.interworldtransport.cladosFExceptions;
26  
27  import org.interworldtransport.cladosF.*;
28  
29  /**
30   * This class is designed to be the top of the binary Field exception family.
31   * All instances of descendants are assumed to originate from Field related
32   * problems. Common elements from each exception class are found here.
33   * <br><br>
34   * @version 1.0
35   * @author Dr Alfred W Differ
36   *
37   */
38  
39  public class FieldBinaryException extends FieldException {
40  	/**
41  	 * Serialization ID
42  	 */
43  	private static final long serialVersionUID = -8985905609966643248L;
44  	/**
45  	 * This is the second object involved in the Exception if the operation is a
46  	 * binary operation.
47  	 */
48  	private ProtoN Second;
49  
50  	/**
51  	 * This method is the main constructor of all Clados Field Binary Exceptions.
52  	 * <br>
53  	 * @param pSource  ProtoN
54  	 * @param pMessage String
55  	 * @param pSecond  ProtoN
56  	 */
57  	public FieldBinaryException(ProtoN pSource, String pMessage, ProtoN pSecond) {
58  		super(pSource, pMessage);
59  		Second = pSecond;
60  	}
61  
62  	/**
63  	 * This method elivers a reference to the object that originated the Exception
64  	 * <br>
65  	 * @return ProtoN
66  	 */
67  	public ProtoN getSecond() {
68  		return Second;
69  	}
70  
71  }
