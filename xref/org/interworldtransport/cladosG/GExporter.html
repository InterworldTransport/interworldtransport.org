33  /**
34   * This is a non-constructable class meant as a collector of all export methods for cladosG objects
35   * They are collected here not because a single object doing data exports is particularly efficient, 
36   * (it is likely to be a bottleneck until everything is parallelized), but because the geometry
37   * objects in real physical models are likely to be numerous. The geometry does not need to take 
38   * up memory knowing how to export its contents to some format for every single piece of geometry.
39   * It is enough that one object does it or even that it is all handled by static methods.
40   */
41  public class GExporter {
42      /*
43  	 * Private constructor means this will only after get used for its class/static methods.
44  	 */
45  	private GExporter(){
46  		;
47  	}

55  	 * 		"name": "NamedPoint"<br>
56  	 * 		}<br>
57  	 * }

60  	 * @return String formatted as JSON containing information about the input
61  	 */
62  	public final static String toJSON(Foot pF) {
63  		StringBuilder rB = new StringBuilder();

70  	/**
71  	 * Export a Blade as a small JSON fragment. Object properties are represented as attributes.
72  	 * This is intended as an output format.
73  	 * <br><br>
74  	 * Example:<br>
75  	 * {<br>
76  	 * 	"Blade": {<br>
77  	 * 		"key": 81985529216486895, <br>
78  	 * 		"bitKey": "0b111111111111111", <br>
79  	 * 		"generators": ["E1","E2","E3","E4","E5","E6","E7","E8","E9","EA","EB","EC","ED","EE","EF"]<br>
80  	 * 		}<br>
81  	 * }
82  	 * <br><br>
83  	 * @param blade Blade to be exported as JSON
84  	 * @param pAsText boolean determines whether generators are exported in text or ordinal representation.
85  	 * @return String formatted as JSON containing information about the input
86  	*/
87  	public final static String toJSON(Blade blade, boolean pAsText) {
88  		StringBuilder rB = new StringBuilder();
89  		rB	.append("{\"blade\": {\"key\": ")
90  			.append(blade.key())
91  			.append(", \"bitKey\": \"0b");
92  		int pad = blade.maxGen - Integer.toBinaryString(blade.bitKey()).length();
93  		while (pad>0) {
94  			rB.append("0");
95  			pad--;
96  		}
97  		rB	.append(Integer.toBinaryString(blade.bitKey()))
98  			.append("\", \"generators\": [");
99  
100 		if(pAsText)	blade	.generatorStream()
101 							.forEachOrdered(g -> rB.append("\""+g.toString()+"\"").append(","));
102 		else		blade	.generatorStream()
103 							.forEachOrdered(g -> rB.append(g.ord).append(","));
104 
105 		if (blade.getGenerators().size() > 0)	rB.deleteCharAt(rB.length() - 1);
106 		rB	.append("]}}");
107 		return rB.toString();
108 	}

150 
151 		rB	.append("]}}");
152 		return rB.toString();
153 	}

157 	 * Export a Basis as a JSON fragment. Object properties are represented as attributes.
158 	 * This is intended as an output format.
159 	 * <br><br>
160 	 * Example:<br>
161 	 * {<br>
162 	 *	"basis": {<br>
163 	 *	"UUID": "6f264df4-0b45-4ac2-9679-3ff1d181a0dc", <br>
164 	 *	"Grades": {<br>
165 	 *		"count": 3,<br> 
166 	 *		"grade": [	{"rank": 0, "range": "0-0"}, <br>
167 	 *					{"rank": 1, "range": "1-2"}, <br>
168 	 *					{"rank": 2, "range": "3-3"}]<br>
169 	 *		}, <br>
170 	 *	"Blades": {<br>
171 	 *		"count": 4,<br> 
172 	 *		"blade": [ 	{"key": 0, "bitKey": "0b00", "generators": []}, <br>
173 	 *					{"key": 1, "bitKey": "0b01", "generators": ["E1"]}, <br>
174 	 *					{"key": 2, "bitKey": "0b10", "generators": ["E2"]}, <br>
175 	 *					{"key": 5, "bitKey": "0b11", "generators": ["E1","E2"]}]<br>
176 	 *		}<br>
177 	 *	}<br>
178 	 * }
179 	 * <br><br>
180 	 * @param basis to be exported as JSON
181 	 * @return String formatted as JSON containing information about the input
182 	 */
183 	public final static String toJSON(Basis basis) {
184 		StringBuilder rB = new StringBuilder();
185 
186 		rB	.append("{\"basis\": {\"UUID\": \"")
187 			.append(basis.uuid)
188 			.append("\", \"Grades\": {\"count\": ")
189 			.append(basis.getGradeCount());
190 			
191 		rB	.append(", \"grade\": [");
192 		for (int k = 0; k < basis.getGradeCount(); k++) {						//loop through grades constructing grade ranges.
193 			rB	.append("{\"rank\": ")
194 				.append(k)
195 				.append(", \"range\": \"");
196 			rB	.append(basis.getGrades().get(k))
197 				.append("-");
198 			if (k == basis.getGradeCount() - 1 )
199 					rB	.append(basis.getGrades().get(k));						//this is why the pscalar is handled separately
200 			else 	rB	.append(basis.getGrades().get(k + 1) - 1);				//there is is a next higher grade
201 			rB	.append("\"}, ");
202 			}
203 		rB	.deleteCharAt(rB.length()-1);										//lop the last character (removes trailing space)
204 		rB	.deleteCharAt(rB.length()-1);										//lop the last character (removes trailing comma)
205 		rB	.append("]}, ");
206 
207 		rB	.append("\"Blades\": {\"count\": ")
208 			.append(basis.getBladeCount());
209 		rB	.append(", \"blade\": [");
210 		basis.bladeStream().forEach(blade -> {
211 			StringBuffer workThis = new StringBuffer(GExporter.toJSON(blade, true));
212 			workThis.delete(0,9);
213 			workThis.deleteCharAt(workThis.length()-1);
214 			workThis.append(",");
215 			rB.append(workThis);
216 		});
217 		rB	.deleteCharAt(rB.length()-1);									//lop the last character (removes trailing comma)
218 		rB	.append("]}}}");
219 		return rB.toString();
220 	}

386 	/**
387 	 * This method produces a printable and parseable string that represents the
388 	 * Basis in a human readable form. return String
389 	 * <br>
390 	 * @param pB The Basis to export as XML
391 	 * @param indent String of 'tab' characters that help space the output correctly
392 	 *               visually. It's not actually necessary except for human
393 	 *               readability of the output.
394 	 * @return String
395 	 */
396 	public static String toXMLString(Basis pB, String indent) {
397 		if (indent == null)
398 			indent = "\t\t\t\t\t\t";
399 		StringBuilder rB = new StringBuilder(indent + "<Basis UUID=\"");
400 		rB	.append(pB.uuid)
401 			.append("\">\n");
402 		// ------------------------------------------------------------------
403 		rB.append(indent)
404 			.append("\t<Grades count=\"")
405 			.append(pB.getGradeCount() + "\">\n");
406 		for (int k = 0; k <= pB.getGradeCount() - 2; k++) // loop to get all but the highest grade
407 			rB	.append(indent)
408 				.append("\t\t<Grade rank=\"")
409 				.append(k)
410 				.append("\" range=\"")
411 				.append(pB.getGrades().get(k))
412 				.append("-")
413 				.append((pB.getGrades().get(k + 1) - 1))
414 				.append("\" />\n");
415 		// Handle last grade separate. There is no k+1 index for the largest grade
416 		rB	.append(indent)
417 			.append("\t\t<Grade rank=\"")
418 			.append((pB.getGradeCount() - 1))
419 			.append("\" range=\"")
420 			.append(pB.getGrades().get(pB.getGradeCount() - 1))
421 			.append("-")
422 			.append(pB.getGrades().get(pB.getGradeCount() - 1))
423 			.append("\" />\n");
424 		rB.append(indent)
425 			.append("\t</Grades>\n");
426 		// ------------------------------------------------------------------
427 		rB	.append(indent)
428 			.append("\t<Blades count=\"")
429 			.append(pB.getBladeCount())
430 			.append("\">\n");
431 		for (int k = 0; k < pB.bladeList.size(); k++) // Appending blades
432 			rB	.append(GExporter.toXMLString(pB.bladeList.get(k), indent + "\t\t"));
433 		rB	.append(indent)
434 			.append("\t</Blades>\n");
435 		// ------------------------------------------------------------------
436 		rB	.append(indent)
437 			.append("</Basis>\n");
438 		return rB.toString();
439 	}

442 	 * This method produces a printable and parseable string that represents the GProduct in a human readable form.
443 	 * NOTE the GProduct no longer exports the Basis on which it is built. It exports the Cayley table and signature.
444 	 * <br>
445 	 * @param pG A geometric product to be exported to XML
446 	 * @param indent A string to use for XML element intentation. Not required.
447 	 * @return String This is the XML string export of an object.
448 	 */
449 	public final static String toXMLString(GProduct pG, String indent) {
450 		if (indent == null)			indent = "\t\t\t\t\t";
451 		StringBuilder rB = new StringBuilder(indent + "<GProduct signature=\""+pG.signature()+"\">\n");
452 		//rB	.append(GExporter.toXMLString(pG.getBasis(), indent + "\t"));
453 		rB	.append(indent)
454 			.append("\t<CayleyTable rows=\"")
455 			.append(pG.getBladeCount())
456 			.append("\">\n");
457 		for (int k = 0; k < pG.getBladeCount(); k++) {		// Appending rows of the Cayley table
458 			rB	.append(indent)
459 				.append("\t\t<row id=\"") 
460 				.append(k)
461 				.append("\" cells=\"");
462 			for (int m = 0; m < pG.getBladeCount(); m++)
463 				rB	.append(pG.getResult(k, m))
464 					.append(",");
465 			rB.deleteCharAt(rB.length() - 1);
466 			rB.append("\" />\n");
467 		}
468 		rB.append(indent + "\t</CayleyTable>\n");
469 		rB.append(indent + "</GProduct>\n");
470 		return rB.toString();
471 	}

474 	 * This is an exporter of internal details to XML. It exists to bypass certain security concerns related to 
475 	 * Java serialization of objects.
476 	 * <br><br>
477 	 * @param pA     Algebra to be exported as XML data
478 	 * @param indent String of tab characters to assist with human readability of output.
479 	 * @return String formatted as XML containing information about the Algebra
480 	 */
481 	public final static String toXMLString(Algebra pA, String indent) {
482 		if (indent == null)		indent = "\t\t\t\t";
483 		StringBuilder rB = new StringBuilder(indent+"<Algebra name=\"");
484 		rB	.append(pA.getAName())
485 			.append("\" UUID=\"")
486 			.append(pA.uuid)
487 			.append("\" >\n");
488
