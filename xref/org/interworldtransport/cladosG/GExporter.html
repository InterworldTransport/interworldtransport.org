33  /**
34   * This is a non-constructable class meant as a collector of all export methods for cladosG objects
35   * They are collected here not because a single object doing data exports is particularly efficient, 
36   * (it is likely to be a bottleneck until everything is parallelized), but because the geometry
37   * objects in real physical models are likely to be numerous. The geometry does not need to take 
38   * up memory knowing how to export its contents to some format for every single piece of geometry.
39   * It is enough that one object does it or even that it is all handled by static methods.
40   */
41  public class GExporter {
42      /*
43  	 * Private constructor means this will only after get used for its class/static methods.
44  	 */
45  	private GExporter(){
46  		;
47  	}

49      /**
50  	 * Export a Foot as a small JSON fragment. Object properties are represented as attributes.
51  	 * This is intended as an output format.
52  	 * <br><br>
53  	 * {"Foot": {"name": "NamedPoint"}}
54  	 * <br><br>
55  	 * @param pF Foot to be exported as JSON
56  	 * @return String formatted as JSON containing information about the Foot
57  	 */
58  	public final static String toJSON(Foot pF) {
59  		StringBuilder rB = new StringBuilder();
60  		rB	.append("{\"Foot\": {\"name\": \"")
61  			.append(pF.getName())
62  			.append("\"}}\n");
63  		return rB.toString();
64  	}

75  		if (indent == null)		indent = "\t\t";
76  		StringBuilder rB = new StringBuilder(indent + "<Foot name=\"");

82  	/**
83  	 * This method produces a printable and parseable string that represents the
84  	 * Blade in a human readable form.
85  	 * <br>
86  	 * This variation uses a Generator's ordinal to name it in the generator list.
87  	 * <br>
88  	 * @param blade  The Blade to be exported to XML.
89  	 * @param indent String of 'tab' characters that help space the output correctly
90  	 *               visually. It's not actually necessary except for human
91  	 *               readability of the output.
92  	 * @return String The XML formated String representing the Blade.
93  	 */
94  	public final static String toXMLOrdString(Blade blade, String indent) {
95  		if (indent == null)		indent = "\t\t\t\t\t\t\t\t";
96  		StringBuilder rB = new StringBuilder(indent);
97  		rB	.append("<Blade key=\"")
98  			.append(blade.key())
99  			.append("\" bitKey=\"0b");
100 
101 		int pad = blade.maxGen - Integer.toBinaryString(blade.bitKey()).length();
102 		while (pad>0) {
103 			rB.append("0");
104 			pad--;
105 		}
106 		rB	.append(Integer.toBinaryString(blade.bitKey()))
107 			.append("\" generators=\"");
108 
109 		blade	.generatorStream()
110 				.forEachOrdered(gen -> rB.append(gen.ord).append(","));
111 
112 		if (blade.rank() > 0)
113 			rB.deleteCharAt(rB.length() - 1);
114 		rB	.append("\" />\n");
115 		return rB.toString();
116 	}

174 	/**
175 	 * This method produces a printable and parseable string that represents the
176 	 * Basis in a human readable form. return String
177 	 * <br>
178 	 * @param pB The Basis to export as XML
179 	 * @param indent String of 'tab' characters that help space the output correctly
180 	 *               visually. It's not actually necessary except for human
181 	 *               readability of the output.
182 	 * @return String
183 	 */
184 	public static String toXMLString(Basis pB, String indent) {
185 		if (indent == null)
186 			indent = "\t\t\t\t\t\t";
187 		StringBuilder rB = new StringBuilder(indent + "<Basis UUID=\"");
188 		rB	.append(pB.uuid)
189 			.append("\">\n");
190 		// ------------------------------------------------------------------
191 		rB.append(indent)
192 			.append("\t<Grades count=\"")
193 			.append(pB.getGradeCount() + "\">\n");
194 		for (int k = 0; k <= pB.getGradeCount() - 2; k++) // loop to get all but the highest grade
195 			rB	.append(indent)
196 				.append("\t\t<Grade rank=\"")
197 				.append(k)
198 				.append("\" range=\"")
199 				.append(pB.getGrades().get(k))
200 				.append("-")
201 				.append((pB.getGrades().get(k + 1) - 1))
202 				.append("\" />\n");
203 		// Handle last grade separate. There is no k+1 index for the largest grade
204 		rB	.append(indent)
205 			.append("\t\t<Grade rank=\"")
206 			.append((pB.getGradeCount() - 1))
207 			.append("\" range=\"")
208 			.append(pB.getGrades().get(pB.getGradeCount() - 1))
209 			.append("-")
210 			.append(pB.getGrades().get(pB.getGradeCount() - 1))
211 			.append("\" />\n");
212 		rB.append(indent)
213 			.append("\t</Grades>\n");
214 		// ------------------------------------------------------------------
215 		rB	.append(indent)
216 			.append("\t<Blades count=\"")
217 			.append(pB.getBladeCount())
218 			.append("\">\n");
219 		for (int k = 0; k < pB.bladeList.size(); k++) // Appending blades
220 			rB	.append(GExporter.toXMLString(pB.bladeList.get(k), indent + "\t\t"));
221 		rB	.append(indent)
222 			.append("\t</Blades>\n");
223 		// ------------------------------------------------------------------
224 		rB	.append(indent)
225 			.append("</Basis>\n");
226 		return rB.toString();
227 	}

229 	/**
230 	 * This method produces a printable and parseable string that represents the
231 	 * Basis in a human readable form.
232 	 * <br>
233 	 * @param pG A geometric product to be exported to XML
234 	 * @param indent A string to use for XML element intentation. Not required.
235 	 * @return String This is the XML string export of an object.
236 	 */
237 	public final static String toXMLString(GProduct pG, String indent) {
238 		if (indent == null)			indent = "\t\t\t\t\t";
239 		StringBuilder rB = new StringBuilder(indent + "<GProduct signature=\""+pG.signature()+"\">\n");
240 		rB	.append(GExporter.toXMLString(pG.getBasis(), indent + "\t"));
241 		rB	.append(indent)
242 			.append("\t<CayleyTable rows=\"")
243 			.append(pG.getBladeCount())
244 			.append("\">\n");
245 		for (int k = 0; k < pG.getBladeCount(); k++) {		// Appending rows of the Cayley table
246 			rB	.append(indent)
247 				.append("\t\t<row id=\"")
248 				.append(k)
249 				.append("\" cells=\"");
250 			for (int m = 0; m < pG.getBladeCount(); m++)
251 				rB	.append(pG.getResult(k, m))
252 					.append(",");
253 			rB.deleteCharAt(rB.length() - 1);
254 			rB.append("\" />\n");
255 		}
256 		rB.append(indent + "\t</CayleyTable>\n");
257 		rB.append(indent + "</GProduct>\n");
258 		return rB.toString();
259 	}

263 	 * Java serialization of objects.
264 	 * <br><br>
265 	 * @param pA     Algebra to be exported as XML data
266 	 * @param indent String of tab characters to assist with human readability of output.
267 	 * @return String formatted as XML containing information about the Algebra
268 	 */
269 	public final static String toXMLString(Algebra pA, String indent) {
270 		if (indent == null)		indent = "\t\t\t\t";
271 		StringBuilder rB = new StringBuilder(indent+"<Algebra name=\"");
272 		rB	.append(pA.getAName())
273 			.append("\" UUID=\"")
274 			.append(pA.uuid)
275 			.append("\" >\n");
276 		
277 		rB	.append(GExporter.toXMLString(pA.getFoot(), indent + "\t"));			//Algebra owns a reference to a Foot
278 		rB	.append(GExporter.toXMLString(pA.getGP(), indent + "\t"));				//Algebra owns a reference to a GP
279 		
280 		rB	.append(indent)
281 			.append("</Algebra>\n");
282 		return rB.toString();

286 	 * This is a short exporter of internal details to XML. It exists to bypass certain security concerns related to Java serialization.
287 	 * <br><br>
288 	 * @param pS The Scale oject to be output as XML
289 	 * @param indent String of 'tab' characters to get spacing right for human readable XML output.
290 	 * @return String formatted as XML containing information about the Scale
291 	 */
292 	public final static String toXMLString(Scale<?> pS, String indent) {
293 		StringBuilder rB = new StringBuilder(indent);		
294 		rB	.append("<Scales mode=\""+pS.getMode()+"\" ");
295 		rB	.append("pans=\""+pS.map.size()+"\" ")
296 			.append("cardinal="+pS.getCardinal().getUnit()+"\">\n");
297 
298 		pS.getBasis().bladeStream().forEach(blade -> {
299 			rB	.append(indent+"\t")
300 				.append("<Pair bitKey=\"0b");
301 			int pad = blade.maxGen - Integer.toBinaryString(blade.bitKey()).length();
302 			while (pad>0) {
303 				rB.append("0");
304 				pad--;
305 			}
306 			rB	.append(Integer.toBinaryString(blade.bitKey()));
307 			//rB	.append("\" cardinal=\""+pS.map.get(blade).getCardinalString());
308 			rB	.append("\" ");
309 			
310 			switch (pS.getMode()){
311 				case REALF:
312 					rB	.append("realvalue=\""+((RealF)pS.map.get(blade)).getReal());
313 					break;
314 				case REALD:
315 					rB	.append("realvalue=\""+((RealD)pS.map.get(blade)).getReal());
316 					break;
317 				case COMPLEXF:
318 					rB	.append("realvalue=\""+((ComplexF)pS.map.get(blade)).getReal())
319 						.append("\" imgvalue=\""+((ComplexF)pS.map.get(blade)).getImg());
320 					break;
321 				case COMPLEXD:
322 					rB	.append("realvalue=\""+((ComplexD)pS.map.get(blade)).getReal())
323 						.append("\" imgvalue=\""+((ComplexD)pS.map.get(blade)).getImg());
324 					break;
325 				default:
326 					break;
327 				
328 			}
329 			rB	.append("\" />\n");
330 		});
331 		rB	.append(indent)
332 			.append("</Scales>\n");
333 		return rB.toString();
334 	}

337 	 * This is an exporter of internal details to XML. It exists to bypass certain security concerns related to Java serialization.
338 	 * <br><br>
339 	 * @param pS The Scale oject to be output as XML
340 	 * @param indent String of 'tab' characters to get spacing right for human readable XML output.
341 	 * @return String formatted as XML containing information about the Scale
342 	 */
343 	public final static String toXMLFullString(Scale<?> pS, String indent) {
344 		StringBuilder rB = new StringBuilder(indent);		
345 		rB	.append("<Scales mode=\""+pS.getMode()+"\" pans=\"")
346 			.append(pS.map.size())
347 			.append("\">\n");
348 
349 		pS.getBasis().bladeStream().forEach(blade -> {
350 			rB	.append(indent)
351 				.append("\t\t\t<Pair>\n");
352 			rB	.append(indent)
353 				.append(GExporter.toXMLString(blade, "\t\t\t\t"));
354 			switch (pS.getMode()){
355 				case COMPLEXD -> {rB.append(indent + "\t\t\t\t").append(ComplexD.toXMLString((ComplexD) pS.map.get(blade))).append("\n");}
356 				case COMPLEXF -> {rB.append(indent + "\t\t\t\t").append(ComplexF.toXMLString((ComplexF) pS.map.get(blade))).append("\n");}
357 				case REALD -> 	{rB.append(indent + "\t\t\t\t").append(RealD.toXMLString((RealD) pS.map.get(blade))).append("\n");}
358 				case REALF -> 	{rB.append(indent + "\t\t\t\t").append(RealF.toXMLString((RealF) pS.map.get(blade))).append("\n");}
359 				default -> 		{rB.append(indent + "\t\t\t\t").append(ProtoN.toXMLString(pS.map.get(blade))).append("\n");}
360 			}	
361 			rB	.append(indent).append("\t\t\t</Pair>\n");
362 		});
363 		rB	.append(indent).append("\t\t</Scales>\n");
364 		return rB.toString();
365 	}

421 	 * @param indent String of tab characters to assist with human readability.

423 	 */

425 		if (indent == null)			indent = "\t";
426 		StringBuilder rB = new StringBuilder(indent+"<Nyad name=\"");
427 		rB	.append(pN.getName())
428 			.append("\" order=\"")
429 			.append(pN.arity())

432 			.append("\" >\n");

435 	
436 		rB	.append(indent)
437 			.append("\t<AlgebraList>\n");
438 		for (Algebra point : pN.algebraList)
439 			rB	.append(indent)

442 				.append("</AlgebraName>\n");
443 		rB	.append(indent)
444 			.append("\t</AlgebraList>\n");
445 		
446 		rB	.append(indent)
447 			.append("\t<MonadList>\n");
448 		for (Monad tSpot : pN.monadList)
449 			rB.append(GExporter.toXMLFullString(tSpot, indent + "\t\t"));
450 		rB	.append(indent)
451 			.append("\t</MonadList>\n");
452
