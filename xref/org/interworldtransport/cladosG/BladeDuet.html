25  package org.interworldtransport.cladosG;
26  
27  import java.util.ArrayList;
28  import java.util.Collections;
29  
30  /**
31   * This class just acts as a bucket two blades when they are combined and
32   * reduced in product result discovery. In Clados v1 the methods were all buried
33   * in the Basis and GProduct classes. They are surfaced here in BladeDuet in
34   * order to support parallelization of product table generation.
35   * <br><br>
36   * BladeDuet makes use of streams, but intentionally avoids parallelizing
37   * computations internally. Most of what each one does must be done in sequence.
38   * <br><br>
39   * @version 2.0
40   * @author Dr Alfred W Differ
41   */
42  public final class BladeDuet {
43  
44  	/**
45  	 * This method reduces pairs of directions in what is ALMOST a sorted bladeDuet
46  	 * list. It's actually two buckets of sorted generators that upon duplication
47  	 * removal MIGHT be sorted. If not, we can jump straight to the sorted order
48  	 * simply by inserting the generators in an EnumSet which happens to be their
49  	 * destination in a Blade anyway. What we don't know immeidately is how many
50  	 * transpositions are necessary to reach that sort order. That's what this
51  	 * method does after removing generator duplicates.
52  	 * <br>
53  	 * The offered numeric signature is used for the reduction to handle sign flips.
54  	 * Generators with a positive square appear as a one (1) while those with
55  	 * negative squares appear as negative one (-1).
56  	 * <br>
57  	 * NOTE that the numeric signature representation is a departure with prior use
58  	 * in Clados where zero(0) implied no sign flip and one(1) implied sign flip for
59  	 * negative squared generator. Prior practice used to add up the sign flips and
60  	 * then look at what was left modulo 2. Ideally, what we want is a 'signed bit'
61  	 * sized data element to track signs.
62  	 * <br>
63  	 * Exception cases NOT checked because this is for CladosG internal use. The
64  	 * method itself is public, but it's really for internal use.
65  	 * <br>
66  	 * @param pB1 Blade appearing on the left/row of a multiplication operation
67  	 * @param pB2 Blade appearing on the right/column of a multiplication operation
68  	 * @param sig signature array to use to reduce duplicate generators
69  	 * @return A fully reduced blade
70  	 */
71  	public static final Blade simplify(Blade pB1, Blade pB2, byte[] sig) {
72  		BladeDuet tBD = new BladeDuet(pB1, pB2);
73  		return tBD.simplify(sig);
74  	}
75  	/**
76  	 * These are the bitKey's of the Blades inserted.
77  	 * They get used to help with the simplify algorithm.
78  	 */
79  	private int bitKeyLeft, bitKeyRight = 0;
80  	
81  	/**
82  	 * This holds the combined list of generators from each blade.
83  	 * Left blade goes first, then right
84  	 */
85  	private ArrayList<Generator> bladeDuet;
86  
87  	/**
88  	 * This byte holds the sign of the blade to which the list of 
89  	 * generators simplifies in the end. A +1 sign means a blade that 
90  	 * is an even permutation of the remaining generators. A -1 sign 
91  	 * means an odd permutation.
92  	 * <br>
93  	 * Calculating the sign ALSO gets the algebra's signature involved,
94  	 * so this DOES involve the metric.
95  	 */
96  	private byte sign = 1;
97  
98  	/**
99  	 * This is where the hint is kept for the largest possible blade in the 
100 	 * basis to which the resulting blade actually belongs. It is capped by the 
101 	 * maximum grade from one of the blades... which really should have the 
102 	 * same maximum grade.
103 	 */
104 	private final byte maxGen;
105 
106 	/**
107 	 * This is a re-use constructor that builds this as a juxtaposition of the two
108 	 * offered blades.
109 	 * <br>
110 	 * @param pB1 A Blade to re-use on the left.
111 	 * @param pB2 A Blade to re-use on the right.
112 	 */
113 	public BladeDuet(Blade pB1, Blade pB2) {
114 		assert (pB1.maxGenerator() == pB2.maxGenerator());
115 		maxGen = pB1.maxGenerator();
116 		// assert (maxGen > 0);
117 		bladeDuet = new ArrayList<>(2 * maxGen);
118 		pB1.generatorStream().forEachOrdered(g -> bladeDuet.add(g));
119 		sign = pB1.sign();
120 		bitKeyLeft = pB1.bitKey();
121 		pB2.generatorStream().forEachOrdered(g -> bladeDuet.add(g));
122 		sign *= pB2.sign();
123 		bitKeyRight = pB2.bitKey();
124 	}
125 
126 	/**
127 	 * This method reduces pairs of directions in what is ALMOST a sorted bladeDuet
128 	 * list. It's actually two buckets of sorted generators that upon duplication
129 	 * removal MIGHT be sorted. If not, we can jump straight to the sorted order
130 	 * simply by inserting the generators in an EnumSet which happens to be their
131 	 * destination in a Blade anyway. What we don't know immediately is how many
132 	 * transpositions are necessary to reach that sort order. That's what this
133 	 * method does after removing generator duplicates.
134 	 * <br>
135 	 * The offered numeric signature is used for the reduction to handle sign flips.
136 	 * Generators with a positive square appear as a one (1) while those with
137 	 * negative squares appear as negative one (-1).
138 	 * <br>
139 	 * NOTE that the numeric signature representation is a departure with prior use
140 	 * in Clados where zero(0) implied no sign flip and one(1) implied sign flip for
141 	 * negative squared generator. Prior practice used to add up the sign flips and
142 	 * then look at what was left modulo 2. Ideally, what we want is a 'signed bit'
143 	 * sized data element to track signs.
144 	 * <br>
145 	 * Exception cases NOT checked because this is for CladosG internal use.
146 	 * <br>
147 	 * @param pSig An array of unboxed short integers that signifies when sign flips
148 	 *             occur as generator pairs are removed from the internal dual list.
149 	 * @return Blade [supporting stream approach]
150 	 */
151 	protected Blade simplify(byte[] pSig) {
152 		int andKey = bitKeyLeft & bitKeyRight;
153 		byte gen = 1;										//start with lowest generator
154 		while (andKey > 0) {								//while any duplicate generators present
155 			if (Integer.lowestOneBit(andKey) == 1) {		//andKey is odd => low bit points at duplicate generator
156 				Generator eq = Generator.get(gen);			//find generator for that lowest bit
157 				sign *= (Integer.lowestOneBit(bladeDuet.lastIndexOf(eq) ^ bladeDuet.indexOf(eq)) == 1) ? (byte) 1 : (byte) -1;
158 															//lastIndexOf = right-most. indexOf = left-most.
159 															//We won't be in this section unless there are exactly two.
160 															//This permutes indexes getting both generators next to each other.
161 				sign *= pSig[gen - 1];
162 				bladeDuet.removeAll(Collections.singleton(eq));
163 			}
164 			gen++;											//move up to the next generator to test
165 			andKey = andKey >>> 1;							//shift andKey right dropping lowest bit
166 		}
167 		Blade returnIt = Blade.createBlade(maxGen); 		//A scalar blade with room to expand.
168 		if (sign == 0)							
169 			return returnIt.setSign((byte) 0);				//In the degenerate case, we are done!
170 		
171 		bladeDuet.stream().forEach(g -> returnIt.add(g));	//Load remaining generators IF NO paired generator was degenerate
172 															//returnIt has the correct generators, but might have the wrong sign
173 		andKey = bitKeyLeft & bitKeyRight;
174 															// if either residue key vanishes, the bladeDuet is already in SORT order.
175 		if ((bitKeyLeft - andKey) != 0 & (bitKeyRight - andKey) != 0) {
176 			ArrayList<Generator> pB = new ArrayList<>(returnIt.getGenerators());
177 			for (Generator pG : pB) { 						// Exploiting the KNOWN correct order.
178 				int found = bladeDuet.indexOf(pG);
179 				int refer = pB.indexOf(pG);
180 
181 				if (found != refer) {
182 					sign *= (byte) -1;
183 					Collections.swap(bladeDuet, found, refer);
184 				}
185 			}
186 		}
187 		return returnIt.setSign(sign);
188 	}
189 
190 	/**
191 	 * This method produces a printable and parseable string that represents the
192 	 * BladeDuet in a human readable form. This is likely ONLY useful during debug
193 	 * efforts.
194 	 * <br>
195 	 * This variation uses a Generator's name in the generator list.
196 	 * <br>
197 	 * @param pBD The blade duet to export as XML.
198 	 * @return String The XML formated String representing the BladeDuet.
199 	 */
200 	public final static String toXMLString(BladeDuet pBD) {
201 		StringBuilder rB = new StringBuilder();
202 		rB.append("<BladeDuet sign=\"").append(pBD.sign).append("\" maxGrade=\"").append(pBD.maxGen).append("\" generators=\"");
203 		pBD.bladeDuet.stream().forEachOrdered(g -> rB.append(g.toString() + ","));
204 		if (pBD.bladeDuet.size() > 0)
205 			rB.deleteCharAt(rB.length() - 1);
206 		rB.append("\" />\n");
207 		return rB.toString();
208 	}
209 }
