27  import java.util.ArrayList;
28  import java.util.Collections;

30  /**
31   * This class serves two highly related purposes.<br>
32   * 1) It supports blade discovery in blade multiplication by acting as a bucket for their generators that can
33   * be reduced using a signature.<br>
34   * 2) It supports blade discovery in blade complement operations and compute the resulting blade's sign so it
35   * correctly multiplies to the pscalar in the basis that contains the input blade.
36   * <br><br>
37   * In Clados v1 the multiplication methods were all buried in the Basis and GProduct classes. BaldeDuet surfaces
38   * them in order to support parallelization of product table generation. As a side benefit, the complement 
39   * operation supportual Dual() in Clados v2 also benefits from parallelization.
40   * <br><br>
41   * NOTE: BladeDuet makes use of streams, but intentionally avoids internal parallelization of computations. Most of 
42   * what must be done must be in sequence. It is a calling objects that may parallelize operations relying on BladeDuet.
43   * <br><br>
44   * @version 2.0
45   * @author Dr Alfred W Differ
46   */
47  public final class BladeDuet implements Comparable<BladeDuet>{
48  
49  	/**
50  	 * This method takes the set of generators in the offered blade and returns a blade with a set of generators that 
51  	 * complements the first in such a way that left multiplication of the initial blade by the complement blade produces
52  	 * the +pscalar associated with the initial blade.
53  	 * <br><br>
54  	 * NOTE: Because generators might be degenerate, it is not possible to extract commute signs from the Cayley table
55  	 * when computing complement blades. If not for degeneracy, complement would involve a table lookup.
56  	 * <br><br>
57  	 * @param pB1 Blade to be complemented with respect to it's basis pscalar blade
58  	 * @param sig byte[] signature array to use when reducing duplicate generators
59  	 * @param pElliptic True if degenerate pairs pretend to square to +1, False if they pretend to square to -1
60  	 * @return Blade complement of pB1 with sign set for left multiply returning +pscalar
61  	 */
62  	public static final Blade complementLeft(Blade pB1, byte[] sig, boolean pElliptic) {
63  		BladeDuet tBD = new BladeDuet(pB1, Blade.createPScalarBlade(Generator.get(pB1.maxGenerator()))) ;
64  		if (pElliptic) 	return tBD.simplifyForEllipticDual(sig);
65  		else			return tBD.simplifyForHyperbolicDual(sig);
66  	}

68  	/**
69  	 * This method takes the set of generators in the offered blade and returns a blade with a set of generators that 
70  	 * complements the first in such a way that right multiplication of the initial blade by the complement blade produces
71  	 * the +pscalar associated with the initial blade.
72  	 * <br><br>
73  	 * NOTE: Because generators might be degenerate, it is not possible to extract commute signs from the Cayley table
74  	 * when computing complement blades. If not for degeneracy, complement would involve a table lookup.
75  	 * <br><br>
76  	 * @param pB1 Blade to be complemented with respect to it's basis pscalar blade
77  	 * @param sig byte[] signature array to use when reducing duplicate generators
78  	 * @param pElliptic True if degenerate pairs pretend to square to +1, False if they pretend to square to -1
79  	 * @return Blade complement of pB1 with sign set for left multiply returning +pscalar
80  	 */
81  	public static final Blade complementRight(Blade pB1, byte[] sig, boolean pElliptic) {
82  		BladeDuet tBD = new BladeDuet(Blade.createPScalarBlade(Generator.get(pB1.maxGenerator())), pB1) ;
83  		if (pElliptic) 	return tBD.simplifyForEllipticDual(sig);
84  		else			return tBD.simplifyForHyperbolicDual(sig);
85  	}

87  	/**
88  	 * This method reduces pairs of directions in what is ALMOST a sorted bladeDuet list. It's actually two buckets of 
89  	 * sorted generators that upon duplication removal MIGHT be sorted. If not, we can jump straight to the sorted order
90  	 * simply by inserting the generators in an EnumSet which happens to be their destination in a Blade anyway. What we 
91  	 * don't know immediately is how many transpositions are necessary to reach that sort order. That's what this method 
92  	 * does after removing generator duplicates.
93  	 * <br><br>
94  	 * The offered numeric signature is used for the reduction to handle sign flips. Generators with a positive square 
95  	 * appear as a one (1) while those with negative squares appear as negative one (-1).
96  	 * <br><br>
97  	 * NOTE that the numeric signature representation is a departure with prior use in Clados where zero(0) implied no sign 
98  	 * flip and one(1) implied sign flip for negative squared generator. Prior practice used to add up the sign flips and 
99  	 * then look at what was left modulo 2. Ideally, what we want is a 'signed bit' sized data element to track signs.
100 	 * <br><br>
101 	 * Exception cases NOT checked because this is for CladosG internal use. The method itself is public, but it's 
102 	 * really for internal use.
103 	 * <br><br>
104 	 * @param pB1 Blade appearing on the left/row of a multiplication operation
105 	 * @param pB2 Blade appearing on the right/column of a multiplication operation
106 	 * @param sig signature array to use to reduce duplicate generators
107 	 * @return A fully reduced blade
108 	 */
109 	public static final Blade simplify(Blade pB1, Blade pB2, byte[] sig) {
110 		BladeDuet tBD = new BladeDuet(pB1, pB2);
111 		return tBD.simplify(sig);
112 	}

114 	/**
115 	 * This is the left blade in the ordered pair that BladeDuet really is.
116 	 */
117 	protected final Blade bladeLeft;
118 
119 	/**
120 	 * This is the right blade in the ordered pair that BladeDuet really is.
121 	 */
122 	protected final Blade bladeRight;
123 	
124 	/**
125 	 * This holds the combined list of generators from each blade.
126 	 * Left blade goes first, then right
127 	 */
128 	protected ArrayList<Generator> bladeDuet;
129 
130 	/**
131 	 * This byte holds the sign of the blade to which the list of 
132 	 * generators simplifies in the end. A +1 sign means a blade that 
133 	 * is an even permutation of the remaining generators. A -1 sign 
134 	 * means an odd permutation.
135 	 * <br>
136 	 * Calculating the sign ALSO gets the algebra's signature involved,
137 	 * so this DOES involve the metric.
138 	 */
139 	protected byte sign = 0;
140 
141 	/**
142 	 * This is where the hint is kept for the largest possible blade in the basis to which the resulting blade actually belongs. 
143 	 * It is capped by the maximum grade from one of the blades... which really should have the same maximum grade.
144 	 */
145 	protected Generator maxGen;
146 
147 	/**
148 	 * This constructor builds what is essentially an ordered pair of blades and then sets up the simplification steps
149 	 * that usually follow.
150 	 * <br><br>
151 	 * @param pB1 A Blade to re-use on the left.
152 	 * @param pB2 A Blade to re-use on the right.
153 	 */
154 	public BladeDuet(Blade pB1, Blade pB2) {
155 		bladeLeft = pB1;
156 		bladeRight = pB2;
157 		bladeDuet = new ArrayList<Generator>(30);
158 	}
159 
160 	/**
161 	 * The method needs to be as efficient as possible because it is used in comparators. It determines the 
162 	 * sort order of ordered pairs of blades, thus the streaming order for key/value pairs.
163 	 * <br><br>
164 	 * The way this works is simple. If the left blades compare to +/-1, return that. If they compare to ZERO,
165 	 * return the comparison of the right blades. This ensures the sort order in maps containing BladeDuets
166 	 * will sort on the left blade first and then the right blade. Remember that blade comparisons use their
167 	 * blade keys, so the net effect of this method for sorting BladeDuet will place them in the same order
168 	 * as the blades appear in a Basis. 
169 	 * <br><br>
170 	 * @param pIn BladeDuet to be compared to this one
171 	 * @return int in {-1, 0, 1} depending on the blade comparisons in the duet.
172 	 */
173 	@Override
174 	public int compareTo(BladeDuet pIn) {
175 		int testThis = this.bladeLeft.compareTo(pIn.bladeLeft);
176 		if (testThis == 0)		return (this.bladeRight.compareTo(pIn.bladeRight));
177 		else 					return testThis;
178 	}

180 	/**
181 	 * Overridden Equals method from Object.
182 	 * This ensures reference equality is the standard. They must literally refer to the same blade objects to be equal.
183 	 * @return boolean check for reference equality
184 	 */
185 	@Override
186 	public boolean equals(Object obj) {
187 		if (obj == null)						return false;
188 		if (! (obj instanceof BladeDuet)) 		return false;
189 
190 		return 		this.bladeLeft  == ((BladeDuet) obj).bladeLeft 
191 				& 	this.bladeRight == ((BladeDuet) obj).bladeRight;
192 	}

194 	/**
195 	 * The details in this method were originally in the Constructor, but have been moved to this private method which gets called 
196 	 * by any other method that ACTUALLY needs the work to be done. The contents of private data elements don't need to be preserved
197 	 * long most of the time, so once the Cayley Table is constructed, the primary use for BladeDuet is as an ordered pair of blades
198 	 * in maps. That use in maps does not require the work done in this method.
199 	 * <br><br>
200 	 * There is a secondary use that does wind up calling here that involves blade complements. The complement methods rely on the 
201 	 * simplify methods, so initialization of the bladeDuet is handled properly.
202 	 */
203 	private void setup() {																						//Only relevant when blades are multiplied
204 		assert (bladeLeft.maxGenerator() == bladeRight.maxGenerator());											//Don't simplify mismatched blades basically
205 		sign = (byte) (bladeLeft.sign() * bladeRight.sign());													
206 		maxGen = Generator.get((byte) Math.max(bladeLeft.maxGenerator(), bladeRight.maxGenerator()));
207 
208 		bladeDuet = new ArrayList<Generator>(2*CladosConstant.MAXGRADE);
209 		bladeLeft.generatorStream().forEachOrdered(g -> bladeDuet.add(g));
210 		bladeRight.generatorStream().forEachOrdered(g -> bladeDuet.add(g));
211 		bladeDuet.trimToSize();
212 	}

214 	/**
215 	 * This method reduces generator pairs in what is an ALMOST sorted bladeDuet list. It is two buckets of sorted generators 
216 	 * that upon pair removal WILL be fully sorted because one of the buckets is a pscalar blade. Every generator removed
217 	 * from the input blade is paired with one in the pscalar blade, so removal of pairs leaves a sorted list ensuring the
218 	 * second half of the 'simplify' algorithm isn't necessary here. The first half eliminates generator pairs and computes
219 	 * transposition counts, so that is all this method does
220 	 * <br><br>
221 	 * The offered metric signature resolves what sign a generator pair produces with one twist. For degenerate generators, 
222 	 * this method treats them as if they squared to +1.
223 	 * <br><br>
224 	 * @param pSig byte[] signature array to use when reducing duplicate generators
225 	 * @return Blade complement of the non-pscalar blade used to initialize the BladeDuet.
226 	 */
227 	private Blade simplifyForEllipticDual(byte[] pSig) {
228 		setup();
229 		int andKey = bladeLeft.bitKey() & bladeRight.bitKey();
230 		byte gen = 1;										//start with lowest generator
231 		while (andKey > 0) {								//while any duplicate generators present
232 			if (Integer.lowestOneBit(andKey) == 1) {		//andKey is odd => low bit names duplicate generator. Action required.
233 				Generator eq = Generator.get(gen);			//find generator for that lowest bit
234 				sign *= (Integer.lowestOneBit(bladeDuet.lastIndexOf(eq) ^ bladeDuet.indexOf(eq)) == 1) ? (byte) 1 : (byte) -1;
235 															//lastIndexOf = right-most. indexOf = left-most.
236 															//We won't be in this section unless there are exactly two.
237 															//This 'permutes' generators without moving them.
238 				sign *= (pSig[gen - 1] == 0) ? 1 : pSig[gen - 1];	//IF SIGNATURE of eq is 0, pretend it is +1. 
239 																	//Otherwise use correct signature.
240 				bladeDuet.removeAll(Collections.singleton(eq));
241 			}
242 			gen++;											//move up to the next generator to test
243 			andKey = andKey >>> 1;							//shift andKey right dropping lowest bit
244 		}
245 		Blade returnIt = Blade.createBlade(maxGen); 		//A scalar blade with room to expand.
246 		bladeDuet.stream().forEach(g -> returnIt.add(g));	//Load remaining generators 
247 															//returnIt has the correct generators AND sign.
248 		return returnIt.setSign(sign);
249 	}

263 	 */
264 	private Blade simplifyForHyperbolicDual(byte[] pSig) {
265 		setup();
266 		int andKey = bladeLeft.bitKey() & bladeRight.bitKey();

268 		while (andKey > 0) {								//while any duplicate generators present

273 															//We won't be in this section unless there are exactly two.

288 	/**
289 	 * This method reduces pairs of directions in what is ALMOST a sorted bladeDuet list. It's actually two buckets of 
290 	 * sorted generators that upon duplication removal MIGHT be sorted. If not, we can jump straight to the sorted order
291 	 * simply by inserting the generators in an EnumSet which happens to be their destination in a Blade anyway. What we 
292 	 * don't know immediately is how many transpositions are necessary to reach that sort order. That's what this method 
293 	 * does after removing generator duplicates.
294 	 * <br>
295 	 * The offered numeric signature is used for the reduction to handle sign flips. Generators with a positive square 
296 	 * appear as a one (1) while those with negative squares appear as negative one (-1).
297 	 * <br>
298 	 * NOTE that the numeric signature representation is a departure with prior use in Clados where zero(0) implied no sign 
299 	 * flip and one(1) implied sign flip for  negative squared generator. Prior practice used to add up the sign flips and 
300 	 * then look at what was left modulo 2. Ideally, what we want is a 'signed bit' sized data element to track signs.
301 	 * <br>
302 	 * Exception cases NOT checked because this is for CladosG internal use.
303 	 * <br>
304 	 * @param pSig 	An array of unboxed short integers that signifies when sign flips occur as generator pairs are removed 
305 	 * 				from the internal dual list.
306 	 * @return Blade [supporting stream approach]
307 	 */
308 	private Blade simplify(byte[] pSig) {
309 		setup();
310 		int andKey = bladeLeft.bitKey() & bladeRight.bitKey();
311 		byte gen = 1;										//start with lowest generator
312 		while (andKey > 0) {								//while any duplicate generators present
313 			if (Integer.lowestOneBit(andKey) == 1) {		//andKey is odd => low bit points at duplicate generator
314 				Generator eq = Generator.get(gen);			//find generator for that lowest bit
315 				sign *= (Integer.lowestOneBit(bladeDuet.lastIndexOf(eq) ^ bladeDuet.indexOf(eq)) == 1) ? (byte) 1 : (byte) -1;
316 															//lastIndexOf = right-most. indexOf = left-most.
317 															//We won't be in this section unless there are exactly two.
318 															//This 'permutes' generators without moving them.
319 				sign *= pSig[gen - 1];						//Adjust again for signature of generators.
320 				bladeDuet.removeAll(Collections.singleton(eq));
321 			}
322 			if (sign == 0) 	break;							//Degenerate eq found. We are done since result is a scalar ZERO.
323 			gen++;											//move up to the next generator to test
324 			andKey = andKey >>> 1;							//shift andKey right dropping lowest bit
325 		}
326 		Blade returnIt = Blade.createBlade(maxGen); 		//A scalar blade with room to expand.
327 		if (sign == 0)							
328 			return returnIt.setSign((byte) 0);				//In the degenerate case, we are done!
329 		
330 		bladeDuet.stream().forEach(g -> returnIt.add(g));	//Load remaining generators IF NO paired generator was degenerate
331 															//returnIt has the correct generators, but might have the wrong sign
332 		andKey = bladeLeft.bitKey() & bladeRight.bitKey();
333 															// if either residue key vanishes, the bladeDuet is already in SORT order.
334 		if ((bladeLeft.bitKey() - andKey) != 0 & (bladeRight.bitKey() - andKey) != 0) {
335 			ArrayList<Generator> pB = new ArrayList<>(returnIt.getGenerators());
336 			for (Generator pG : pB) { 						// Exploiting the KNOWN correct order.
337 				int found = bladeDuet.indexOf(pG);
338 				int refer = pB.indexOf(pG);
339 
340 				if (found != refer) {
341 					sign *= (byte) -1;
342 					Collections.swap(bladeDuet, found, refer);
343 				}
344 			}
345 		}
346 		return returnIt.setSign(sign);
347 	}
