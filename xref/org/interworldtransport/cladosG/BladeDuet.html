25  package org.interworldtransport.cladosG;
26  
27  import java.util.ArrayList;
28  import java.util.Collections;
29  
30  /**
31   * This class serves two highly related purposes.<br>
32   * 1) It supports blade discovery in blade multiplication by acting as a bucket for their generators that can
33   * be reduced using a signature.<br>
34   * 2) It supports blade discovery in blade complement operations and compute the resulting blade's sign so it
35   * correctly multiplies to the pscalar in the basis that contains the input blade.
36   * <br><br>
37   * In Clados v1 the multiplication methods were all buried in the Basis and GProduct classes. BaldeDuet surfaces
38   * them in order to support parallelization of product table generation. As a side benefit, the complement 
39   * operation supportual Dual() in Clados v2 also benefits from parallelization.
40   * <br><br>
41   * NOTE: BladeDuet makes use of streams, but intentionally avoids internal parallelization of computations. Most of 
42   * what must be done must be in sequence. It is a calling objects that may parallelize operations relying on BladeDuet.
43   * <br><br>
44   * @version 2.0
45   * @author Dr Alfred W Differ
46   */
47  public final class BladeDuet {
48  
49  	/**
50  	 * This method takes the set of generators in the offered blade and returns a blade with a set of generators that 
51  	 * complements the first in such a way that left multiplication of the initial blade by the complement blade produces
52  	 * the +pscalar associated with the initial blade.
53  	 * <br><br>
54  	 * NOTE: Because generators might be degenerate, it is not possible to extract commute signs from the Cayley table
55  	 * when computing complement blades. If not for degeneracy, complement would involve a table lookup.
56  	 * <br><br>
57  	 * @param pB1 Blade to be complemented with respect to it's basis pscalar blade
58  	 * @param sig byte[] signature array to use when reducing duplicate generators
59  	 * @return Blade complement of pB1 with sign set for left multiply returning +pscalar
60  	 */
61  	public static final Blade complementLeft(Blade pB1, byte[] sig) {
62  		BladeDuet tBD = new BladeDuet(pB1, Blade.createPScalarBlade(Generator.get(pB1.maxGenerator()))) ;
63  		return tBD.simplifyForDual(sig);
64  	}
65  
66  	/**
67  	 * This method takes the set of generators in the offered blade and returns a blade with a set of generators that 
68  	 * complements the first in such a way that right multiplication of the initial blade by the complement blade produces
69  	 * the +pscalar associated with the initial blade.
70  	 * <br><br>
71  	 * NOTE: Because generators might be degenerate, it is not possible to extract commute signs from the Cayley table
72  	 * when computing complement blades. If not for degeneracy, complement would involve a table lookup.
73  	 * <br><br>
74  	 * @param pB1 Blade to be complemented with respect to it's basis pscalar blade
75  	 * @param sig byte[] signature array to use when reducing duplicate generators
76  	 * @return Blade complement of pB1 with sign set for left multiply returning +pscalar
77  	 */
78  	public static final Blade complementRight(Blade pB1, byte[] sig) {
79  		BladeDuet tBD = new BladeDuet(Blade.createPScalarBlade(Generator.get(pB1.maxGenerator())), pB1) ;
80  		return tBD.simplifyForDual(sig);
81  	}

83  	/**
84  	 * This method reduces pairs of directions in what is ALMOST a sorted bladeDuet list. It's actually two buckets of 
85  	 * sorted generators that upon duplication removal MIGHT be sorted. If not, we can jump straight to the sorted order
86  	 * simply by inserting the generators in an EnumSet which happens to be their destination in a Blade anyway. What we 
87  	 * don't know immediately is how many transpositions are necessary to reach that sort order. That's what this method 
88  	 * does after removing generator duplicates.
89  	 * <br><br>
90  	 * The offered numeric signature is used for the reduction to handle sign flips. Generators with a positive square 
91  	 * appear as a one (1) while those with negative squares appear as negative one (-1).
92  	 * <br><br>
93  	 * NOTE that the numeric signature representation is a departure with prior use in Clados where zero(0) implied no sign 
94  	 * flip and one(1) implied sign flip for negative squared generator. Prior practice used to add up the sign flips and 
95  	 * then look at what was left modulo 2. Ideally, what we want is a 'signed bit' sized data element to track signs.
96  	 * <br><br>
97  	 * Exception cases NOT checked because this is for CladosG internal use. The method itself is public, but it's 
98  	 * really for internal use.
99  	 * <br><br>
100 	 * @param pB1 Blade appearing on the left/row of a multiplication operation
101 	 * @param pB2 Blade appearing on the right/column of a multiplication operation
102 	 * @param sig signature array to use to reduce duplicate generators
103 	 * @return A fully reduced blade
104 	 */
105 	public static final Blade simplify(Blade pB1, Blade pB2, byte[] sig) {
106 		BladeDuet tBD = new BladeDuet(pB1, pB2);
107 		return tBD.simplify(sig);
108 	}
109 	/**
110 	 * These are the bitKey's of the Blades inserted.
111 	 * They get used to help with the simplify algorithm.
112 	 */
113 	private int bitKeyLeft, bitKeyRight = 0;
114 	
115 	/**
116 	 * This holds the combined list of generators from each blade.
117 	 * Left blade goes first, then right
118 	 */
119 	protected ArrayList<Generator> bladeDuet;
120 
121 	/**
122 	 * This byte holds the sign of the blade to which the list of 
123 	 * generators simplifies in the end. A +1 sign means a blade that 
124 	 * is an even permutation of the remaining generators. A -1 sign 
125 	 * means an odd permutation.
126 	 * <br>
127 	 * Calculating the sign ALSO gets the algebra's signature involved,
128 	 * so this DOES involve the metric.
129 	 */
130 	protected byte sign = 1;
131 
132 	/**
133 	 * This is where the hint is kept for the largest possible blade in the 
134 	 * basis to which the resulting blade actually belongs. It is capped by the 
135 	 * maximum grade from one of the blades... which really should have the 
136 	 * same maximum grade.
137 	 */
138 	protected final Generator maxGen;
139 
140 	/**
141 	 * This is a re-use constructor that builds this as a juxtaposition of the two offered blades.
142 	 * <br>
143 	 * @param pB1 A Blade to re-use on the left.
144 	 * @param pB2 A Blade to re-use on the right.
145 	 */
146 	public BladeDuet(Blade pB1, Blade pB2) {
147 		assert (pB1.maxGenerator() == pB2.maxGenerator());
148 		maxGen = Generator.get((byte) pB1.maxGenerator());
149 		bladeDuet = (maxGen != null) ? new ArrayList<>(2 * maxGen.ord) : new ArrayList<>(2);
150 		pB1.generatorStream().forEachOrdered(g -> bladeDuet.add(g));
151 		sign = pB1.sign();
152 		bitKeyLeft = pB1.bitKey();
153 		pB2.generatorStream().forEachOrdered(g -> bladeDuet.add(g));
154 		sign *= pB2.sign();
155 		bitKeyRight = pB2.bitKey();
156 	}
157 
158 	/**
159 	 * This method reduces generator pairs in what is an ALMOST sorted bladeDuet list. It is two buckets of sorted generators 
160 	 * that upon pair removal WILL be fully sorted because one of the buckets is a pscalar blade. Every generator removed
161 	 * from the input blade is paired with one in the pscalar blade, so removal of pairs leaves a sorted list ensuring the
162 	 * second half of the 'simplify' algorithm isn't necessary here. The first half eliminates generator pairs and computes
163 	 * transposition counts, so that is all this method does
164 	 * <br><br>
165 	 * The offered metric signature resolves what sign a generator pair produces with one twist. For degenerate generators, 
166 	 * this method treats them as if they squared to +1.
167 	 * <br><br>
168 	 * @param pSig byte[] signature array to use when reducing duplicate generators
169 	 * @return Blade complement of the non-pscalar blade used to initialize the BladeDuet.
170 	 */
171 	private Blade simplifyForDual(byte[] pSig) {
172 		int andKey = bitKeyLeft & bitKeyRight;
173 		byte gen = 1;										//start with lowest generator
174 		while (andKey > 0) {								//while any duplicate generators present
175 			if (Integer.lowestOneBit(andKey) == 1) {		//andKey is odd => low bit names duplicate generator. Action required.
176 				Generator eq = Generator.get(gen);			//find generator for that lowest bit
177 				sign *= (Integer.lowestOneBit(bladeDuet.lastIndexOf(eq) ^ bladeDuet.indexOf(eq)) == 1) ? (byte) 1 : (byte) -1;
178 															//lastIndexOf = right-most. indexOf = left-most.
179 															//We won't be in this section unless there are exactly two.
180 															//This 'permutes' generators without moving them.
181 				sign *= (pSig[gen - 1] == 0) ? 1 : pSig[gen - 1];	//IF SIGNATURE of eq is 0, pretend it is +1. 
182 																	//Otherwise use correct signature.
183 				bladeDuet.removeAll(Collections.singleton(eq));
184 			}
185 			gen++;											//move up to the next generator to test
186 			andKey = andKey >>> 1;							//shift andKey right dropping lowest bit
187 		}
188 		Blade returnIt = Blade.createBlade(maxGen); 		//A scalar blade with room to expand.
189 		bladeDuet.stream().forEach(g -> returnIt.add(g));	//Load remaining generators 
190 															//returnIt has the correct generators AND sign.
191 		return returnIt.setSign(sign);
192 	}
193 
194 	/**
195 	 * This method reduces pairs of directions in what is ALMOST a sorted bladeDuet list. It's actually two buckets of 
196 	 * sorted generators that upon duplication removal MIGHT be sorted. If not, we can jump straight to the sorted order
197 	 * simply by inserting the generators in an EnumSet which happens to be their destination in a Blade anyway. What we 
198 	 * don't know immediately is how many transpositions are necessary to reach that sort order. That's what this method 
199 	 * does after removing generator duplicates.
200 	 * <br>
201 	 * The offered numeric signature is used for the reduction to handle sign flips. Generators with a positive square 
202 	 * appear as a one (1) while those with negative squares appear as negative one (-1).
203 	 * <br>
204 	 * NOTE that the numeric signature representation is a departure with prior use in Clados where zero(0) implied no sign 
205 	 * flip and one(1) implied sign flip for  negative squared generator. Prior practice used to add up the sign flips and 
206 	 * then look at what was left modulo 2. Ideally, what we want is a 'signed bit' sized data element to track signs.
207 	 * <br>
208 	 * Exception cases NOT checked because this is for CladosG internal use.
209 	 * <br>
210 	 * @param pSig 	An array of unboxed short integers that signifies when sign flips occur as generator pairs are removed 
211 	 * 				from the internal dual list.
212 	 * @return Blade [supporting stream approach]
213 	 */
214 	private Blade simplify(byte[] pSig) {
215 		int andKey = bitKeyLeft & bitKeyRight;
216 		byte gen = 1;										//start with lowest generator
217 		while (andKey > 0) {								//while any duplicate generators present
218 			if (Integer.lowestOneBit(andKey) == 1) {		//andKey is odd => low bit points at duplicate generator
219 				Generator eq = Generator.get(gen);			//find generator for that lowest bit
220 				sign *= (Integer.lowestOneBit(bladeDuet.lastIndexOf(eq) ^ bladeDuet.indexOf(eq)) == 1) ? (byte) 1 : (byte) -1;
221 															//lastIndexOf = right-most. indexOf = left-most.
222 															//We won't be in this section unless there are exactly two.
223 															//This 'permutes' generators without moving them.
224 				sign *= pSig[gen - 1];						//Adjust again for signature of generators.
225 				bladeDuet.removeAll(Collections.singleton(eq));
226 			}
227 			if (sign == 0) 	break;							//Degenerate eq found. We are done since result is a scalar ZERO.
228 			gen++;											//move up to the next generator to test
229 			andKey = andKey >>> 1;							//shift andKey right dropping lowest bit
230 		}
231 		Blade returnIt = Blade.createBlade(maxGen); 		//A scalar blade with room to expand.
232 		if (sign == 0)							
233 			return returnIt.setSign((byte) 0);				//In the degenerate case, we are done!
234 		
235 		bladeDuet.stream().forEach(g -> returnIt.add(g));	//Load remaining generators IF NO paired generator was degenerate
236 															//returnIt has the correct generators, but might have the wrong sign
237 		andKey = bitKeyLeft & bitKeyRight;
238 															// if either residue key vanishes, the bladeDuet is already in SORT order.
239 		if ((bitKeyLeft - andKey) != 0 & (bitKeyRight - andKey) != 0) {
240 			ArrayList<Generator> pB = new ArrayList<>(returnIt.getGenerators());
241 			for (Generator pG : pB) { 						// Exploiting the KNOWN correct order.
242 				int found = bladeDuet.indexOf(pG);
243 				int refer = pB.indexOf(pG);
244 
245 				if (found != refer) {
246 					sign *= (byte) -1;
247 					Collections.swap(bladeDuet, found, refer);
248 				}
249 			}
250 		}
251 		return returnIt.setSign(sign);
252 	}
