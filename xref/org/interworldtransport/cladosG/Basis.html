40  /**
41   * All geometry objects within the cladosG package have elements that form a
42   * basis to span the vector space related to the algebra. This basis is
43   * represented in the algebra as various products of the generators. This class
44   * representation uses sets of generators as Blades and a list of Blades. It
45   * also maintains a few methods that help manipulate them.
46   * <br><br>
47   * The gradeCount is tracked using byte integers as it isn't expected that
48   * anyone will work with algebras of more than 126 generators any time soon.
49   * <br><br>
50   * The bladeCount is computed as (1 &lt;&lt; gradeCount - 1) instead of stored.
51   * It is reported as an integer though that DOES limit the size of a basis to 31
52   * generators. Again, it isn't expected anyone will need more any time soon.
53   * <br><br>
54   * The bladeList is stored as an ArrayList of Blades that should be no longer
55   * than bladeCount. There is nothing to stop it from being longer or shorter,
56   * though, so this is a potential source of error. It really SHOULD be
57   * immutable once constructed correctly. Same goes for the Blades contained in
58   * the ArrayList.
59   * <br><br>
60   * Blade keys are now stored inside the Blades. There is no separate array for
61   * them in the Basis. The primary use for keys is sorting the Blades and
62   * detecting which blade is found at the head of the list for each grade. This
63   * is useful when multiplying multivectors that are sparsely populated in terms
64   * of their coefficients. The multiplication algorithm can skip over grades not
65   * present in the multivector and skip potentially large blocks in the sums with
66   * vanishing contributions. However, sparseness is detected using a
67   * multivector's key and NOT a Blade's key. Once Blades are sorted into a basis,
68   * blade keys are only used to name them in maps.
69   * <br><br>
70   * NOTE that Blade keys are currently kept as long integers. The key for a
71   * pscalar in a 14 generator basis is 2234152501943160L. For 15 generator
72   * algebras the key is 81985529216486896L. Long integers can't hold keys much
73   * larger than that. At some point keys will shift to Java's BigInteger class
74   * and impose another performance penalty. Not yet, though. Best practice would
75   * be to avoid computations demanding heavy use of blade keys.
76   * <br><br>
77   * The data in this class is stored in objects instead of arrays of primitives.
78   * This is intentional. Doing so allows a system to lay them out in memory in
79   * any way it finds convenient. There IS an overhead associated with this plan,
80   * but it is in recognition that virtualization puts distance between an
81   * application and the hardware on which it runs.
82   * <br><br>
83   * It is expected that Basis objects will be cached, though. There is no reason
84   * to create copies within running applications. One MAY do so as no singleton
85   * enforcement occurs, but every basis of the same number of generators passes
86   * the equality test. A convenient cache already exists in the singleton
87   * GCache.
88   * <br><br>
89   * The choice limit of 15 generators produces a maximum basis size of 32,768.
90   * More can be used, but one must change the 'magic numbers' in the
91   * CladosConstant class and recompile.
92   * <br><br>
93   * There IS a sort buried in the constructor for a Basis. After Blades are
94   * generated using the private powerSet method, they are sorted on the 'natural
95   * order'. Blades implement Comparable and use their keys for compareTo(). THAT
96   * is why blade keys are now buried in Blade objects. Doing so enables one line
97   * of code here to use Java's Collections class to decide what sorting algorithm
98   * to use. That means this class no longer implements its own sort algorithm. It
99   * is likely the developer community is much better at writing sort algorithms,
100  * so this is recognition of that reality.
101  * <br><br>
102  * This class probably should be implemented as a Java enumeration. It might be
103  * some day. The problem with that is construction time scales as O(N^2).
104  * Pre-construction of small basis objects makes good sense, but larger ones
105  * become problematic especially for users who never intend to use them. If
106  * one's primary interest involves Euclidian 3-space, what need is there of a
107  * basis with 14 geometric directions and 16384 linear dimensions that takes a
108  * sizeable fraction of a minute to construct? So it is suggested that Best
109  * Practice among those who build physical models is to prebuild what you need
110  * and load it all to the cache. Use GBuilder to do it for you. 
111  * That said, it is obvious why the copy constructor was removed.
112  * <br><br>
113  * @version 2.0
114  * @author Dr Alfred W Differ
115  */
116 public final class Basis implements CanonicalBasis {

148 	/*
149 	 * Deliver the powerset of generators present in the pscalar of a basis. Members
150 	 * of this set ARE the other blades.
151 	 * <br>
152 	 * This is O(n^2), so improvements here matter.
153 	 */
154 	private final static Set<EnumSet<Generator>> powerSet(Set<Generator> inSet) {
155 		Set<EnumSet<Generator>> sets = new HashSet<EnumSet<Generator>>(inSet.size(), 0.75f);
156 		if (inSet.isEmpty()) {
157 			sets.add(EnumSet.noneOf(Generator.class));
158 			return sets;
159 		}
160 		List<Generator> list = new ArrayList<Generator>(inSet);
161 		Generator head = list.get(0);
162 		Set<Generator> tailset = new HashSet<Generator>(list.subList(1, list.size()));
163 		for (EnumSet<Generator> set : powerSet(tailset)) {
164 			EnumSet<Generator> newSet = EnumSet.noneOf(Generator.class);
165 			newSet.add(head);
166 			newSet.addAll(set);
167 			sets.add(newSet);
168 			sets.add(set);
169 		}
170 		return sets;
171 	}

173 	/**
174 	 * This list holds the representation of a Basis for the 'vector space'
175 	 * associated with an algebra. The list size is ALWAYS bladeCount which is
176 	 * ALWAYS 2^(# of generators available). Also, the list is ALWAYS ordered such
177 	 * that Blade keys ascend as one iterates along the list.
178 	 * <br>
179 	 * Every list entry is a blade. ArrayList DOES allow for null entries, but
180 	 * methods in this class MUST prevent that from happening.
181 	 * <br>
182 	 * While modern lingo refers to this as the canonical basis, Ken Greider tended
183 	 * to describe it as the EDDINGTON BASIS. The reason for that comes from the
184 	 * physical interpretation which he felt traced to Eddington. Greider preferred
185 	 * to avoid interpreting all the elements as 'vector' directions because that
186 	 * confused and conflated the geometric meaning for 'vector'. Most of them are
187 	 * much bigger than one-blades, so a different name that directed students
188 	 * toward Physics history seemed appropriate. In Clados, "Eddington Basis" is
189 	 * used if there is a chance of confusing an algebra's "Generator Basis" with
190 	 * the basis of the vector space spanned by all the blades.
191 	 */
192 	private final ArrayList<Blade> bladeList;
193 
194 	/**
195 	 * This integer is the number of grades in the algebra. It is one more than the
196 	 * number of generators and is used often enough to be worth keeping.
197 	 */
198 	private final byte gradeCount;
199 
200 	/**
201 	 * This list is used for tracking of where grades start and stop in a bladeList.
202 	 * The difference from grade k to k+1 is binomial(GradeCount-1, k) =
203 	 * (GradeCount-1)! / (k! * ((GradeCount-1)-k)!) GradeRange[j] is the first
204 	 * position for a blade of grade j.
205 	 * <br>
206 	 * This list enables the CladosG library to avoid implementing a factorial
207 	 * method to repeatedly calculate binomial coefficients. Construction of the
208 	 * basis provides the information, so for the sake of efficiency they are 
209 	 * stored here for later use.
210 	 * <br>
211 	 * The size of this list is always the same as the grade count.
212 	 * <br>
213 	 * grade 0: The first entry always points to the start as that is where the
214 	 * first (and only) scalar blade is found.
215 	 * <br>
216 	 * grade 1: The second entry always points to the start of the one-blades.
217 	 * <br>
218 	 * grade 2: N+2 where N is the number of generators is where two-blades start.
219 	 * <br>
220 	 * grade N: The last entry always points to the blade that represents the
221 	 * pscalar in the basis.
222 	 * <br>
223 	 * All other entries in this list are calculated using a Blade's key.
224 	 */
225 	private final ArrayList<Integer> gradeList;
226 
227 	/**
228 	 * This map connects a Blade's internal key to its indexed location in the
229 	 * basis. This map is what allows us to put the indexed position in the
230 	 * multiplicaton table instead of the blade's key which is longer and would 
231 	 * inflate the size of the Cayley table unnecessarily.
232 	 * <br>
233 	 * The index "1" always refers to the scalar (no generators blade) of the basis. 
234 	 * The index 2^(p+q+r) always refers to the pscalar of the basis.
235 	 * <br>
236 	 * Nothing should be indexed at "0" because a "0" in the Cayley Table is used 
237 	 * to refer to an outcome of products of degenerate generators.
238 	 */
239 	private final TreeMap<Long, Integer> keyIndexMap;
240 
241 	/**
242 	 * This is the basic constructor. It takes the number of generators as its only
243 	 * parameter. It can be instantiated on its own for demonstration purposes, but
244 	 * it has no awareness of the addition and multiplication operations in an
245 	 * algebra, so all it does is show the basis.
246 	 * <br>
247 	 * @param pGens byte This is the number of generators that make up the basis
248 	 * @throws GeneratorRangeException This exception is thrown when the integer
249 	 *                                 number of generators for the basis is out of
250 	 *                                 the supported range. {0, 1, ..., CladosConstant.MAXGRADE}
251 	 */
252 	public Basis(byte pGens) throws GeneratorRangeException {
253 		if (!CanonicalBasis.validateSize(pGens))
254 			throw new GeneratorRangeException("Supported range is 0<->CladosConstant.MAXGRADE using 8 bit integers");
255 		// ------Initialize
256 		gradeCount = (byte) (pGens + 1);
257 		gradeList = new ArrayList<Integer>(gradeCount);
258 		bladeList = new ArrayList<Blade>(1 << pGens);
259 		keyIndexMap = new TreeMap<>();
260 		// ------Build bladeList
261 		if (pGens == 0) {
262 			bladeList.add(Blade.createBlade(pGens));
263 			gradeList.add(Integer.valueOf(0));
264 			keyIndexMap.put(0L, 1);
265 		} else if (pGens == 1) {
266 			bladeList.add(Blade.createScalarBlade(Generator.E1));
267 			gradeList.add(Integer.valueOf(0));
268 			keyIndexMap.put(0L, 1);
269 			bladeList.add(Blade.createPScalarBlade(Generator.E1));
270 			gradeList.add(Integer.valueOf(1));
271 			keyIndexMap.put(1L, 2);
272 		} else {
273 			EnumSet<Generator> offer = EnumSet.range(CladosConstant.GENERATOR_MIN, Generator.get(pGens));
274 			TreeSet<Blade> sorted = new TreeSet<>(); // Expects things that have a natural order
275 			for (EnumSet<Generator> pG : powerSet(offer))
276 				sorted.add(new Blade(pGens, pG)); // Adds in SORTED ORDER because... TreeSet
277 				
278 			sorted.iterator().forEachRemaining(blade -> { // Iterator works in ascending order
279 				bladeList.add(blade); // causing bladeList to be in ascending (by key) order
280 				keyIndexMap.put(blade.key(), Integer.valueOf(bladeList.indexOf(blade) + 1));
281 			});
282 
283 			// ------Build gradeList
284 			gradeList.add(Integer.valueOf(0)); // First entry in gradeList is for scalar grade
285 			gradeList.add(Integer.valueOf(1)); // Second entry in gradeList is for vector grade
286 			IntStream.range(2, gradeCount - 1).forEachOrdered(i -> {
287 				gradeList
288 						.add(keyIndexMap.ceilingEntry(Long.valueOf((long) Math.pow(gradeCount, i - 1))).getValue() - 1);
289 				}); // keyIndexMap uses bladeKey (known to blade) to get bladeIndex for products.
290 			gradeList.add(getBladeCount() - 1); // Last entry in gradeList is for pscalar grade
291 		}
292 	}

294 	/**
295 	 * This is a 'from point' constructor. It takes a generator and treats it as the pscalar to factor.
296 	 * It can be instantiated on its own for demonstration purposes, but it has no awareness of the 
297 	 * addition and multiplication operations in an algebra, so all it does is show the basis.
298 	 * <br>
299 	 * @param pGen Generator This is the pscalar to factor to make up the basis
300 	 */
301 	public Basis(Generator pGen) {
302 		// ------Initialize
303 		gradeCount = (byte) (pGen.ord + 1);
304 		gradeList = new ArrayList<Integer>(gradeCount);
305 		bladeList = new ArrayList<Blade>(1 << pGen.ord);
306 		keyIndexMap = new TreeMap<>();
307 		// ------Build bladeList
308 		switch (pGen.ord) {
309 			case 1 -> {
310 				bladeList.add(Blade.createScalarBlade(Generator.E1));
311 				gradeList.add(Integer.valueOf(0));
312 				keyIndexMap.put(0L, 1);
313 				bladeList.add(Blade.createPScalarBlade(Generator.E1));
314 				gradeList.add(Integer.valueOf(1));
315 				keyIndexMap.put(1L, 2);
316 				break;
317 			}
318 			default -> {
319 				EnumSet<Generator> offer = EnumSet.range(CladosConstant.GENERATOR_MIN, pGen);
320 				TreeSet<Blade> sorted = new TreeSet<>(); // Expects things that have a natural order
321 				for (EnumSet<Generator> pG : powerSet(offer))
322 					sorted.add(new Blade(pGen, pG)); // Adds in SORTED ORDER because... TreeSet
323 				
324 				sorted.iterator().forEachRemaining(blade -> { // Iterator works in ascending order
325 					bladeList.add(blade); // causing bladeList to be in ascending (by key) order
326 					keyIndexMap.put(blade.key(), Integer.valueOf(bladeList.indexOf(blade) + 1));
327 				});
328 
329 				// ------Build gradeList
330 				gradeList.add(Integer.valueOf(0)); // First entry in gradeList is for scalar grade
331 				gradeList.add(Integer.valueOf(1)); // Second entry in gradeList is for vector grade
332 				IntStream.range(2, gradeCount - 1).forEachOrdered(i -> {
333 					gradeList
334 						.add(keyIndexMap.ceilingEntry(Long.valueOf((long) Math.pow(gradeCount, i - 1))).getValue() - 1);
335 					}); // keyIndexMap uses bladeKey (known to blade) to get bladeIndex for products.
336 				gradeList.add(getBladeCount() - 1); // Last entry in gradeList is for pscalar grade
337 			}
338 		}
339 	}

376 	 * Old-fashioned 'find' method that reports index location in the basis where a
377 	 * Blade is found OR -1 if it wasn't found.
378 	 * <br>
379 	 * @param pIn Blade to be found
380 	 * @return integer index pointing to element of a list containing the Blade OR
381 	 *         -1 if the blade wasn't found.
382 	 */
383 	@Override
384 	public int find(Blade pIn) {
385 		if (!(pIn == null)) {
386 			Integer loc = keyIndexMap.get(pIn.key());
387 			if (!(loc == null))
388 				return loc.intValue();
389 		}
390 		return -1;
391 	}

531 	/**
532 	 * This method produces a printable and parseable string that represents the
533 	 * Basis in a human readable form. return String
534 	 * <br>
535 	 * @param pB The Basis to export as XML
536 	 * @param indent String of 'tab' characters that help space the output correctly
537 	 *               visually. It's not actually necessary except for human
538 	 *               readability of the output.
539 	 * @return String
540 	 */
541 	public static String toXMLString(Basis pB, String indent) {
542 		if (indent == null)
543 			indent = "\t\t\t\t\t\t";
544 		StringBuilder rB = new StringBuilder(indent + "<Basis>\n");
545 		// ------------------------------------------------------------------
546 		rB.append(indent)
547 			.append("\t<Grades count=\"")
548 			.append(pB.getGradeCount() + "\">\n");
549 		for (int k = 0; k <= pB.gradeCount - 2; k++) // loop to get all but the highest grade
550 			rB.append(indent)
551 				.append("\t\t<Grade number=\"")
552 				.append(k)
553 				.append("\" range=\"")
554 				.append(pB.gradeList.get(k))
555 				.append("-")
556 				.append((pB.gradeList.get(k + 1) - 1))
557 				.append("\" />\n");
558 		// Handle last grade separate. There is no k+1 index for the largest grade
559 		rB.append(indent)
560 			.append("\t\t<Grade number=\"")
561 			.append((pB.getGradeCount() - 1))
562 			.append("\" range=\"")
563 			.append(pB.gradeList.get(pB.gradeCount - 1))
564 			.append("-")
565 			.append(pB.gradeList.get(pB.gradeCount - 1))
566 			.append("\" />\n");
567 		rB.append(indent)
568 			.append("\t</Grades>\n");
