89  	/**
90  	 * This small function is for key building and exist soley because java's 
91  	 * Math.pow() accepts doubles and returns a double. I get why they do that, 
92  	 * but I have fairly tight control over the inputs on this method in the 
93  	 * Blade and Basis classes, so overflow isn't likely. 
94  	 * <br><br>
95  	 * Having said that (!), future expansion the number of generators that can 
96  	 * be used in blades and bases will have to revisit this method since long 
97  	 * integers have a limit... even if it is very large by current standards.
98  	 * <br><br>
99  	 * @param base byte to use as the exponential base
100 	 * @param exponent integer to use as the power to raise the base
101 	 * @return Long wrapped result of base^exponent
102 	 */
103 	protected final static Long pow(byte base, int exponent) {
104 		long result = 1;					//Establish default return value... anything^zero is 1.
105 		long sq = base;						//Establish default return value if exponent is 1.
106 		int power = Math.abs(exponent);		//Just in case some quirk hands in negative numbers
107 		while (power > 0) {					//Start looping on the exponent
108 			if (power % 2 == 1) 			//If power is odd
109 				result *= sq;				//update result by multiplying it by the base^(some power of 2)
110 			sq *= sq;						//else power is odd and we square the base
111 			power /= 2;						//Cut the power in half trimming fractions then
112 											//return to the top of the loop
113 		}
114 		return Long.valueOf(result);
115 
116 		/*
117 		Try it out.
118 
119 		Call: 
120 		pow(16, 15)
121 			result starts as 1			
122 			sq starts as 16		
123 			power starts as 15
124 			loop starts with power>0	Decision spots that power is odd
125 											result becomes 1*16
126 										sq becomes 16^2
127 										power becomes 7
128 			loop detects power>0		Decision spots power is odd
129 											result becomes 16*16^2
130 										sq becomes 16^4
131 										power becomes 3
132 			loop detects power>0		Decison spots power is odd
133 											result becomes 16^3*16^4
134 										sq becomes 16^8
135 										power becomes 1
136 			loop detects power>0		Decison spots power is odd
137 											result becomes 16^7*16^8
138 										sq becomes 16^16
139 										power becomes 0
140 			loop detects termination condition
141 		return Long.valueOf(16^15) 
142 
143 		task complete in 4 iterations. (Ceiling of log_2() of exponent )
144 		but did overflows happen?
145 
146 		The longest long integer (32bits) is 	9,223,372,036,854,775,807 (or 2^63 - 1)
147 		The pscalar in a 15 generator algebra is   81,985,529,216,486,896 (grade count = 16)
148 		while 16^15 (15 bit binary key) is		1,152,921,504,606,846,976 (meaning result doesn't overflow)
149 		but setting sq = 16^16 does overflow. Fortunately sq isn't need again in this maximum case.
150 			This overflow is why java's Math.pow() using double precision floats, but it does so at a small cost.
151 			It isn't clear that Clados should roll it's own pow function, but it isn't used often by Blade 
152 			or Basis and developers are discouraged from re-using it. Even large byte entries can blow up.
153 		Speed tests suggest any change in efficiency is negligible for our purposes.
154 		*/
155 	}
