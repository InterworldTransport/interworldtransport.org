25  package org.interworldtransport.cladosG;
26  
27  import java.util.ArrayList;
28  import java.util.EnumSet;
29  import java.util.stream.IntStream;
30  import java.util.stream.LongStream;
31  import java.util.stream.Stream;
32  
33  /**
34   * This interface represents the 'contract' obeyed by a canonical basis in cladosG. 
35   * Most of the interface focuses upon gettor methods that return blade information, 
36   * but there are a few convenience methods too that support grade information and 
37   * ranges as detected by the number of generators found in a blade. Finally there 
38   * are also a few methods for dealing with keys and basis size validation.
39   * <br><br>
40   * The farther one gets from basic methods dealing with blades and span, the more 
41   * one is wandering off into conveniences. The various keys matter, but as computational 
42   * short-cuts for product tables having to deal with very large vector spaces.
43   * <br><br>
44   * @version 2.0
45   * @author Dr Alfred W Differ
46   */
47  public interface CanonicalBasis {
48  
49  	/**
50  	 * This method detects whether or not the number of generators being used is
51  	 * currently supported by Clados. The point for it is to prevent internal errors
52  	 * in other classes.
53  	 * <br>
54  	 * For example, someone attempting to create a basis with a negative number of
55  	 * generators should be stopped up front without having to check for that
56  	 * situation themselves.
57  	 * <br>
58  	 * The trickier one, though, involves a number of generators that will blow up
59  	 * an internal representation in a basis or CliffordProduct class. For example,
60  	 * trying to build a basis with 64 generators will create a list of 2E64
61  	 * blades... or at least the library will try if the size isn't validated first.
62  	 * Perhaps the list representation of a Basis can hold that many Blades, but the
63  	 * keys identifying them might not if they rely on java primitives. Also, the
64  	 * related CliffordProduct holds a product table of all blades. Its internal
65  	 * representation might break too.
66  	 * <br>
67  	 * This method provides a single place to validate the largest possible grade of 
68  	 * a basis. Having it be static avoids having this test be specialized for each 
69  	 * Basis object. This version AVOIDS placing magic numbers in the interface or 
70  	 * objects. Actual limits are in the CladosConstant class as static elements.
71  	 * <br>
72  	 * @param pGens int This is the number of generators of the basis.
73  	 * @return TRUE returned if the number of generators is in the supported range.
74  	 *         FALSE otherwise.
75  	 */
76  	public static boolean validateSize(int pGens) {
77  		return (pGens >= CladosConstant.SCALARGRADE & pGens <= CladosConstant.MAXGRADE);
78  	}
79  
80  	/**
81  	 * In support of streams, a basis should offer a stream of its blades of a
82  	 * particular grade.
83  	 * <br>
84  	 * @param pIn byte integer for magnitude of grade of blades in stream.
85  	 * @return Stream of Blades in the basis of a particular grade.
86  	 */
87  	public abstract Stream<Blade> bladeOfGradeStream(byte pIn);
88  
89  	/**
90  	 * In support of streams, a basis should offer a stream of its blades.
91  	 * <br>
92  	 * @return Stream of Blades in the basis
93  	 */
94  	public abstract Stream<Blade> bladeStream();
95  
96  	/**
97  	 * Old-fashioned 'find' method that reports index location in the basis where a
98  	 * Blade is found OR -1 if it wasn't found.
99  	 * <br>
100 	 * @param pIn Blade to be found
101 	 * @return integer index pointing to element of a list containing the Blade OR
102 	 *         -1 if the blade wasn't found.
103 	 */
104 	public abstract int find(Blade pIn);
105 
106 	/**
107 	 * Return the number of independent blades in the basis. This is the same as the
108 	 * linear dimension of an algebra that uses this basis.
109 	 * <br>
110 	 * @return int
111 	 */
112 	public abstract int getBladeCount();
113 
114 	/**
115 	 * There is a similar sounding method in the old basis that isn't being used, so
116 	 * it is replaced by this one that returns the enumerated set of generators in
117 	 * the requested blade.
118 	 * <br>
119 	 * @param p1 integer pointing to the blade in the internal list
120 	 * @return EnumSet of Generator representing the blade without the context
121 	 *         necessary for knowing much about the enclosing space for the blade.
122 	 */
123 	public abstract EnumSet<Generator> getBladeSet(int p1);
124 
125 	/**
126 	 * Return the number of grades in the basis. Since there is no geometry in the
127 	 * basis this is a measure of the number of distinct generator subset types that
128 	 * can be formed where the element count determines the type. Because the empty
129 	 * set includes no generators, GradeCount will always be one more than the
130 	 * number of generators.
131 	 * <br>
132 	 * @return byte
133 	 */
134 	public abstract byte getGradeCount();
135 
136 	/**
137 	 * This method simply delivers the otherwise private grade range list. Useful
138 	 * for testing purposes, but should be avoided as much as possible.
139 	 * <br>
140 	 * @return ArrayList of Integers A list of grades boxed as Integers.
141 	 */
142 	public abstract ArrayList<Integer> getGrades();
143 
144 	/**
145 	 * Get an index to the first blade of grade specified by the parameter.
146 	 * <br>
147 	 * @param p1 byte This is for choosing which grade index range to return.
148 	 * @return int Index within the basis where requested grade starts.
149 	 */
150 	public abstract int getGradeStart(byte p1);
151 
152 	/**
153 	 * Return the long at p1 in the EddingtonKey array.
154 	 * <br>
155 	 * @param p1 short This is the desired key at p1 .
156 	 * @return long
157 	 */
158 	public abstract long getKey(int p1);
159 
160 	/**
161 	 * This is a short-hand method for getSingleBlade('last'). It just returns the
162 	 * last blade in the basis.
163 	 * <br>
164 	 * @return Blade that is the last in the basis
165 	 */
166 	public abstract Blade getPScalarBlade();
167 	
168 	/**
169 	 * This is a special version of getGradeStart() that finds the highest grade.
170 	 * <br>
171 	 * @return int Index within the basis where pscalar grade starts.
172 	 */
173 	public abstract int getPScalarStart();
174 	
175 	/**
176 	 * This is a short-hand method for getSingleBlade(0). It just returns the first
177 	 * blade in the basis.
178 	 * <br>
179 	 * @return Blade that is the first in the basis
180 	 */
181 	public abstract Blade getScalarBlade();
182 
183 	/**
184 	 * Simple gettor method retrieves the Blade at the indexed position in the
185 	 * Basis.
186 	 * <br>
187 	 * @param p1 integer index
188 	 * @return Blade at the indexed position.
189 	 */
190 	public abstract Blade getSingleBlade(int p1);
191 
192 	/**
193 	 * This should essentially be an integer stream, but grades are very small. The
194 	 * maximum grade in a basis is equal to the number of generators used to create
195 	 * it. For now, a single byte integer suffices, but IntStream is better
196 	 * supported by Java.
197 	 * <br>
198 	 * @return Stream of Grades in the basis
199 	 */
200 	public abstract IntStream gradeStream();
201 
202 	/**
203 	 * This method should answer the question whether the offered blade is in the basis.
204 	 * How it does so is up to the concrete class to decide, but the usual way involves
205 	 * testing for keys in a list or map containing the blades of the basis.
206 	 * <br>
207 	 * @param pB	Blade to be checked if it is in the basis.
208 	 * @return boolean answering true of false about whether the blade is in the basis
209 	 */
210 	public abstract boolean hasBlade(Blade pB);
211 
212 	/**
213 	 * Similar to bladestream(), this method returns a stream of boxed long integers
214 	 * that represents blade keys.
215 	 * <br>
216 	 * @return LongStream of blade keys.
217 	 */
218 	public abstract LongStream keyStream();
219 
220 	/**
221 	 * This is a validator detects blade out of range issues. If one tries to name a
222 	 * blade by its index, it is always possible for the offered integer to be out
223 	 * of range.
224 	 * <br>
225 	 * This method demands a contract term for validating the blade index to be referenced.
226 	 * It is NOT static because the maximum grade of a basis is object dependent.
227 	 * Implementations of this method should AVOID placing magic numbers in the method 
228 	 * body by using CladosConstant static elements like validateSize(int) does.
229 	 * <br>
230 	 * @param pIn Integer representing the integer index of the blade
231 	 * @return boolean True if parameter in the supported range [0, bladeCount]
232 	 */
233 	public abstract boolean validateBladeIndex(int pIn);
234 
235 	/**
236 	 * This validator detects grade out of range issues. If one tries to name a
237 	 * grade by its index, it is always possible for the offered integer to be out
238 	 * of range.
239 	 * <br>
240 	 * This method demands a contract term for validating the grade index to be referenced.
241 	 * It is NOT static because the maximum blade of a basis is object dependent.
242 	 * Implementations of this method should AVOID placing magic numbers in the method 
243 	 * body by using CladosConstant static elements like validateSize(int) does.
244 	 * <br>
245 	 * @param pIn Integer representing the integer index of the grade
246 	 * @return boolean True if parameter in the supported range [0, gradeCount]
247 	 */
248 	public abstract boolean validateGradeIndex(int pIn);
249 
250 }
