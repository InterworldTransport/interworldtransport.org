36  /**
37   * This class is essentially a connection patching the basis of one algebra to the basis of another. Think about
38   * Cristoffel coefficients and you get the rough idea. As such, there will always be two different algebras
39   * with two different Feet.
40   * <br><br>
41   * This class has a usage doc with the primary scenario that drives its design. Rather than duplicate the explanation
42   * from there to here and risk getting the documentation out of sync, check out the Connection usage page.
43   * <br><br> 
44   * @param <D> ProtoN child class is used in the inner maps for weights of blades. (Linear Combinations)
45   * @version 2.0
46   * @author Dr Alfred W Differ
47   */
48  public final class Connection<D extends ProtoN & Field & Normalizable> implements Unitized, Modal{
49  
50      /**
51  	 * The outer context for this Frame is an Algebra containing a basis with blades to BE represented as a linear 
52       * combination other blades. This first algebra is the image space of the map of maps.
53  	 */
54  	protected Algebra algebra1;
55  
56      /**
57  	 * The inner context for this Frame is an Algebra containing a basis with blades to act IN a linear 
58       * combination of blades. This second algebra is the domain space of the map of maps. 
59       * <br><br>
60       * NOTE there is no reason why algebra2 can't be the same as algebra1. This is typical of maps that 
61       * represent rotations, translations, reflections, and so on. 
62  	 */
63  	protected Algebra algebra2;
64  
65      /**
66  	 * When scales are appended to the internal map, they should all share the same cardinal. That cardinal is 
67  	 * referenced here for ease of access and to act as a standard.
68  	 */
69  	private Cardinal card;
70  
71      /**
72       * This map is is a helper for transposing the cast operation. It has key/value pairs at every blade in the first
73       * algebra that is mapped to a blade in the second algebra. One key/value pair for each relationship. That makes
74       * this map a direct container for ordered pairs of blades. Both keys and values are fundamentally similar even
75       * if they are from different algebras because they are JUST blades. That makes the relationship is symmetric.
76       * Both sets of blades can be streamed to find related blades in the other algebra.
77       * <br><br>
78       * In this map, blades from algebra1 act as keys. Blades from algebra2 act as values. If a key/value pair is 
79       * inserted for (bladeX, bladeY) it means there is a weighted entry in mapOfMaps for bladeX's Scale for bladeY.
80       * <br><br>
81       */
82      private TreeMap<Blade, Blade> mapOfBlades;
83      
84      /**
85       * This map is the heart of this class. At the top level the key blades from the basis in 'algebra1' are used to 
86       * point at other maps (Scales) that contain key blades from the basis in 'algebra2' to number values. 
87       * That means the inner map is a linear combination of blades that collected into a set with the other blades 
88       * are a transformation from one blade set to the other. That makes this map of maps an extensor.
89       * <br><br>
90       */
91      private TreeMap<Blade, Scale<D>> mapOfMaps;
92  
93      /**
94       * This map is the inversed heart of this class. At the top level the key blades from the basis in 'algebra2' 
95       * are used to point at other maps (Scales) that contain key blades from the basis in 'algebra1' to number values. 
96       * That means the inner map is a linear combination of blades that collected into a set with the other blades 
97       * are a transformation from one blade set to the other. That makes this map an inverse of mapOfMaps if the weights
98       * are multiplicative inverses.
99       * <br><br>
100      */
101     private TreeMap<Blade, Scale<D>> mapOfMapsInverse;
102 
103     /**
104      * This map is the counterpart to mapOfBlades that has an ordered pair of blades (BladeDuet) as keys and the kind 
105      * of numbers found in the Scales from mapOfMaps as values. The value IS the weight of the relationship between 
106      * the two blades in the ordered pair.
107      */
108     private TreeMap<BladeDuet, D> mapOfWeights;
109 
110     /**
111 	 * This is the type of ProtoN child that should be present in the map of scales referenced by this class. 
112      * For example, if mode = CladosField.REALF, then all elements in the list will be the RealF child of ProtoN. 
113 	 * <br><br>
114 	 * Mode ensures the scale elements all have the same precision and come from the same numeric field.
115 	 */
116 	private final CladosField mode;
117     
118      /**
119       * Construct a Connection with everything required being provided up front except the numbers acting as weights for linear combinations.
120       * Without numbers, assume the two bases connect directly. For example E1 in algebra1 is E1 in algebra2... and so on.
121       * <br><br>
122       * Note: Because the two algebras can be of different sizes there is an asymmetry in the map of maps due to how 'equivalent' blades are 
123       * found in the filter on the second algebra. There can be blades in the inner algebra that are never found which leaves the Scale for
124       * the outer blade set to ZERO. If the outer algebra is smaller (e.g. Cl(3,0,0)) while the inner algebra is larger (e.g. Cl(3,0,1)) then
125       * there are a number of blades in the outer algebra with ZERO scales in the map of maps. If the outer algebra is larger than the inner
126       * one, the same thing happens because no equivalent blade is found in the inner stream's filter. 
127       * <br><br>
128       * @param pA1      Algebra #1 providing context
129       * @param pA2      Algebra #2 providing context
130       * @param pMode    Precision mode used by numbers in the transformation maps.
131       * @param pCard    Cardinal used by the numbers in the transformation maps.
132       */
133     public Connection(Algebra pA1, Algebra pA2, CladosField pMode, Cardinal pCard) {
134         mode = pMode;
135         card = pCard;
136 
137         algebra1 = pA1;
138         algebra2 = pA2;
139         
140         mapOfBlades = new TreeMap<>();
141         mapOfMaps = new TreeMap<>();
142         mapOfWeights = new TreeMap<>();
143         pA1.getBasis().bladeStream().forEach(b1 -> {                                                             //Pick a blade in the outer algebra. Don't go parallel.
144             Scale<D> tScale = new Scale<D>(mode, pA2.getBasis(), card);                                          //Create a zero Scale using the inner algebra
145             Optional<Blade> similar = pA2.getBasis().bladeStream().filter(b2 -> CanonicalBlade.equivalent(b1, b2)).findFirst(); //Find equivalent blade in inner algebra
146             if (similar.isPresent()) {                                                                          //If inner algebra has equivalent blade
147                 switch (mode) {
148                     case COMPLEXD -> tScale.put(similar.get(), (D) ComplexD.create(card, 1.0D, 0.0D));   //Replace weight at that blade to ONE
149                     case COMPLEXF -> tScale.put(similar.get(), (D) ComplexF.create(card, 1.0F, 0.0F));   //Replace weight at that blade to ONE
150                     case REALD -> tScale.put(similar.get(), (D) RealD.create(card, 1.0D));                  //Replace weight at that blade to ONE
151                     case REALF -> tScale.put(similar.get(), (D) RealF.create(card, 1.0F));                  //Replace weight at that blade to ONE
152                 }
153             }                                                                                                   //ELSE not needed. tScale initiated with zero weights.
154                                                                                                                 //Zero scale or ONE at equivalent blade
155             mapOfMaps.put(b1, tScale);                                                                          //Happens for every b1. Processing order doesn't matter.
156         });
157         setAltMaps();
158     }
159 
160     /**
161      * This method sifts through the map of maps and rebuilds the blade lists to support connection transpose operations.
162      */
163     private void setAltMaps() {
164         bladeStream().forEachOrdered(b1 -> {
165             Scale<D> value = mapOfMaps.get(b1);
166             value.getMap().keySet().stream().forEach(b2 -> {
167                 mapOfBlades.put(b1, b2);
168                 mapOfWeights.put(new BladeDuet(b1, b2), mapOfMaps.get(b1).get(b2));
169             });
170         });
171     }
172 
173     /**
174      * This stream should produce the same output (blades from algebra1) as bladeStream(), but it does so by looking 
175      * at the mapOfBlades instead of the mapOfMaps. That means when you get the corresponding value from the pair you 
176      * get a single blade from algebra2.
177      * <br><br>
178      * The counterpart stream is blade2PairStream().
179      * <br><br>
180      * @return Stream of Blades from algebra1 that appear as keys in mapOfMaps
181      */
182     public Stream<Blade> blade1PairStream() {
183         return mapOfBlades.keySet().stream();
184     }
185 
186     /**
187      * This stream would produce output blades from algebra2 like bladeStream() if mapOfMaps was transposed. It does so 
188      * by looking at the mapOfBlades instead. That means when you get the corresponding key from the pair you get a 
189      * single blade from algebra1.
190      * <br><br>
191      * The counterpart stream is blade1PairStream().
192      * <br><br>
193      * @return Stream of Blades from algebra2 that appear any Scale in mapOfMaps.
194      */
195     public Stream<Blade> blade2PairStream() {
196         return mapOfBlades.values().stream();
197     }

200     /**
201      * This is the blade stream of the outer map. These blades from algebra1 are keys paired up with Scales using algebra2.
202 	 * <br><br>
203      * @return Stream of Blades from algebra1 that are in the outer map
204      */
205     public Stream<Blade> bladeStream() {
206         return mapOfMaps.keySet().stream();
207     }

209     /**
210      * A monad in one algebra is remapped to another algebra using a Connection.
211      * <br><br>
212      * Cast involves turning a monad's scale from a map defined in terms of blades of algebra1 into blades of algebra2. Where a blade in pM's 
213      * scale matches a blade in the outer mapOfMaps basis, the weight from pM's scale is used to weight the inner map. Once all pM's blades 
214      * are considered, the inner maps are summed and the result becomes the new Scale for pM.
215      * <br><br>
216      * See the usage documentation for Connection for more details.
217      * <br><br>
218      * @param pM Monad to be cast
219      * @return Monad transformed by the cast operation
220      */
221     public Monad cast(Monad pM) {
222         if (pM == null)                 return null;                                        //Oops. Nothing to do.
223         if (pM.getMode() != mode)       return pM;                                          //No. Mixed modes makes messes.
224         if (pM.getCardinal() != card)   return pM;                                          //No. Apples and Oranges.
225         Scale<D> newScale = new Scale<>(mode, algebra2.getBasis(), card);                   //new zeroed Scale uses THIS mode and cardinal
226 
227         if (pM.sparseFlag)                                                  //Few grades in use, so blocks of zero weights are skipped
228             pM  .bladeOfGradesStream()                                      //No parallelization (I think) because aggregating
229                 .forEach(b1 -> {newScale.aggregate(                         //aggregate into the replacement Scale
230                                 GBuilder.copyOfScale(get(b1))             //a copy of the relevant Scale
231                                         .scale(pM.get(b1)));                //weighted correctly for that blade. (Could be scaled by ZERO.)
232                                }    //[the action for each non-zero blade in pM]
233                         );          //[far edge of forEach loop]
234         else                                                                //Many grades in use, so individual zero weights are skipped.
235             pM  .getWeights()
236                 .bladesNotZeroStream()                                      //No parallelization (I think) because aggregating
237                 .forEach(b1 -> {newScale.aggregate(                         //aggregate into the replacement Scale
238                                 GBuilder.copyOfScale(get(b1))             //a copy of the relevant Scale
239                                         .scale(pM.get(b1)));                //weighted correctly for that blade. (Never scaled by ZERO.)
240                                }    //[the action for each non-zero blade in pM]
241                         );          //[far edge of forEach loop]
242 
243         pM.setAlgebra(algebra2);
244         try {pM.setScale(newScale);}                                                        //This should never fail because...
245         catch (CladosMonadException e) {                                                    //the possible ways for it are...
246             throw new IllegalArgumentException("Connection.cast error shouldn't happen");//prevented by use of algebra2.
247         }
248         return pM;
249     }

251     /**
252      * Get the Scale object associated with the blade in the outer layer of the map.
253      * <br><br>
254      * @param pB    Blade to use as the index for finding the Scale map
255      * @return Scale of D which extend ProtoN and other numeric interfaces
256      */
257     public Scale<D> get(Blade pB) {
258         if (algebra1.getBasis().hasBlade(pB))
259             return mapOfMaps.get(pB);
260         return null;
261     }

263     /**
264      * Retrieve a reference to one of the algebras used in this object. The boolean input picks between
265      * outer/inner (algebra1/algebra2) choices.
266      * <br><br>
267      * @param pOuter boolean True returns the outer map's algebra, False returns the inner maps algebra
268      * @return Algebra (Either the first or second one depending on the boolean)
269      */
270     public Algebra getAlgebra(boolean pOuter) {
271         if (pOuter)     return algebra1;
272         else            return algebra2;
273     }

295     /**
296      * PUT a Blade, Scale key/value pair into the mapOfMaps. Check that the operation is legitimate first, though.
297      * <br><br>
298      * @param pB    Blade to use as the key for finding the Scale map
299      * @param pS    Scale of D which extend ProtoN and other numeric interfaces
300      * @return Frame of D which extend ProtoN and other numeric interfaces. Basically... this object.
301      */
302     public Connection<D> put(Blade pB, Scale<D> pS) {
303         if (algebra1.getBasis().hasBlade(pB) & pS.getBasis().hasBlade(pB))   //This is enough to ensure Scale's basis matches Algebra's basis.
304             mapOfMaps.put(pB, pS);
305         return this;
306     }
307 
308     /**
309      * Remove a Blade, Scale key/value pair into the mapOfMaps. Check for legitimate use isn't needed, though, because remove
310      * fails quietly if the key isn't present.
311      * <br><br>
312      * @param pB    Blade to use as the key for finding the Scale map
313      * @return Connection of D which extend ProtoN and other numeric interfaces. Basically... this object.
314      */
315     public Connection<D> remove(Blade pB) {
316         if (algebra1.getBasis().hasBlade(pB))       mapOfMaps.remove(pB);        
317         return this;
318     }

337     /**
338      * This method causes all coefficients to be set to zero re-using their cardinals.
339      * <br><br>
340      * @param pMode CladosField mode in which the numbers operate.
341      * @return Frame after it has had all the numbers zero'd out.
342      */
343     protected Connection<D> zeroAll(CladosField pMode) {
344 		algebra1.getBasis()  .bladeStream().forEach(b -> {
345 			    mapOfMaps   .get(b).weightsParallelStream().forEach(scl -> {
346                     scl =   FBuilder.createZERO(pMode, scl.getCardinal());}
347             );
348         });
349 		return this;
350 	}

354      * <br><br>

357      * @return Frame after it has had all the numbers zero'd out.
358      */
359     protected Connection<D> zeroAll(CladosField pMode, Cardinal pCard) {

361 			     mapOfMaps   .get(b).weightsParallelStream().forEach(scl -> {
362                     scl =   FBuilder.createZERO(pMode, pCard);}
363             );
364         });
365 		return this;
366 	}
