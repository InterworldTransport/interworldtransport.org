45  /**
46   * This class is essentially a connection patching the basis of one algebra to the basis of another. Think about
47   * Cristoffel coefficients and you get the rough idea. As such, there will always be two different algebras
48   * with two different Feet.
49   * <br><br>
50   * This class has a usage document with the primary scenario that drives its design. Rather than duplicate the
51   * explanation from there to here and risk getting the documentation out of sync, check out the site usage pages.
52   * <br><br> 
53   * @param <D> ProtoN child class is used in the inner maps for weights of blades. (Linear Combinations)
54   * @version 2.0
55   * @author Dr Alfred W Differ
56   */
57  public final class Connection<D extends ProtoN & Field & Normalizable> implements Unitized, Modal{
58  
59      /**
60  	 * The outer context for this Frame is an Algebra containing a basis with blades to BE represented as a linear 
61       * combination other blades. This first algebra is the image space of the map of maps.
62  	 */
63  	protected Algebra algebra1;
64  
65      /**
66  	 * The inner context for this Frame is an Algebra containing a basis with blades to act IN a linear 
67       * combination of blades. This second algebra is the domain space of the map of maps. 
68       * <br><br>
69       * NOTE there is no reason why algebra2 can't be the same as algebra1. This is typical of maps that 
70       * represent rotations, translations, reflections, and so on. 
71  	 */
72  	protected Algebra algebra2;
73  
74  
75      /**
76  	 * When scales are appended to the internal map, they should all share the same cardinal. That cardinal is 
77  	 * referenced here for ease of access and to act as a standard.
78  	 */
79  	private Cardinal card;
80  
81      /**
82       * This map is the heart of this class. At the top level the key blades from the basis in 'algebra1' are used to 
83       * point at other maps (Scales) that contain key blades from the basis in 'algebra2' to number values. 
84       * That means the inner map is a linear combination of blades that collected into a set with the other blades 
85       * are a transformation from one blade set to the other. That makes this map of maps an extensor.
86       * <br><br>
87       */
88      private TreeMap<Blade, Scale<D>> mapOfMaps;
89  
90      /**
91  	 * This is the type of ProtoN that should be present in the map of scales referenced by this class. For example, 
92  	 * if mode = CladosField.REALF, then all elements in the list will be the RealF child of ProtoN. 
93  	 * <br><br>
94  	 * Mode ensures the scale elements all have the same precision and come from the same numeric field.
95  	 */
96  	private final CladosField mode;
97      
98       /**
99        * Construct a Connection with everything required being provided up front except the numbers acting as weights for linear combinations.
100       * Without numbers, assume the two bases connect directly. For example E1 in algebra1 is E1 in algebra2... and so on.
101       * <br><br>
102       * Note: Because the two algebras can be of different sizes there is an asymmetry in the map of maps due to how 'equivalent' blades are 
103       * found in the filter on the second algebra. There can be blades in the inner algebra that are never found which leaves the Scale for
104       * the outer blade set to ZERO. If the outer algebra is smaller (e.g. Cl(3,0,0)) while the inner algebra is larger (e.g. Cl(3,0,1)) then
105       * there are a number of blades in the outer algebra with ZERO scales in the map of maps. If the outer algebra is larger than the inner
106       * one, the same thing happens because no equivalent blade is found in the inner stream's filter. 
107       * <br><br>
108       * @param pA       Algebra #1 providing context
109       * @param pB       Algebra #2 providing context
110       * @param pMode    Precision mode used by numbers in the transformation maps.
111       * @param pCard    Cardinal used by the numbers in the transformation maps.
112       */
113     public Connection(Algebra pA, Algebra pB, CladosField pMode, Cardinal pCard) {
114         algebra1 = pA;
115         algebra2 = pB;
116         mode = pMode;
117         card = pCard;
118         mapOfMaps = new TreeMap<>();
119         pA.getBasis().bladeStream().parallel().forEach(b1 -> {                                                  //Pick a blade in the outer algebra
120             Scale<D> tScale = new Scale<D>(mode, pB.getBasis(), card);                                          //Create a zero Scale using the inner algebra
121             Optional<Blade> similar = pB.getBasis().bladeStream().filter(b2 -> CanonicalBlade.equivalent(b1, b2)).findFirst(); //Find equivalent blade in inner algebra
122             if (similar.isPresent()) {                                                                          //If inner algebra has equivalent blade
123                 switch (mode) {
124                     case COMPLEXD -> tScale.put(similar.get(), (D) ComplexD.create(card, 1.0D, 0.0D));   //Replace weight at that blade to ONE
125                     case COMPLEXF -> tScale.put(similar.get(), (D) ComplexF.create(card, 1.0F, 0.0F));   //Replace weight at that blade to ONE
126                     case REALD -> tScale.put(similar.get(), (D) RealD.create(card, 1.0D));                  //Replace weight at that blade to ONE
127                     case REALF -> tScale.put(similar.get(), (D) RealF.create(card, 1.0F));                  //Replace weight at that blade to ONE
128                 }
129             }                                                                                                   //ELSE not needed. tScale initiated with zero weights.
130                                                                                                                 //Zero scale or ONE at equivalent blade
131             mapOfMaps.put(b1, tScale);                                                                          //Happens for every b1. Processing order doesn't matter.
132         });    
133     }

135     /**
136      * TODO CAST: A monad in one algebra is remapped to another algebra using a Connection.<br>
137      * Cast involves turning a monad's scale from a map defined in terms of blades of algebra1 into blades of algebra2. Where a blade in pM's 
138      * scale matches a blade in the outer mapOfMaps basis, the weight from pM's scale is used to weight the inner map. Once all pM's blades 
139      * are considered, the inner maps are summed a blade at a time with the result becoming the new Scale for pM.<br>
140      * Examples: <br>
141      * 1) The simplest cast involves projecting a monad from one algebra to another exactly parallel one. Weights simply transfer.<br>
142      * 2) Another casts a monad into a subalgebra where a k-blade in the larger one is the pscalar in the subalgebra.
143      * <br><br>
144      * Map(Blades1, Map(Blades2, Weights2)) casts Map(Blades1, Weights1) -> Map(Blades2, some linear combination of Weights1 and Weights2) 
145      * <br><br>
146      * @param pM Monad to be cast
147      * @return Monad transformed by the cast operation
148      */
149     public Monad cast(Monad pM) {
150         return null;
151     }

153     /**
154      * Get the Scale object associated with the blade in the outer layer of the map.
155      * <br><br>
156      * @param pB    Blade to use as the index for finding the Scale map
157      * @return Scale of D which extend ProtoN and other numeric interfaces
158      */
159     public Scale<D> getAt(Blade pB) {
160         if (algebra1.getBasis().hasBlade(pB))
161             return mapOfMaps.get(pB);
162         return null;
163     }
164 
165     /**
166 	 * Simple gettor method for the Cardinal associated with these objects.
167 	 * <br><br>
168 	 * @return Cardinal in use in this.
169 	 */
170     @Override
171     public Cardinal getCardinal() {
172         return card;
173     }
174 
175     /**
176 	 * Simple gettor method reporting the Connection's internal mode.
177 	 * <br><br>
178 	 * @return CladosField element reporting which ProtoN child is expected in the Scale's used.
179 	 */
180     @Override
181     public CladosField getMode() {
182         return mode;
183     }
184 
185     /**
186      * PUT a Blade, Scale key/value pair into the mapOfMaps. Check that the operation is legitimat first, though.
187      * <br><br>
188      * @param pB    Blade to use as the index for finding the Scale map
189      * @param pS    Scale of D which extend ProtoN and other numeric interfaces
190      * @return Frame of D which extend ProtoN and other numeric interfaces. Basically... this object.
191      */
192     public Connection<D> put(Blade pB, Scale<D> pS) {
193         if (algebra1.getBasis().hasBlade(pB) & pS.getBasis().hasBlade(pB))   //This is enough to ensure Scale's basis matches Algebra's basis.
194             mapOfMaps.put(pB, pS);
195         return this;
196     }
197 
198     /**
199      * This is the compliment of a blade stream involving the scaling maps 'multiplied' by blades in the sense 
200 	 * of a linear combination in a vector space. When framing a new blade in terms of others in the basis, these scales
201      * are the linear combinations of blades to construct them.
202 	 * <br><br>
203      * 
204 	 * Since the internal map can accept any of the CladosF numbers as values, there is a cast to a 'generic' type 
205 	 * within this method. This would normally cause warnings by the compiler since the generic named in the internal 
206 	 * map IS a ProtoN child AND casting an unchecked type could fail at runtime.
207 	 * <br><br>
208 	 * That won't happen when CladosF builders are used because they dan't build anything that is NOT a ProtoN child. 
209 	 * Scale's internal map only accepts ProtoN child classes, so there is no danger of a failed cast operation... 
210 	 * until someone creates a new ProtoN child class and fails to update the builders.
211 	 * <br><br>
212      * @return Stream of Scales of numbers (ProtoN children)
213      */
214     public Stream<Scale<D>> scaleStream() {
215         return mapOfMaps.values().stream();
216     }
217 
218     /**
219 	 * This method returns a parallelizable stream of the Scales in this Frame. It is intended for wholesale 
220      * operations on the weights that may be done in any order. It is mostly for use by the owning object of this Scale.
221 	 * <br><br>
222 	 * @return A stream of weights as children of ProtoN.
223 	 */
224     public Stream<Scale<D>> scaleParallelStream() {
225         return mapOfMaps.values().parallelStream();
226     }
227 
228     /**
229      * This method causes all coefficients to be set to zero re-using their cardinals.
230      * <br><br>
231      * @param pMode CladosField mode in which the numbers operate.
232      * @return Frame after it has had all the numbers zero'd out.
233      */
234     protected Connection<D> zeroAll(CladosField pMode) {
235 		algebra1.getBasis()  .bladeStream().forEach(b -> {
236 			    mapOfMaps   .get(b).weightsParallelStream().forEach(scl -> {
237                     scl =   FBuilder.createZERO(pMode, scl.getCardinal());}
238             );
239         });
240 		return this;
241 	}
242 
243     /**
244      * This method causes all coefficients to be set to zero using the new cardinal.
245      * <br><br>
246      * @param pMode CladosField mode in which the numbers operate.
247      * @param pCard Cardinal to use when rebuilding the numbers.
248      * @return Frame after it has had all the numbers zero'd out.
249      */
250     protected Connection<D> zeroAll(CladosField pMode, Cardinal pCard) {
251 		algebra1.getBasis()  .bladeStream().forEach(b -> {
252 			    mapOfMaps   .get(b).weightsParallelStream().forEach(scl -> {
253                     scl =   FBuilder.createZERO(pMode, pCard);}
254             );
255         });
256 		return this;
257 	}
