25  package org.interworldtransport.cladosG;
26  
27  import java.util.stream.Stream;
28  
29  /**
30   * This interface represents the 'contract' obeyed by canonical blades in cladosG. 
31   * Implementing classes should be usable for blade multiplications.
32   * <br><br>
33   * The interface mostly focuses upon gettor methods, but eqivalent() is intended 
34   * for support of a Comparator. A few other static boolean tests appear here too.
35   * <br><br>
36   * @version 2.0
37   * @author Dr Alfred W Differ
38   */
39  public interface CanonicalBlade {
40  
41      /**
42       * This method compares the offered blade to the current one and reports true if they
43       * both use the same set of generators.
44       * <br><br>
45  	 * Examples<br>
46       * 1) Current blade uses e_1 and e_2. Offered blade uses e_1 and e_3. Report false.<br>
47       * 2) Current blade is for a scalar. Offered blade has no generators. Report true.<br>
48       * i) Current blade uses e_4 from a 5-generator basis. Offered blade uses only e_4. Report true.<br>
49       * <br><br>
50       * This method purposely ignores information available in a basis that would speak to 
51       * how many generators are present in a pscalar and checks ONLY the generator list kept
52       * by a blades internal representation. HOW the generators are stored doesn't matter to
53       * the interface. What matters is recognizing equivalent generator sets.
54       * <br><br>
55       * @param pCB1      is a blade to be used in the comparison
56       * @param pCB2      is a blade to be used in the comparison
57       * @return boolean  answers the question "Are the blades equivalent spaces?"
58       */
59      public static boolean equivalent(CanonicalBlade pCB1, CanonicalBlade pCB2) {
60          return (pCB1.bitKey() == pCB2.bitKey());
61      }
62  
63      /**
64  	 * Simple grade tester. Does the Blade contain 'n' generators?
65  	 * <br>
66  	 * @param blade Blade to be tested
67  	 * @param n     grade value
68  	 * @return TRUE if the blade has a number of generators matching the grade value
69  	 *         being tested. FALSE otherwise.
70  	 */
71      public static boolean isNBlade(CanonicalBlade blade, byte n) {
72          return blade.rank() == n;
73      }
74  
75      /**
76  	 * "Bit Key" is a base-2 representation of a blade's generators. All classes acting as blades
77       * must be able to return this bitKey even if they don't store it internally.
78  	 * <br><br>
79  	 * Examples<br>
80       * 1) A blade using e_2 must have +2 added to the bit key.<br>
81       * 2) A blade using e_4 must have +8 added to the bit key.<br>
82       * i) A blade using e_i must have 2^(i-1) added to the bit key.<br>
83       * <br><br>
84  	 * How a blade stores the bitKey (or if it even does) doesn't matter to the interface.
85       * What matters is that it be able to return a base-10 representation of the bit key.
86       * @return int base-10 representation of the blades' bit key.
87  	 */
88      public abstract int bitKey();
89  
90      /**
91       * This method delivers a stream of generators in the representing this blade. All blade classes
92       * must be able to stream their generators no matter how they represent generators.
93       * <br><br>
94       * <br><br>
95       * @return Stream of generators in the blade
96       */
97      public abstract Stream<?> generatorStream();
98  
99      /**
100 	 * "Key" is a representation of a blade used by a comparator. All classes acting as blades
101      * must be able to return this key even if they don't store it internally.
102 	 * <br><br>
103 	 * It is most likely that the "Key" will be the "Bit Key", but this interface doesn't 
104      * require that to be true. A Blades's comparator gets used to sort blades into a basis
105      * and Bit Key suffices, but there might be other requirements of the comparator that 
106      * Bit Key cannot serve. That's why this interface distinguishes Key from Bit Key.
107      * <br><br>
108 	 * How a blade stores the Key (or if it even does) doesn't matter to the interface.
109      * What matters is being able to return a base-10 long integer representation of the key.
110      * @return long base-10 representation of the blade's key.
111 	 */
112 	public abstract long key();
113 
114     /**
115      * This method reports how many unique generators are in the set representing this blade. All classes
116      * representing blades must be able to report 'rank' no matter how they represent generators.
117      * <br><br>
118      * @return byte describing the size of the generator set.
119      */
120     public abstract byte rank();
121 
122     /**
123      * This method reports on the permutation order of the generators in the set representing
124      * this blade. 
125      * <br><br>
126      * Cases:<br>
127      * 1) The generators are arranged in an even permutation of some standard order. Report +1.<br>
128      * 2) The generators are arranged in an odd permutation of some standard order. Report -1.<br>
129      * 3) The generator set has two of the same generators that are known to be degenerate. Report 0.<br>
130      * <br><br>
131      * Special Note about permutation order and degeneracy: A blade doesn't have to know about the 
132      * permutation order or whether degenerate generators are present. It has to be able to STORE 
133      * the information and report it when requested. That means blade implementations will have a 
134      * data element for 'sign' and a way to set it. This method is just the 'gettor'.
135      * <br><br>
136      * @return byte in the set {-1, 0, +1}
137      */
138     public abstract byte sign();
139     
140 }
