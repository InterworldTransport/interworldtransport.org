47  /**
48   * Nyads are for all practical purposes just lists of monads that share a common
49   * Foot, but not necessarily common or unique algebras. They can be used as mere
50   * lists, but they are intended to act more like transformations. For example, a
51   * nyad of order two contains two monads. If they are of different algebras,
52   * there is no path to simplifying them. No product or addition operation exists
53   * between the monads even though they share the same Foot. However, if one of
54   * the monads is multiplied against a different monad resulting in a scalar, the
55   * nyad can be contracted to one monad. There are other ways to accomplish this
56   * contraction as well and all of them imitate operations upon an operand.
57   * <br><br>
58   * The Nyad class in it's current form is immature. The list capability works,
59   * but the operation behaviors are yet to be written. This will most likely be
60   * done as the library gets used in physical models for field theories that
61   * require multi-algebra currents and potentials. The expected physical behavior
62   * of a 'classical' field theory from physics will inform the behaviors expected
63   * of CladosG Nyads.
64   * <br><br>
65   * Nyads ARE Modal because they contain modal objects. Nothing in the List
66   * nature of Nyads requires Modal, but specific Monad handling behavior does.
67   * <br><br>
68   * (Single monad nyads are essentially monads, but can be expanded.)
69   * <br><br>
70   * @version 2.0
71   * @author Dr Alfred W Differ
72   */
73  public class Nyad implements Modal {
74  	/**
75  	 * Return true if the Monads in the two lists are GEqual and the nyads are
76  	 * reference matches. Only monads sharing the same algebra name need to be
77  	 * checked against each other. No check is to be made for equality between the
78  	 * monad names.
79  	 * <br>
80  	 * This method is needed to compare Nyads since comparing instances via their
81  	 * variable names only checks to see if both variables reference the same place
82  	 * in memory
83  	 * <br>
84  	 * @param pT Nyad to be tested (first one)
85  	 * @param pN  Nyad to be tested (the other one)
86  	 * @return boolean
87  	 */
88  	public static final boolean isMEqual(Nyad pT, Nyad pN) {
89  		if (pT.getNyadMOrder() != pN.getNyadMOrder())		// Check if the Nyads are of the same order
90  			return false;									// Return false if they are not
91  
92  		if (pT.getNyadAOrder() != pN.getNyadAOrder())		// Check if the nyads algebra orders are the same
93  			return false;									// Return false if they are not
94  		
95  		if (pT.getFoot() != pN.getFoot())					// Check if the feet match
96  			return false;									// Return false if they don't	
97  		
98  		boolean fcheck = pT.monadStream().allMatch(y -> pN.monadStream().anyMatch(x -> x.isGEqual(y)));
99  		boolean bcheck = pN.monadStream().allMatch(y -> pT.monadStream().anyMatch(x -> x.isGEqual(y)));
100 		return fcheck & bcheck;								// These two ensure reflexive testing is complete
101 															// Failures in one should stop the other, but the streams will
102 															// halt in a protected way where loops risk mutability.	
103 	}

105 	/**
106 	 * This method performs a strong test for a reference match. All properties of Nyads must match  except for names. 
107 	 * Each monad in a nyad must have a counterpart in the other that is a reference matches.  There must be NO unpaired 
108 	 * monads, so the algebra lists have to be identical to within sorting as well.
109 	 * 
110 	 * Only monads sharing the same algebra need to be checked against each other for reference matches. 
111 	 * For those in the same algebra, we make use of the isRefereceMatch method and compare the two.
112 	 * <br>
113 	 * @param pT Nyad
114 	 * @param pN  Nyad
115 	 * @return boolean
116 	 */
117 	public static final boolean isStrongReferenceMatch(Nyad pT, Nyad pN) {
118 		if (pT.getNyadMOrder() != pN.getNyadMOrder())		// Check if the Nyads are of the same order
119 			return false;									// Return false if they are not
120 		
121 		if (pT.getNyadAOrder() != pN.getNyadAOrder())		// Check if the nyads algebra orders are the same
122 			return false;									// Return false if they are not
123 
124 		if (pT.getFoot() != pN.getFoot())					// Check if the feet match
125 			return false;									// Return false if they do not		
126 		
127 		boolean fcheck = pT.monadStream().allMatch(y -> pN.monadStream().anyMatch(x -> Monad.isReferenceMatch(x, y)));
128 		boolean bcheck = pN.monadStream().allMatch(y -> pT.monadStream().anyMatch(x -> Monad.isReferenceMatch(x, y)));
129 		return fcheck & bcheck;								// These two ensure reflexive testing is complete
130 															// Failures in one should stop the other, but the streams will
131 															// halt in a protected way where loops risk mutability.	
132 	}

134 	/**
135 	 * This method performs a weak test for a reference match. It is similar to the strong reference match, but 
136 	 * tolerates dangling monads. Only monads in each nyad that share an algebra must reference match.
137 	 * <br>
138 	 * Unpaired monads (those with no algebra matching counterpart in the other nyad) are counted as matches 
139 	 * against unit scalars even though the matching unit scalar monad isn't present. The isUnitMatch*() method 
140 	 * compares two monads this time because the monad lists are already filtered to ensure algebras already match.
141 	 * <br>
142 	 * @param pT Nyad
143 	 * @param pN  Nyad
144 	 * @return boolean
145 	 */
146 	public static final boolean isWeakReferenceMatch(Nyad pT, Nyad pN) {
147 		if (pT.getFoot() != pN.getFoot())					// Check to see if the Feet match
148 			return false;									// Return false if they do not	
149 
150 		if (pT.getNyadAOrder() == 0 | pN.getNyadAOrder() == 0)
151 			return true;									// If one nyad is empty, the other has danglers and passes.
152 
153 		if (!pT.algebraStream().anyMatch(y -> pN.algebraStream().anyMatch(x -> x.equals(y)))) 										
154 			return true;									// Edge case: ALL monads are danglers. Weak reference match occurs by default
155 		
156 		return  pT.algebraStream().anyMatch(y -> 			// Pick an algebra in pT [Failure to match means algebra in both and unit match failed]
157 					pT.monadInAlgebraStream(y).anyMatch(pTm -> 		// Pick a monad in pT using the algebra and find anyMatch of
158 						pN.monadInAlgebraStream(y).anyMatch(pNm -> 			// Pick a monad in pN using the algebra and find anyMatch of
159 							Monad.isUnitMatch(pNm, pTm) | !pN.hasAlgebra(y)			// The two monads being unit matches or pT's being a dangler
160 								)));
161 															// Any monad in pN not chosen for a check is a known dangler. No need to check it.
162 															// Any unit match for monads sharing an algebra suffices, so two nyads with 
163 															// two monads in the same algebra with different units will pass weak matching
164 															// but if they had differnt algebras (swapped) and the units swapped, they'd fail.
165 															// Examples filling in the truth table are obviously needed as test cases.
166 	}

278 	 * A simple copy constructor of a Nyad. The passed Nyad will be copied
279 	 * without the name. This constructor is used most often to clone other objects
280 	 * in every way except name.
281 	 * <br>
282 	 * The Foot object is re-used. The Algebra object is re-used. The Nyad's
283 	 * proto-number object is re-used. The Nyad's monad objects are copyied OR
284 	 * re-used depending on pCopy. 
285 	 * <br>
286 	 * @param pName String
287 	 * @param pN    Nyad
288 	 * @param pCopy boolean True - Copy monads first False - Re-use monads from Nyad
289 	 * @throws CladosNyadException  This exception is thrown when the offered Nyad
290 	 *                              is malformed. Make no assumptions!
291 	 */
292 	public Nyad(String pName, Nyad pN, boolean pCopy) throws CladosNyadException {
293 		if (pN.getNyadMOrder() == 0)
294 			throw new IllegalArgumentException("Offered Nyad to copy is empty.");
295 
296 		setName(pName);
297 		setFoot(pN.getFoot());
298 		mode = pN.getMonadAt(0).getMode();
299 		if (pN.monadList != null) {
300 			monadList = new ArrayList<Monad>(pN.monadList.size());
301 			algebraList = new ArrayList<Algebra>(pN.algebraList.size());
302 			if (pCopy)
303 				for (Monad tSpot : pN.monadList)
304 					appendMonadCopy(tSpot);
305 			else
306 				for (Monad tSpot : pN.monadList)
307 					appendMonad(tSpot);
308 		}
309 	}

321 	 * Add another Monad to the list of monads in this nyad. This method re-uses the Monad offered 
322 	 * as a parameter, so the Nyad DOES reference it.
323 	 * <br>
324 	 * Be aware that this method silently disallows adding the same monad to the list again. 
325 	 * This prevents two entries in the monad list pointing at the same monad. If two ARE needed, 
326 	 * use the appendMonadCopy method.
327 	 * <br>
328 	 * @param pM Monad to append to the list
329 	 * @throws CladosNyadException Exception thrown if the foot of the new monad fails to match
330 	 * @return Nyad
331 	 */
332 	public Nyad appendMonad(Monad pM) throws CladosNyadException {
333 		if (hasMonad(pM))									// If it is in the list...
334 			return this;									// silently return.
335 
336 		if (!pM.getAlgebra().getFoot().equals(getFoot()))
337 			throw new CladosNyadException(this, "Nyad / New Monad Foot mismatch");
338 
339 		monadList.ensureCapacity(monadList.size() + 1);		// Append monad to the list
340 		monadList.add(pM);
341 		resetFlags();
342 		return this;
343 	}

366 	/**
367 	 * Dyad anti-symmetric compression: 1/2 (left right - right left) Monads are placed
368 	 * in the same algebra and antisymmetrically multiplied. This ensures most reference
369 	 * match tests will succeed because the one on the right loses a reference to the
370 	 * algebra to which it once belonged. It is still possible for the test to fail, 
371 	 * though, because the monad modes and cardnials might still be different.
372 	 * <br><br>
373 	 * In a lot of cases, this method will produce nonsense. Simply defining the right
374 	 * side monad to be in the algebra for the left monad destroys the meaning carried
375 	 * by the right monad unless the two algebras are essentially the same. This happens,
376 	 * though, for cases where algebras are kept as book-keeping devices preventing 
377 	 * simplification of operations.
378 	 * <br><br>
379 	 * @param pInto int
380 	 * @param pFrom int
381 	 * @throws CladosNyadException 	This happens with a couple of edge cases involving
382 	 * 								list range errors and basis mis-match issues.
383 	 */
384 	public void compressAntiSymm(int pInto, int pFrom) throws CladosNyadException {
385 		if (pInto >= 0 & pFrom >=0 
386 				& pInto < monadList.size() 
387 				& pFrom < monadList.size()) { 					// Check for array out of bounds errors.
388 			Monad tLeft = monadList.get(pInto);
389 			Algebra tLeftAlg = tLeft.getAlgebra();				// Project into this one
390 			Monad tRight = monadList.get(pFrom);
391 			Algebra tRightAlg = tRight.getAlgebra();			// Keep a temp reference to the old one
392 
393 			if (tLeftAlg.getGBasis() == tRightAlg.getGBasis()) {// Proceed only if Basis is exact match
394 				tRight = projectOntoAlgebra(tLeft, tRight);		// second Monad is ALTERED HERE!
395 				tLeft.multiplyAntisymm(tRight);					// Only now can we do the deed.
396 
397 				monadList.remove(pFrom);						// Cleanup
398 				monadList.trimToSize();
399 				if (tLeftAlg != tRightAlg) {
400 					algebraList.remove(tRightAlg);				// More cleanup if needed
401 					algebraList.trimToSize();
402 				}
403 			} else {
404 				throw new CladosNyadException(this, "Anti-Symmetric Compression requires exact Basis match.");
405 			}
406 		} else {
407 			throw new CladosNyadException(this, "Anti-Symmetric Compression out of range error");
408 		}	
409 	}

412 	 * Dyad symmetric compression: 1/2 (left right + right left) Monads are placed
413 	 * in the same algebra and symmetrically multiplied. This ensures most reference
414 	 * match tests will succeed because the one on the right loses a reference to the
415 	 * algebra to which it once belonged. It is still possible for the test to fail, 
416 	 * though, because the monad modes and cardnials might still be different.
417 	 * <br><br>
418 	 * In a lot of cases, this method will produce nonsense. Simply defining the right
419 	 * side monad to be in the algebra for the left monad destroys the meaning carried
420 	 * by the right monad unless the two algebras are essentially the same. This happens,
421 	 * though, for cases where algebras are kept as book-keeping devices preventing 
422 	 * simplification of operations.
423 	 * <br><br>
424 	 * @param pInto int
425 	 * @param pFrom int
426 	 * @throws CladosNyadException 	This happens with a couple of edge cases involving
427 	 * 								list range errors and basis mis-match issues.
428 	 */
429 	public void compressSymm(int pInto, int pFrom) throws CladosNyadException {
430 		if (pInto >= 0 & pFrom >=0 
431 				& pInto < monadList.size() 
432 				& pFrom < monadList.size()) { 						// Check for array out of bounds errors.
433 			Monad tLeft = monadList.get(pInto);
434 			Algebra tLeftAlg = tLeft.getAlgebra();					// Project into this one
435 			Monad tRight = monadList.get(pFrom);
436 			Algebra tRightAlg = tRight.getAlgebra();				// Keep a temp reference to the old one
437 
438 			if (tLeftAlg.getGBasis() == tRightAlg.getGBasis()) {	// Proceed only if Basis is exact match
439 				tRight = Nyad.projectOntoAlgebra(tLeft, tRight);	// second Monad is ALTERED HERE!
440 				tLeft.multiplySymm(tRight);							// Only now can we do the deed.
441 
442 				monadList.remove(pFrom);							// Cleanup
443 				monadList.trimToSize();
444 				if (tLeftAlg != tRightAlg) {
445 					algebraList.remove(tRightAlg);					// More cleanup if needed
446 					algebraList.trimToSize();
447 				}
448 			} else {
449 				throw new CladosNyadException(this, "Symmetric Compression requires exact Basis match.");
450 			}
451 		} else {
452 			throw new CladosNyadException(this, "Symmetric Compression out of range error");
453 		}		
454 	}

456 	/**
457 	 * Create a new monad for this nyad using the prototype field and then append it
458 	 * to the end of the monadList. A 'zero' for the new algebra will be added to
459 	 * the list. This method creates a new algebra using the offered name and
460 	 * signature. It is not a copy method.
461 	 * <br>
462 	 * @param pMonadName    String
463 	 * @param pAlgebraName String
464 	 * @param pSig     String
465 	 * @param pCard    String
466 	 * @throws CladosMonadException    This exception is thrown when the new monad
467 	 *                                 constructor fails.
468 	 * @throws BadSignatureException   This exception is thrown when signature is
469 	 *                                 rejected as invalid.
470 	 * @throws CladosNyadException     This exception is thrown when the new monad
471 	 *                                 cannot be appended. Perhaps there is a
472 	 *                                 reference mismatch or the new monad failed
473 	 *                                 construction.
474 	 * @throws GeneratorRangeException This exception is thrown when the integer
475 	 *                                 number of generators for the basis is out of
476 	 *                                 the supported range. {0, 1, 2, ..., 14}
477 	 * @return Nyad
478 	 */
479 	public Nyad createMonad(String pMonadName, String pAlgebraName, String pSig, String pCard)
480 							throws 			BadSignatureException, GeneratorRangeException,
481 											CladosMonadException, CladosNyadException {
482 		//Prepare Cardinal and append to nyad's shared Foot if needed. Re-use where possible.
483 		Cardinal tCard = null;
484 		if (pCard == null) {								//No unit offered			
485 			if (getFoot().getCardinals().size() > 0) 		//Units present in Foot
486 				tCard = sharedFoot.getCardinal(0);		//Use the first one
487 			else {											//No units present in Foot
488 				tCard = Cardinal.generate(getMode());		//Create default mode unit
489 				sharedFoot.appendCardinal(tCard);			//Ensure Foot has default cardinal
490 			}
491 		} else {											//Unit specifically offered
492 			Optional<Cardinal> foundThis = getFoot().findCardinal(pCard);	//Find it in Nyad foot
493 			if (foundThis.isPresent()) 						//If Cardinal found
494 				tCard = foundThis.get();					//re-use it
495 			else {											//If not
496 				tCard = FBuilder.createCardinal(pCard);		//construct the Cardinal
497 				sharedFoot.appendCardinal(tCard);			//and ensure Foot knows about it.
498 			}
499 		}
500 		//Prepare algebra for monad if needed. Re-use where possible.
501 		Algebra tAlg = null;
502 		Optional<Algebra> foundAlg = algebraStream().filter(x -> x.getAlgebraName().equals(pAlgebraName)).findFirst();
503 		if(foundAlg.isPresent())							//Algebra found by name
504 			tAlg = foundAlg.get();							//and simply referenced
505 		else {											//Have to construct algebra not found
506 			Optional<GProduct> foundGP = GCache.INSTANCE.findGProduct(pSig);
507 			if (foundGP.isPresent()) 						//GP for new algebra already constructed
508 				tAlg = GBuilder.createAlgebraWithFootGP(sharedFoot, foundGP.get(), pAlgebraName);
509 			else 											//Don't need to hunt basis for re-use
510 				tAlg = GBuilder.createAlgebraWithFoot(sharedFoot, pAlgebraName, pSig);			
511 		}
512 		//With Algebra and a Cardinal to use, we can construct a viable Scale<ProtoN child> now.
513 		Scale<?> tScale = null;
514 		switch (mode) {
515 			case COMPLEXD -> new Scale<ComplexD>(mode, tAlg.getGBasis(), tCard);
516 			case COMPLEXF -> new Scale<ComplexF>(mode, tAlg.getGBasis(), tCard);
517 			case REALD -> new Scale<RealD>(mode, tAlg.getGBasis(), tCard);
518 			case REALF -> new Scale<RealF>(mode, tAlg.getGBasis(), tCard);
519 			default -> throw new IllegalArgumentException("Unexpected ProtoN child: " + mode);
520 		}
521 		//The point was to use GBuilder.createMonadWithAlgebra(Scale<T> pNumbers, Algebra pA, String pName)
522 		this.appendMonad(GBuilder.createMonadWithAlgebra(tScale, tAlg, pMonadName));
523 		return this;										//All done!
524 	}

755 	 * This method determines whether or not the Nyad is a pscalar in the algebra in question. 
756 	 * It works essentially the same way as isScalarAt. It checks all monads using the offered algebra.

991 	/**
992 	 * This method resets the internal list of algebras associated with the nyad. It streams the monads 
993 	 * and copies references to unique algebras found along the way. The juxtapositon and composition 
994 	 * flags are reset based on what conditions are found in the monad and algebra lists.
995 	 * <br><br>
996 	 * |-------Flags-------|---compositionFlag = True---|---compositionFlag = False---|<br>
997 	 * |-------------------|-------algebra # = 0 or 1---|-------------algebra # > 1---|<br>
998 	 * |---jFlag = True----|----[Zero or One Monad]-----|----[monad # = algebra #]----|<br>
999 	 * |-------------------|----------------------------|-----------------------------|<br>
1000 	 * |---jFlag = False---|--------[One Algebra]-------|----[monad # > algebra #]----|<br>
1001 	 * |-------------------|----------------------------|-----------------------------|<br>
1002 	 */
1003 	protected void resetFlags() {
1004 		algebraList.clear();
1005 		algebraList.ensureCapacity(monadList.size());
1006 
1007 		if (monadList.size() == 0) {							//Empty set edge case
1008 			jFlag = true;										
1009 			compositionFlag = true;
1010 			return;												//and we're done.
1011 		}
1012 		monadStream().forEach(m -> {							//At least one monad to process
1013 			if (!algebraList.contains(m.getAlgebra()))			//Stash a reference to the algebra if 
1014 				algebraList.add(m.getAlgebra());				//it isn't already stashed.
1015 		});
1016 		//Collections.sort(algebraList); 						//Can't think of why we should bother sorting the list.
1017 																//Now we set/reset the state flags.
1018 		if (monadList.size() == 1) {							//Singlton nyad edge case is similar to empty set, so...
1019 			jFlag = true;										//is a juxtaposition by default
1020 			compositionFlag = true;								//and a composition by default.
1021 		} else if (algebraList.size() == 1) {					//Multiple monads all sharing an algebra
1022 			jFlag = false;										//are not juxtapositions
1023 			compositionFlag = true;								//but ARE compositions.
1024 		} else if (monadList.size() == algebraList.size()) {	//Multiple monads and just as many algebras
1025 			jFlag = true;										//ARE juxtapositions
1026 			compositionFlag = false;							//and never compositions.
1027 		} else {												//Monads outnumber Algebras and more than one of each, so...
1028 			jFlag = false;										//it is NOT a juxtaposition
1029 			compositionFlag = false;							//and NOT a composition. (Mixed Case)
1030 		}	
1031 	}
