33  /**
34   * This class defines a geometric product on an associated basis within a Clifford Algebra. The flat space on which 
35   * multiplication is defined is assumed to be locally tangent to a manifold, but the difference in product results 
36   * form one tangent space to another are NOT tracked here. Only the structure of the geometric product on a canonical 
37   * basis is. Other distinctions are kept in the algebra objects that reference a GProduct.
38   * <br><br>
39   * A GProduct object actually assumes it is OK to perform a requested operation and will throw an exception if it 
40   * discovers later that it isn't. This is true most everywhere except in the constructor where input is examined first.
41   * <br><br>
42   * Most errors can be avoided by using GBuilder to construct this object. However, it shouldn't be necessary to 
43   * construct a GProduct directly. Best practice is to create an algebra and let it construct its product.
44   * <br><br>
45   * The implemented interface is currently all the methods available in this class. That will change in the future as 
46   * helper methods are built here that need not be exposed elsewhere.
47   * <br><br>
48   * @version 2.0
49   * @author Dr Alfred W Differ
50   */
51  public class GProduct implements CliffordProduct, Comparable<GProduct> {
52  
53  	/**
54  	 * This basis holds a representation of all the elements that can be built from the generators to span the 
55  	 * algebra's vector space. It is the object that Ken Greider called the Eddington Basis.
56  	 */
57  	private final Basis canonBasis;
58  
59  	/**
60  	 * This array holds the geometric multiplication table for a Clifford algebra using the associated basis. 
61  	 * The array contains numbers that represent the blade # one would produce with a product of blades (row) 
62  	 * and (column) of result.
63  	 * <br><br>
64  	 * Negative results in the array imply the resulting blade is scaled by -1.
65  	 */
66  	private final int[][] result;
67  
68  	/**
69  	 * This string holds the signature information describing the squares of all geometry generators present on 
70  	 * the multiplication table.
71  	 * <br><br>
72  	 * The term 'signature' is currently overloaded with meanings. The one being used here is a long form aggregate 
73  	 * of '+', '-', and '0' characters encoding the squares of an algebra's generators. The short form that reports 
74  	 * the count for each character as a list of three integers is NOT in use here. That means this signature string 
75  	 * has the details one expects in a quadratic form after generators have been assigned roles in a basis.
76  	 */
77  	private final String signature;
78  
79  	/**
80  	 * This array is an integer representation of the signature string. Generators with positive squares are one (1), 
81  	 * negative squares are minus one (-1), and degenerate ones as (0). The order of the array ALWAYS matches the order
82  	 * of the generators in the basis.
83  	 * <br><br>
84  	 * This array is kept to increase the speed of product calculations.
85  	 */
86  	private final byte[] nSignature;
87  
88  	/**
89  	 * Main constructor of GProduct with signature information passed in. It figures out the rest of what it needs.
90  	 * <br><br>
91  	 * @param pSig String form of the signature. Looks like "-+++0".
92  	 * @throws BadSignatureException   Thrown when an invalid signature is found
93  	 */
94  	public GProduct(String pSig) throws BadSignatureException {
95  		this(Optional.ofNullable(null), pSig);
96  	}
97  
98  	/**
99  	 * A re-use constructor of GProduct with signature and Basis passed in. It figures out the rest of what it needs.
100 	 * <br>
101 	 * The size of the signature string used to be checked using a static method on Basis, but that was duplicating the 
102 	 * effort performed by CliffordProduct when it checks the validity of the string. Size and characters ARE checked.
103 	 * <br>
104 	 * @param pSig String form of the signature. Looks like "-+++0".
105 	 * @param pB   Optional Basis to re-use in constructing this product.
106 	 * @throws BadSignatureException   Thrown when an invalid signature is found
107 	 */
108 	public GProduct(Optional<Basis> pB, String pSig) throws BadSignatureException {
109 		if (!CliffordProduct.validateSignature(pSig))	throw new BadSignatureException(this, "Valid signature required.");
110 		signature = pSig;
111 		nSignature = (pSig.length() == 0) ? new byte[1] : new byte[pSig.length()];				//initialize nSignature[]
112 		IntStream.range(0, pSig.length()).forEach(i -> {
113 															switch (pSig.charAt(i)){
114 																case '+' -> nSignature[i] = 1;
115 																case '0' -> nSignature[i] = 0;
116 																case '-' -> nSignature[i] = -1;
117 																}
118 															});						
119 		canonBasis = (pB.isPresent()) 	? pB.get() 												//Get A Basis. 
120 										: GBuilder.createBasis((byte) pSig.length());			//Create it if necessary.
121 		result = new int[getBladeCount()][getBladeCount()];										//Initialize the Cayley Table
122 		canonBasis.bladeStream().parallel().forEach(bladeLeft -> {								//streaming the row blades
123 									int row = canonBasis.find(bladeLeft) - 1;					//discover the blade's basis index
124 									canonBasis.bladeStream().parallel().forEach(bladeRight -> {	//streaming the column blades
125 													int col = canonBasis.find(bladeRight) - 1;	//discover the blade's basis index
126 		/* Compute result blade */					Blade bMult = BladeDuet.simplify(bladeLeft, bladeRight, nSignature);
127 		/* Time to fill the Cayley Table */						result[row][col] = ((int) bMult.sign() != 0)
128 		/* Discover result blade's basis index & fix sign */	? (int) bMult.sign() * (int) canonBasis.find(bMult)
129 		/* Result sign was zero */								: 0;							//Degenerate generator paired up
130 													});											//0 & 1 indicate a scalar blade
131 									});
132 	}

134 	@Override
135 	public boolean equals(Object obj) {
136 		if (this == obj)					return true;
137 		if (obj == null)					return false;
138 		if (getClass() != obj.getClass())	return false;
139 		if (!canonBasis.equals(((GProduct) obj).getBasis()))	return false;
140 		if (!signature.equals(((GProduct) obj).signature))		return false;		
141 		return true;
142 	}

146 	 * anticommute. Return a 0 otherwise.
147 	 * <br>
148 	 * @param pRow int
149 	 * @param pCol int
150 	 * @return int

154 		return (result[pRow][pCol] == result[pCol][pRow]) ? 0 : 1;

178 	 * Return a measure of whether blades pj and pk commute. Return a 1 if they
179 	 * commute. Return a 0 otherwise.

190 	/**
191 	 * This method takes the offered blade and hunts through the Cayley table to find another one such that the offered one 
192 	 * left multiplied by the other produces the +pscalar blade.
193 	 * <br><br>
194 	 * This method is a brute force search
195 	 * <br><br>
196 	 * @param pB1 Blade for which a complement blade is sought
197 	 * @return Blade that is the complement of the input blade
198 	 */
199 	public final Blade getComplementLeft(Blade pB1) {
200 		int row = canonBasis.find(pB1) - 1;			//pB1 determines row to hunt through
201 		int hunt = getBladeCount() - 1;				//index of the pscalar blade... always at the end of the basis.
202 		
203 		Optional<Blade> maybeThis = canonBasis	.bladeStream().parallel()
204 												.filter(b -> Math.abs(result[row][canonBasis.find(b)-1]) - 1 == hunt)
205 												.findAny();		//THIS IS A BLADE IN THE BASIS. Alter it at your peril.
206 		byte sign = (byte) getSign(pB1, canonBasis.getPScalarBlade());
207 		if (maybeThis.isPresent())	return maybeThis.get().setSign((sign == 0) ? 1 : sign);
208 		else						return null;
209 	}
210 
211 	/**
212 	 * This method takes the offered blade and hunts through the Cayley table to find another one such that the offered one 
213 	 * right multiplied by the other produces the +pscalar blade.
214 	 * <br><br>
215 	 * This method is a brute force search
216 	 * <br><br>
217 	 * @param pB1 Blade for which a complement blade is sought
218 	 * @return Blade that is the complement of the input blade
219 	 */
220 	public final Blade getComplementRight(Blade pB1) {
221 		int col = canonBasis.find(pB1) - 1;			//pB1 determines column to hunt down
222 		int hunt = getBladeCount() - 1;				//index of the pscalar blade... always at the end of the basis.
223 
224 		Optional<Blade> maybeThis = canonBasis	.bladeStream().parallel()
225 												.filter(b -> Math.abs(result[canonBasis.find(b)-1][col]) - 1 == hunt)
226 												.findAny();		//THIS IS A BLADE IN THE BASIS. Alter it at your peril.
227 		byte sign = (byte) getSign(canonBasis.getPScalarBlade(), pB1);
228 		if (maybeThis.isPresent())	return maybeThis.get().setSign((sign == 0) ? 1 : sign);
229 		else						return null;
230 	}

242 	/**
243 	 * Get start and end index from the GradeRange array for grade pGrade.
244 	 * <br>
245 	 * There is currently no protection on this method. If someone asks for a grade
246 	 * that isn't in range, they WILL get -1 in the cells.
247 	 * <br>
248 	 * @param pGrade byte primitive = grade for which the range is needed
249 	 * @return int[] start and end indexes returned as a int[] array
250 	 */
251 	@Override
252 	public final int[] getGradeRange(byte pGrade) {
253 		int[] tR = new int[2];
254 		tR[0] = (int) canonBasis.getGradeStart(pGrade);
255 		tR[1] = (int) ((pGrade == canonBasis.getGradeCount() - 1) // is this MaxGrade? If so, top=bottom
256 				? tR[0]
257 				: (canonBasis.getGradeStart((byte) (pGrade + 1)) - 1));
258 		return tR;
259 	}

262 	public final int[] getPScalarRange() {
263 		int[] tR = new int[2];
264 		tR[0] = canonBasis.getPScalarStart();
265 		tR[1] = tR[0];
266 		return tR;
267 	}

291 	/**
292 	 * This method takes two blades, finds their index values in the basis, and then retrieves
293 	 * the index value of their product from the Cayley table. With the result index, it return the 
294 	 * basis blade at that location.
295 	 * <br><br>
296 	 * One could just multiply the two blades using BladeDuet.simplify() and get the blade returned.
297 	 * That's how the Cayley table was constructed in the first place.
298 	 * <br><br>
299 	 * @param pRow 	Blade acting as the row entry for the Cayley table result
300 	 * @param pCol	Blade acting as the column entry for the Cayley table result
301 	 * @return Blade representing the result found in the Cayley table. If none found, null is returned.
302 	 */
303 	public final Blade getResult(Blade pRow, Blade pCol) {
304 		if (canonBasis.hasBlade(pRow) & canonBasis.hasBlade(pCol)) {
305 			int p = canonBasis.find(pRow) - 1;
306 			int q = canonBasis.find(pCol) - 1;	
307 			return canonBasis.getSingleBlade(	(getResult(p, q)==0) ? 0 : Math.abs(getResult(p, q)) - 1	);	
308 		}		
309 		return null;
310 	}

312 	/**
313 	 * Return the sign of an element in the geometric multiplication result table.
314 	 * <br>
315 	 * @param pRow int
316 	 * @param pCol int
317 	 * @return int
318 	 */
319 	@Override
320 	public final int getSign(int pRow, int pCol) {
321 		return (result[pRow][pCol] < 0) ? -1 : (result[pRow][pCol] > 0) ? 1 : 0;
322 	}
323 
324 	/**
325 	 * This method takes two blades, finds their index values in the basis, and then retrieves
326 	 * the sign variation at that location in the Cayley table. 
327 	 * <br><br>
328 	 * @param pRow 	Blade acting as the row entry for the Cayley table result
329 	 * @param pCol	Blade acting as the column entry for the Cayley table result
330 	 * @return int representing the sign of the result found in the Cayley table. If none found, 0 is returned.
331 	 */
332 	public final int getSign(Blade pRow, Blade pCol) {
333 		if (canonBasis.hasBlade(pRow) & canonBasis.hasBlade(pCol)) {
334 			int p = canonBasis.find(pRow) - 1;
335 			int q = canonBasis.find(pCol) - 1;	
336 			return getSign(p, q);
337 		}		
338 		return 0;
339 	}
