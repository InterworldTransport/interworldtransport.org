25  package org.interworldtransport.cladosG;
26  
27  import java.util.ArrayList;
28  import java.util.Optional;
29  
30  /**
31   * Any classes within CladosG that benefit from a supporting cache make use of this 
32   * singleton enumeration as a 'builder'. Nothing fancy here otherwise. Just simple 
33   * create, append, find, and remove capabilities backed by ArrayLists of cached objects.
34   * <br><br>
35   * @version 2.0
36   * @author Dr Alfred W Differ
37   */
38  public enum GCache {
39  	/**
40  	 * There is an implicit private constructor for this, but we won't override it.
41  	 */
42  	INSTANCE;
43  
44  	/**
45  	 * Heads Up! These ArrayLists ensure this 'enumeration' is mutable. This class
46  	 * is a cache, so this should suprise no one. It is supposed to keep track of
47  	 * the CladosG objects that can be safely shared in use.
48  	 */
49  	private ArrayList<Basis> listOfBases = new ArrayList<>(1);
50  	/**
51  	 * Heads Up! These ArrayLists ensure this 'enumeration' is mutable. This class
52  	 * is a cache, so this should suprise no one. It is supposed to keep track of
53  	 * the CladosG objects that can be safely shared in use.
54  	 */
55  	private ArrayList<GProduct> listOfGProducts = new ArrayList<>(1);
56  
57  	/**
58  	 * Method appends offered basis to cache IF one like it is not already present.
59  	 * If it IS, nothing is done and the method silently returns.
60  	 * <br>
61  	 * By 'like it' we mean the basis objects have the same number of generators
62  	 * used to construct them. It doesn't matter what signatures are used as a basis
63  	 * isn't aware of products except as lists. NO generator duplications should
64  	 * exist in blades within a basis, thus all N-generator basis objects are
65  	 * structurally the same even if represented by different instances.
66  	 * <br>
67  	 * Important NOTE. Basis objects capture structural meaning implied by a list of
68  	 * generators of an algebra. They do NOT capture the meaning of the generators
69  	 * themselves. Generators in one algebra need not mean the same thing as
70  	 * generators in another algebra, but the structure created in a basis is the
71  	 * same between algebras if the basis for each shares the same number of
72  	 * generators.
73  	 * <br>
74  	 * @param pB Basis to be appended to the cache IF not already present.
75  	 */
76  	public void appendBasis(Basis pB) {
77  		if (!listOfBases.contains(pB))
78  			listOfBases.add(pB);
79  	}
80  
81  	/**
82  	 * Method appends offered product to cache IF one like it is not already
83  	 * present. If it IS, nothing is done and the method silently returns.
84  	 * <br>
85  	 * By 'like it' we mean the product objects have the same number of generators
86  	 * and identical signatures used to construct them. NO generator duplications
87  	 * will survive in a product table after reducing blade combinations using the
88  	 * product's signature, so there is structural similarity between product tables
89  	 * that use the same number of generators (because of basis similarities) and
90  	 * the same signatures.
91  	 * <br>
92  	 * Important Note. GProduct objects capture structural meaning implied in
93  	 * a product table of elements of a Basis. As with a basis, no meaning
94  	 * to the generators or blades is implied in a product table.
95  	 * <br>
96  	 * @param pGP GProduct to be appended to the cache IF not already
97  	 *            present.
98  	 */
99  	public void appendGProduct(GProduct pGP) {
100 		if (!listOfGProducts.contains(pGP)){
101 			listOfGProducts.add(pGP);
102 			if(!listOfBases.contains(pGP.getBasis()))
103 				listOfBases.add(pGP.getBasis());
104 		}
105 	}
106 
107 	/**
108 	 * This is for resetting the cache of basis objects. It should rarely be used
109 	 * since they don't take up a lot of space, but it is faster than removing one
110 	 * at a time.
111 	 */
112 	public void clearBases() {
113 		listOfBases = new ArrayList<>(1);
114 	}
115 
116 	/**
117 	 * This is for resetting the cache of gproduct objects. It should be used sparingly
118 	 * since gproducts can be time consuming to recreate for large algebras. It 
119 	 * shouldn't HAVE to be done, but it is faster than removing one at a time.
120 	 */
121 	public void clearGProducts() {
122 		new ArrayList<>(1);
123 	}
124 
125 	/**
126 	 * This method returns an Optional of Basis using the integer number of
127 	 * generators offered for the search. If found, the optional will be engaged. If
128 	 * not, it will be disengaged. IF by some chance there are two basis instances
129 	 * in the cache by the same number of generators (which should NOT happen) the
130 	 * first one found will be returned in the Optional.
131 	 * <br>
132 	 * @param pGen byte integer of generators in a basis to be found in the cache
133 	 * @return Optional of Basis matching the number of generators offered.
134 	 */
135 	public Optional<Basis> findBasis(byte pGen) {
136 		return listOfBases.stream().filter(x -> (x.getGradeCount() - 1) == pGen).findFirst();
137 	}
138 
139 	/**
140 	 * This method returns an Optional of GProduct using the String signature
141 	 * offered for the search. If found, the optional will be engaged. If not, it
142 	 * will be disengaged. IF by some chance there are two product instances in the
143 	 * cache by the same signatures (which should NOT happen) the first one found
144 	 * will be returned in the Optional.
145 	 * <br>
146 	 * @param pSig String signature in a product to be found in the cache
147 	 * @return Optional of GProduct matching the signature offered.
148 	 */
149 	public Optional<GProduct> findGProduct(String pSig) {
150 		return listOfGProducts.stream().filter(x -> x.signature().equals(pSig)).findFirst();
151 	}
152 
153 	/**
154 	 * Simple gettor for the size of the basis cache. Since there is a limit to the
155 	 * Generator enumeration, there is also a limit to the basis cache. One should
156 	 * NEVER see more than CladosConstant.MAXGRADE basis objects in the cache.
157 	 * <br>
158 	 * @return byte integer of the size of the cache of basis instances.
159 	 */
160 	public byte getBasisListSize() {
161 		return (byte) listOfBases.size();
162 	}
163 
164 	/**
165 	 * Simple gettor for the size of the product cache. Since there is a limit to
166 	 * the Generator enumeration, there is also a limit to the product cache. One
167 	 * should NEVER see more than 2^(CladosConstant.MAXGRADE+1)-1 product objects in
168 	 * the cache.
169 	 * <br>
170 	 * @return integer of the size of the cache of basis instances.
171 	 */
172 	public int getGProductListSize() {
173 		return listOfGProducts.size();
174 	}
175 
176 	/**
177 	 * Method removes explicit basis from cache IF present. If it IS NOT, nothing is
178 	 * done and the method silently returns.
179 	 * <br>
180 	 * @param pB Basis to remove from the cache IF present.
181 	 * @return boolean TRUE if removal succeed. FALSE otherwise.
182 	 */
183 	public boolean removeBasis(Basis pB) {
184 		return listOfBases.remove(pB);
185 	}
186 
187 	/**
188 	 * Method removes implied basis from cache IF one like it is present. If it IS
189 	 * NOT, nothing is done and the method silently returns.
190 	 * <br>
191 	 * An implied basis is simply one that matches the integer number of generators
192 	 * passed in as a parameter. If no basis is found, nothing is done. That covers
193 	 * error conditions too. For example, no basis exists with -1 generators, but
194 	 * this method will report TRUE as though it removed it. There is no harm in
195 	 * this since the point of this method is to clean out the cache and NOT to
196 	 * error check the calling object.
197 	 * <br>
198 	 * So... Generator size quality is NOT checked.
199 	 * <br>
200 	 * @param pGen byte integer number of generators in a basis to remove from the
201 	 *             cache IF present.
202 	 * @return boolean TRUE if removal succeed. FALSE otherwise.
203 	 */
204 	public boolean removeBasis(byte pGen) {
205 		Optional<Basis> B = findBasis(pGen);
206 		if (B.isEmpty())
207 			return true;
208 		return removeBasis(B.get());
209 	}
210 
211 	/**
212 	 * Method removes explicit product from cache IF present. If it IS NOT, nothing
213 	 * is done and the method silently returns.
214 	 * <br>
215 	 * @param pGP GProduct to remove from the cache IF present.
216 	 * @return boolean TRUE if removal succeed. FALSE otherwise.
217 	 */
218 	public boolean removeGProduct(GProduct pGP) {
219 		return listOfGProducts.remove(pGP);
220 	}
221 
222 	/**
223 	 * Method removes implied product from cache IF one like it is present. If it IS
224 	 * NOT, nothing is done and the method silently returns.
225 	 * <br>
226 	 * An implied product is simply one that matches the signature parameter and its
227 	 * integer size. If no product is found, nothing is done. That covers error
228 	 * conditions too. For example, no GProduct exists with -1 generators (no
229 	 * matter the signature) or 3 generators with "+*-" signature, but this method
230 	 * will report TRUE as though it removed them. There is no harm in this since
231 	 * the point of this method is to clean out the cache and NOT to error check the
232 	 * calling object.
233 	 * <br>
234 	 * So... Signature quality is NOT checked.
235 	 * <br>
236 	 * @param pSig String signature of product to remove from the cache IF present.
237 	 * @return boolean TRUE if removal succeed. FALSE otherwise.
238 	 */
239 	public boolean removeGProduct(String pSig) {
240 		Optional<GProduct> GP = findGProduct(pSig); // This function tries to find the passed signature
241 		if (GP.isEmpty())
242 			return true;									  // If not found, no worries. It is 'removed'.
243 		return removeGProduct(GP.get());
244 	}
245 }
