35  /**
36   * The algebra object holds all geometric details that support the definition of
37   * a multivector over a division field {Cl(p,q) x ProtoN} except for the
38   * actual field. That makes this a partial abstraction of an algebra. Once an
39   * actual division field is in the mix we are there, but that structure is
40   * reserved for the Monad class.
41   * <br><br>
42   * The primary data structures in a Algebra are a Basis and a GProduct.
43   * Between them they define the structure of operations an Algebra can support.
44   * The basis provides for most behaviors people know from vector spaces. The
45   * product provides the other behaviors people know from differential forms.
46   * Together, though, they enable linear combinations of multi-ranked sums, thus
47   * they step beyond familiar ground from forms and outer products AND the
48   * familiar ground of scalar-only multiplication in vector spaces. All elements
49   * of an algebra an participate in addition and multiplication and
50   * multiplicative commutativity is NOT expected.
51   * <br><br>
52   * This isn't the place to explain what Clifford Algebras are and what they do.
53   * This IS the place to point that that Clados extends the idea slightly in
54   * order to support future uses.
55   * <br><br>
56   * 1. An Algebra references a 'Foot' object to imitate a location where the
57   * algebra's geometry is expected to be a tangent space to some underlying
58   * curved sub-manifold. No attempt at curvature is made here, but the Foot
59   * object IS used in reference match tests. This is intentionally done to
60   * prevent different tangent spaces being compared. In a model that assumes
61   * curvature on the manifold, one must first transport their frame before making
62   * comparisons. No 'transport' capability is written for Clados, but it might be
63   * some day.
64   * <br><br>
65   * Anyone wanting to get around this feature need only declare one 'Foot' and
66   * then re-use it everywhere. The computational penalty is miniscule.
67   * <br><br>
68   * 2. There is a UUID string kept internally for use an XML variant of
69   * serialization. It has no geometric meaning. Think of it as a digital name.
70   * <br><br>
71   * 3. There is also a 'name' string for the human readable name of an algebra.
72   * It has no geometric meaning and is not used for anything important.
73   * <br><br>
74   * @version 2.0
75   * @author Dr Alfred W Differ
76   */
77  public final class Algebra implements Comparable<Algebra> {
78  	/**
79  	 * This is an exporter of internal details to XML. It exists to bypass certain
80  	 * security concerns related to Java serialization of objects.
81  	 * <br>
82  	 * @param pA     Algebra to be exported as XML data
83  	 * @param indent String of tab characters to assist with human readability of
84  	 *               output.
85  	 * @return String formatted as XML containing information about the Algebra
86  	 */
87  	public final static String toXMLString(Algebra pA, String indent) {
88  		if (indent == null)
89  			indent = "\t\t\t\t";
90  		StringBuilder rB = new StringBuilder(indent).append("<Algebra UUID=\"").append(pA.uuid).append("\" >\n");
91  		rB.append(indent).append("\t<Name>").append(pA.getAlgebraName()).append("</Name>\n");
92  		// -----------------------------------------------------------------------
93  		rB.append(Foot.toXMLString(pA.getFoot(), indent + "\t"));
94  		rB.append(GProduct.toXMLString(pA.getGProduct(), indent + "\t"));
95  		rB.append(indent).append("</Algebra>\n");
96  		return rB.toString();
97  	}

99  	/**
100 	 * All algebra types share some elements that are not dependent on number types.
101 	 * The first among them is the 'tangent point' of the sub-manifold represented
102 	 * by the algebra. This is the Foot.
103 	 */
104 	protected Foot foot;
105 
106 	/**
107 	 * The second among the common elements is the Eddington basis formed from all
108 	 * blades that can be produced through exterior products of generating
109 	 * 'coordinate' vectors. For N generators, there are 2^N blades.
110 	 */
111 	protected Basis gBasis;
112 	/**
113 	 * The third among the common elements is the geometric product table formed by
114 	 * every product possible using members of the Eddington basis. This class has a
115 	 * few helper methods for dealing with symmetric and antisymmetric products and
116 	 * detection of other useful conditions.
117 	 */
118 	protected GProduct gProduct;
119 	/**
120 	 * Finally, the algebra has a name because this helps distinguish different
121 	 * reference frames associated with the same Foot.
122 	 */
123 	protected String name;
124 	/**
125 	 * Unique string (hopefully) that provides a machine readable name more likely
126 	 * to be unique. Used by apps that need more than the human readable name to
127 	 * avoid duplicating objects unnecessarily.
128 	 * <br>
129 	 * 
130 	 */
131 	protected String uuid;
132 
133 	/**
134 	 * This is the constructor that assumes a full Algebra has already been
135 	 * constructed. This new one re-uses the objects in the one offered. No
136 	 * independent objects are made in this constructor except the algebra itself
137 	 * <br>
138 	 * THIS CONSTRUCTOR is one that enables algebras to function as light weight frames.
139 	 * <br>
140 	 * @param pNewName 	This is the Algebra's name
141 	 * @param pA 		This is the other Algebra to copy.
142 	 */
143 	public Algebra(String pNewName, Algebra pA) {
144 		setAlgebraName(pNewName);
145 		setFoot(pA.getFoot());							//RE-USE of Foot
146 		setGProduct(pA.getGProduct());					//RE-USE of GP
147 		gBasis = pA.getGProduct().getBasis();			//RE-USE of Basis
148 		uuid = UUID.randomUUID().toString();
149 		GCache.INSTANCE.appendBasis(gBasis);
150 		GCache.INSTANCE.appendGProduct(gProduct);
151 	}
152 
153 	/**
154 	 * This is the constructor that assumes a Foot and GProduct have been
155 	 * instantiated and will simply point at them. It takes in one string for 
156 	 * the algebra name as well and then produces the algebra. Nothing can be 
157 	 * wrong with the signature since the GProduct is already constructed.
158 	 * <br>
159 	 * THIS CONSTRUCTOR is the one that most enables algebras to function as light
160 	 * weight frames. Two algebras can have different names but share everything
161 	 * else and cause reference matches to fail. The effect is that the canonical
162 	 * basis in both algebras is the same, but the name differences ensure the
163 	 * mismatch needed to prevent unphysical operations.
164 	 * <br>
165 	 * @param pNewName	This is the Algebra's name
166 	 * @param pF    	This is the foot being offered for reference
167 	 * @param pGP   	This is the geometric product being offered for reference
168 	 */
169 	public Algebra(String pNewName, Foot pF, GProduct pGP) {
170 		setAlgebraName(pNewName);
171 		setFoot(pF);
172 		setGProduct(pGP);							//RE-USE of GP
173 		gBasis = pGP.getBasis();					//RE-USE of Basis
174 		uuid = UUID.randomUUID().toString();
175 		GCache.INSTANCE.appendBasis(gBasis);
176 		GCache.INSTANCE.appendGProduct(gProduct);
177 	}

179 	/**
180 	 * This is the constructor that assumes a Foot has been instantiated. 
181 	 * It takes in two strings (one name and a product signature) and the Foot 
182 	 * and produces an Algebra. If anything is wrong with the signature it throws 
183 	 * an exception. Any other error throws a general monad exception.
184 	 * <br>
185 	 * THIS CONSTRUCTOR is the one that enables algebras to function as medium
186 	 * weight frames. Two algebras can have different names and GProducts but share
187 	 * a Foot and cause reference matches to fail. This is the behavior necessary to
188 	 * prevent unintended operations between monads expressed using different
189 	 * signatures in their geometric products.
190 	 * <br>
191 	 * @param pNewName	This is the Algebra's name
192 	 * @param pF    	This is the foot being offered for reference
193 	 * @param pSig  	This is the signature of the GProduct
194 	 * @throws BadSignatureException   This constructor creates a new GProduct which
195 	 *                                 requires a signature for the generators. This
196 	 *                                 signature string must be parse-able or this
197 	 *                                 exception is thrown.
198 	 * @throws GeneratorRangeException This exception catches when the supported
199 	 *                                 number of generators is out of range.
200 	 */
201 	public Algebra(String pNewName, Foot pF, String pSig)
202 			throws BadSignatureException, GeneratorRangeException {
203 		this(	pNewName, 
204 				pF, 
205 				GBuilder.createGProduct(	GCache.INSTANCE.findBasis((byte) pSig.length()),
206 											pSig));
207 		GCache.INSTANCE.appendBasis(gBasis);
208 		GCache.INSTANCE.appendGProduct(gProduct);
209 	}

211 	/**
212 	 * This is the raw constructor that assumes only the number type has been
213 	 * instantiated. It takes in three strings (two names and a product signature)
214 	 * and the example ProtoN and produces an Algebra. If anything is wrong with
215 	 * the signature it throws one of two exceptions.
216 	 * <br>
217 	 * This is the constructor that ensures algebra reference match failures even
218 	 * when exactly the same string names are used to construct all its parts.
219 	 * Because the Foot object is constructed within, the algebra will be distinct
220 	 * by definition.
221 	 * <br>
222 	 * @param <D>  		This is the type of "Number" being offered
223 	 * @param pNewName	This is the Algebra's name
224 	 * @param pFootName This is the Foot's name
225 	 * @param pSig      This is the signature of the GProduct
226 	 * @param pF        This is the number type to use expressed as a ProtoN
227 	 * @throws BadSignatureException   This constructor creates a new GProduct which
228 	 *                                 requires a signature for the generators. This
229 	 *                                 signature string must be parse-able or this
230 	 *                                 exception is thrown.
231 	 * @throws GeneratorRangeException This exception catches when the supported
232 	 *                                 number of generators is out of range.
233 	 */
234 	public <D extends ProtoN & Field & Normalizable>  Algebra(	String pNewName, 
235 																String pFootName, 
236 																String pSig, 
237 																D pF)
238 			throws BadSignatureException, GeneratorRangeException {
239 		
240 		this(	pNewName, 
241 				GBuilder.createFoot(		pFootName, 
242 											pF.getCardinalString()),
243 				GBuilder.createGProduct(	GCache.INSTANCE.findBasis((byte) pSig.length()),
244 											pSig));
245 		GCache.INSTANCE.appendBasis(gBasis);
246 		GCache.INSTANCE.appendGProduct(gProduct);
247 	}

249 	/**
250 	 * This method is present to enable sorting of lists of algebras. It isn't
251 	 * critical in the geometric sense, but it might be useful in certain physical
252 	 * models.
253 	 * <br>
254 	 * @param pAnother Algebra This is the algebra to be name compared
255 	 * @return int -1 if the name of 'this' algebra is 'less' than that of pAnother.
256 	 *         0 if the two names are the same +1 if the name of this algebra is
257 	 *         'greater' than that of pAnother.
258 	 */
259 	@Override
260 	public int compareTo(Algebra pAnother) {
261 		if (this.name == null)						return (pAnother.name == null) ? 0 : 1; 			//Null name is larger than a non-null name
262 		else if (pAnother.name == null)				return -1;	//Null name is larger than a non-null name
263 		else if (this.name.equals(pAnother.name))	return 0;	
264 		else {													//Neither name is null. Nor are they equal
265 			char[] first = this.name.toCharArray();
266 			char[] second = pAnother.name.toCharArray();
267 			int loopLimit = (first.length <= second.length) ? first.length : second.length;
268 			for (int j = 0; j < loopLimit; j++) {
269 				if (first[j] < second[j])			return -1;	//Character in first name is smaller. Decision done.
270 				if (first[j] > second[j])			return +1;	//Character in second name is smaller. Decision done.
271 			}													//No decision. Loop to the next character
272 
273 			return (first.length < second.length) ? -1 : +1;	//We know the two names are the same up to the characters 
274 																//in the shorter one. We also know they can't be equal. 
275 																//So the longer one wins.
276 		}
277 	}

375 	/**
376 	 * This method is a little dangerous and should use used only with great
377 	 * caution. Ideally, it would never be used because an algebra is defined
378 	 * relative to a tangent point on a sub-manifold. Sometimes, however, one might
379 	 * find that two seemingly distinct feet are actually the same. In this limited
380 	 * case it should be possible for a model writer to adjust an algebra to point
381 	 * at a different foot after construction.
382 	 * <br>
383 	 * @param footPoint Foot for the Algebra to use as its 'tangent' contact point.
384 	 */
385 	public void setFoot(Foot footPoint) {
386 		foot = footPoint;
387 	}

389 	/**
390 	 * This method is a little dangerous and should use used only with great
391 	 * caution. Ideally, it would never be used because an algebra is defined
392 	 * relative to a tangent point and the coordinates there form both a geometric
393 	 * basis on which a geometric product is defined. Sometimes, however, one might
394 	 * find that two seemingly distinct feet are actually the same. In this limited
395 	 * case it should be possible to adjust an algebra to at a different gProduct
396 	 * after construction.
397 	 * <br>
398 	 * @param pGP GProduct GProduct object for the Algebra to use as its geometric
399 	 *            product operation on the canonical basis.
400 	 */
401 	protected void setGProduct(GProduct pGP) {
402 		gProduct = pGP;
403 		gBasis = pGP.getBasis();
404 	}
