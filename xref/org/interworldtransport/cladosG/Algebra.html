36  /**
37   * The algebra object holds all geometric details that support the definition of
38   * a multivector over a division field {Cl(p,q) x ProtoN} except for the
39   * actual field. That makes this a partial abstraction of an algebra. Once an
40   * actual division field is in the mix we are there, but that structure is
41   * reserved for the Monad class.
42   * <br><br>
43   * The primary data structures in a Algebra are a Basis and a GProduct.
44   * Between them they define the structure of operations an Algebra can support.
45   * The basis provides for most behaviors people know from vector spaces. The
46   * product provides the other behaviors people know from differential forms.
47   * Together, though, they enable linear combinations of multi-ranked sums, thus
48   * they step beyond familiar ground from forms and outer products AND the
49   * familiar ground of scalar-only multiplication in vector spaces. All elements
50   * of an algebra an participate in addition and multiplication and
51   * multiplicative commutativity is NOT expected.
52   * <br><br>
53   * This isn't the place to explain what Clifford Algebras are and what they do.
54   * This IS the place to point that that Clados extends the idea slightly in
55   * order to support future uses.
56   * <br><br>
57   * 1. An Algebra references a 'Foot' object to imitate a location where the
58   * algebra's geometry is expected to be a tangent space to some underlying
59   * curved sub-manifold. No attempt at curvature is made here, but the Foot
60   * object IS used in reference match tests. This is intentionally done to
61   * prevent different tangent spaces being compared. In a model that assumes
62   * curvature on the manifold, one must first transport their frame before making
63   * comparisons. No 'transport' capability is written for Clados, but it might be
64   * some day.
65   * <br><br>
66   * Anyone wanting to get around this feature need only declare one 'Foot' and
67   * then re-use it everywhere. The computational penalty is miniscule.
68   * <br><br>
69   * 2. There is a UUID string kept internally for use as an XML/JSON variant of
70   * serialization. It has no geometric meaning. Think of it as a digital name.
71   * <br><br>
72   * 3. There is also a 'name' string for the human readable name of an algebra.
73   * It has no geometric meaning and is not used for anything important.
74   * <br><br>
75   * @version 2.0
76   * @author Dr Alfred W Differ
77   */
78  public final class Algebra implements Comparable<Algebra> {

80  	/**
81  	 * This method constructs a subalgebra using the offered blade to produce its 'Support' vector space. The blade
82  	 * contains a set of generators. Use those generators as vectors for a vector space and the span of them is the
83  	 * 'Support'. All elements of that span are the vector space over which a Clifford algebra can be constructed,
84  	 * but for CladosG it is sufficient to extract the sub-signature relevant to the blade and then simply construct 
85  	 * the algebra for it.
86  	 * <br><br>
87  	 * One important point about the smaller algebra is it will NOT re-use the exact generators of the offered blade.
88  	 * If the input blade is the bivector E5,E7 the output algebra will have E1,E2 as its pscalar blade. If the 
89  	 * original relationship must be preserved, a nyad will have to be constructed relating E5(E7) in the larger 
90  	 * algebra to E1(E2) in the smaller one. It is a simple nyad to construct, but it is not done here.
91  	 * <br><br>
92  	 * The returned algebra re-uses the Foot and adopts a name similar to the input algebra.
93  	 * <br><br>
94  	 * @param pA 		Algebra to use as source material for the returned algebra
95  	 * @param pPScalar 	Blade to be used to indicate which element of the old basis acts as the pscalar in 
96  	 *		 			the new basis.
97  	 * @return Algebra	finished product that re-uses what it can from the offered algebra
98  	 * @throws BadSignatureException is likely never thrown unless the input algebra has a corrupted signature
99  	 */
100 	public final static Algebra supportOf(Algebra pA, Blade pPScalar) throws BadSignatureException {
101 		
102 		if (pPScalar == null & pA == null)		
103 			return null;										//Nothing to use for context
104 
105 		if (pPScalar == null )									//pA is NOT null here
106 			return new Algebra("CopyOf-"+pA.getAName(), pA);	//Assume pA's pscalar and just copy pA
107 
108 		if (pA == null)	{										//No context, but pPScalar is NOT null at least
109 			byte[] sig = new byte[pPScalar.rank()];					
110 			Arrays.fill(sig, (byte) 0);							//Default to degeneracy
111 			return new Algebra(	"", 					//nameless
112 								new Foot(""), 			//nameless
113 								new String(sig));				//k-blades will exist, but only barely
114 		}
115 																//Blade and context algebra exist. Now find relevant signature part.
116 		Generator[] pointor = (Generator[]) pPScalar.getGenerators().toArray();	//Array of generators in pPScalar. pointor.length=rank.
117 		byte[] refSignature = pA.getGP().nsignature();							//Recall that array index = generator.ord - 1
118 		StringBuffer charSig = new StringBuffer(pPScalar.rank());
119 		IntStream	.range(0, pointor.length)					//Determine index range from pPScalar.rank()
120 					.forEach(i -> {												//Iterate through pointer on index 'i'.
121 						switch (refSignature[(byte) (pointor[i].ord - 1)]) {	//switch on nsignature's byte at a generator.ord - 1
122 							case +1 -> charSig.append("+");				//pointer's generator sqaures to +1
123 							case 0 	-> charSig.append("0");				//pointer's generator sqaures to 0
124 							case -1 -> charSig.append("-");				//pointer's generator sqaures to -1
125 						}
126 					});															//charSig has the relevant part of pA's signature
127 		return new Algebra(	"SubAlgebraOf-"+pA.getAName(),						//This algebra will have the relevant signature but
128 							pA.getFoot(),										//not use the same exact generators given by the
129 							new String(charSig.toString()));					//input blade (pointor) whose maxgen was ignored.
130 	}

134 	 * The first among them is the 'tangent point' of the sub-manifold represented

146 	 * The third among the common elements is the geometric product table formed by

151 	protected GProduct gProduct;

166 	/**
167 	 * This is the constructor that assumes a full Algebra has already been
168 	 * constructed. This new one re-uses the objects in the one offered. No
169 	 * independent objects are made in this constructor except the algebra itself
170 	 * <br>
171 	 * THIS CONSTRUCTOR is one that enables algebras to function as light weight frames.
172 	 * <br>
173 	 * @param pNewName 	This is the Algebra's name
174 	 * @param pA 		This is the other Algebra to copy.
175 	 */
176 	public Algebra(String pNewName, Algebra pA) {
177 		setAName(pNewName);
178 		setFoot(pA.getFoot());							//RE-USE of Foot
179 		setGP(pA.getGP());					//RE-USE of GP
180 		gBasis = pA.getGP().getBasis();			//RE-USE of Basis
181 		uuid = UUID.randomUUID().toString();
182 		GCache.INSTANCE.appendBasis(gBasis);
183 		GCache.INSTANCE.appendGProduct(gProduct);
184 	}

186 	/**
187 	 * This is the constructor that assumes a Foot and GProduct have been
188 	 * instantiated and will simply point at them. It takes in one string for 
189 	 * the algebra name as well and then produces the algebra. Nothing can be 
190 	 * wrong with the signature since the GProduct is already constructed.
191 	 * <br>
192 	 * THIS CONSTRUCTOR is the one that most enables algebras to function as light
193 	 * weight frames. Two algebras can have different names but share everything
194 	 * else and cause reference matches to fail. The effect is that the canonical
195 	 * basis in both algebras is the same, but the name differences ensure the
196 	 * mismatch needed to prevent unphysical operations.
197 	 * <br>
198 	 * @param pNewName	This is the Algebra's name
199 	 * @param pF    	This is the foot being offered for reference
200 	 * @param pGP   	This is the geometric product being offered for reference
201 	 */
202 	public Algebra(String pNewName, Foot pF, GProduct pGP) {
203 		setAName(pNewName);
204 		setFoot(pF);
205 		setGP(pGP);							//RE-USE of GP
206 		gBasis = pGP.getBasis();					//RE-USE of Basis
207 		uuid = UUID.randomUUID().toString();
208 		GCache.INSTANCE.appendBasis(gBasis);
209 		GCache.INSTANCE.appendGProduct(gProduct);
210 	}

212 	/**
213 	 * This is the constructor that assumes a Foot has been instantiated. 
214 	 * It takes in two strings (one name and a product signature) and the Foot 
215 	 * and produces an Algebra. If anything is wrong with the signature it throws 
216 	 * an exception. Any other error throws a general monad exception.
217 	 * <br>
218 	 * THIS CONSTRUCTOR is the one that enables algebras to function as medium
219 	 * weight frames. Two algebras can have different names and GProducts but share
220 	 * a Foot and cause reference matches to fail. This is the behavior necessary to
221 	 * prevent unintended operations between monads expressed using different
222 	 * signatures in their geometric products.
223 	 * <br>
224 	 * @param pNewName	This is the Algebra's name
225 	 * @param pF    	This is the foot being offered for reference
226 	 * @param pSig  	This is the signature of the GProduct
227 	 * @throws BadSignatureException   This constructor creates a new GProduct which
228 	 *                                 requires a signature for the generators. This
229 	 *                                 signature string must be parse-able or this
230 	 *                                 exception is thrown.
231 	 */
232 	public Algebra(String pNewName, Foot pF, String pSig)
233 			throws BadSignatureException {
234 		this(	pNewName, 
235 				pF, 
236 				GBuilder.createGProduct(	GCache.INSTANCE.findBasis((byte) pSig.length()),
237 											pSig));
238 		GCache.INSTANCE.appendBasis(gBasis);
239 		GCache.INSTANCE.appendGProduct(gProduct);
240 	}

242 	/**
243 	 * This is the raw constructor that assumes only the number type has been
244 	 * instantiated. It takes in three strings (two names and a product signature)
245 	 * and the example ProtoN and produces an Algebra. If anything is wrong with
246 	 * the signature it throws one of two exceptions.
247 	 * <br>
248 	 * This is the constructor that ensures algebra reference match failures even
249 	 * when exactly the same string names are used to construct all its parts.
250 	 * Because the Foot object is constructed within, the algebra will be distinct
251 	 * by definition.
252 	 * <br>
253 	 * @param <D>  		This is the type of "Number" being offered
254 	 * @param pNewName	This is the Algebra's name
255 	 * @param pFootName This is the Foot's name
256 	 * @param pSig      This is the signature of the GProduct
257 	 * @param pF        This is the number type to use expressed as a ProtoN
258 	 * @throws BadSignatureException   This constructor creates a new GProduct which
259 	 *                                 requires a signature for the generators. This
260 	 *                                 signature string must be parse-able or this
261 	 *                                 exception is thrown.
262 	 */
263 	public <D extends ProtoN & Field & Normalizable>  Algebra(	String pNewName, 
264 																String pFootName, 
265 																String pSig, 
266 																D pF)
267 			throws BadSignatureException {
268 		
269 		this(	pNewName, 
270 				GBuilder.createFoot(		pFootName, 
271 											pF.getCardinalString()),
272 				GBuilder.createGProduct(	GCache.INSTANCE.findBasis((byte) pSig.length()),
273 											pSig));
274 		GCache.INSTANCE.appendBasis(gBasis);
275 		GCache.INSTANCE.appendGProduct(gProduct);
276 	}

278 	/**
279 	 * This method is present to enable sorting of lists of algebras. It isn't
280 	 * critical in the geometric sense, but it might be useful in certain physical
281 	 * models.
282 	 * <br>
283 	 * @param pAnother Algebra This is the algebra to be name compared
284 	 * @return int -1 if the name of 'this' algebra is 'less' than that of pAnother.
285 	 *         0 if the two names are the same +1 if the name of this algebra is
286 	 *         'greater' than that of pAnother.
287 	 */
288 	@Override
289 	public int compareTo(Algebra pAnother) {
290 		if (this.name == null)						return (pAnother.name == null) ? 0 : 1; 			//Null name is larger than a non-null name
291 		else if (pAnother.name == null)				return -1;	//Null name is larger than a non-null name
292 		else if (this.name.equals(pAnother.name))	return 0;	
293 		else {													//Neither name is null. Nor are they equal
294 			char[] first = this.name.toCharArray();
295 			char[] second = pAnother.name.toCharArray();
296 			int loopLimit = (first.length <= second.length) ? first.length : second.length;
297 			for (int j = 0; j < loopLimit; j++) {
298 				if (first[j] < second[j])			return -1;	//Character in first name is smaller. Decision done.
299 				if (first[j] > second[j])			return +1;	//Character in second name is smaller. Decision done.
300 			}													//No decision. Loop to the next character
301 
302 			return (first.length < second.length) ? -1 : +1;	//We know the two names are the same up to the characters 
303 																//in the shorter one. We also know they can't be equal. 
304 																//So the longer one wins.
305 		}
306 	}
