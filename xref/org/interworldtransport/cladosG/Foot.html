25  package org.interworldtransport.cladosG;
26  
27  import java.util.ArrayList;
28  
29  import org.interworldtransport.cladosF.Cardinal;
30  import org.interworldtransport.cladosF.FBuilder;
31  import org.interworldtransport.cladosF.ProtoN;
32  
33  /**
34   * Objects within the cladosG package have a number of attributes in common that
35   * relate to a point that describes where the geometry of the algebra is tangent
36   * to the manifold underneath it. In a flat space it is not really necessary to
37   * use a foot point, but a simple one is still used in the cladosG package to
38   * help avoid the trap of assuming transport of objects along the manifold
39   * doesn't cause them to re-orient and change into other objects.
40   * <br><br>
41   * At a minimum, a Foot is the name of an event and the type of numbers used 
42   * as wieghts for the tangent space. At this location it is
43   * assumed the geometry of the algebra is aligned with the tangent geometry of 
44   * the manifold at that point. This makes a Foot the root reference.
45   * <br><br>
46   * @version 2.0
47   * @author Dr Alfred W Differ
48   */
49  public final class Foot {
50  	/**
51  	 * This factory build method produces a new Foot with an empty cardinal list. It
52  	 * is intend to name an otherwise opaque constructor that creates a new Foot but
53  	 * re-uses a Cardinal.
54  	 * <br>
55  	 * @param pFootName String name for the new Foot
56  	 * @return Foot Factory method builds a new Foot re-using a Cardinal object.
57  	 */
58  	public static final Foot buildAsType(String pFootName) {
59  		return new Foot(pFootName);
60  	}
61  
62  	/**
63  	 * This factory build method produces a new Foot using the offered Cardinal. It
64  	 * is intend to name an otherwise opaque constructor that creates a new Foot but
65  	 * re-uses a Cardinal.
66  	 * <br>
67  	 * @param pFootName    String name for the new Foot
68  	 * @param pNewCardinal Cardinal to re-use for this new Foot
69  	 * @return Foot Factory method builds a new Foot re-using a Cardinal object.
70  	 */
71  	public static final Foot buildAsType(String pFootName, Cardinal pNewCardinal) {
72  		return new Foot(pFootName, pNewCardinal);
73  	}
74  
75  	/**
76  	 * This object defines the type of numbers used by all objects that share this
77  	 * footPoint.
78  	 */
79  	private ArrayList<Cardinal> cardinalList;
80  	/**
81  	 * This String is the name the footPoint. Think of it as naming the tangent
82  	 * point between a flat algebra and the curvy manifold.
83  	 */
84  	private String footName;
85  
86  	/**
87  	 * Build the Foot from scratch.
88  	 * <br>
89  	 * @param pName String This string will be the name of the foot point.
90  	 */
91  	public Foot(String pName) {
92  		setFootName(pName);
93  		cardinalList = new ArrayList<Cardinal>(1);
94  		cardinalList.add(FBuilder.createCardinal(pName));
95  	}
96  
97  	/**
98  	 * Build the Foot from scratch then put the Cardinal in the internal list.
99  	 * <br>
100 	 * @param pName String This string will be the name of the foot point.
101 	 * @param pFT   Cardinal This object defines the kind of numbers that are
102 	 *              meaningful for this foot point
103 	 */
104 	public Foot(String pName, Cardinal pFT) {
105 		setFootName(pName);
106 		cardinalList = new ArrayList<Cardinal>(1);
107 		cardinalList.add(pFT);
108 	}
109 
110 	/**
111 	 * Build the Foot from scratch then put the number's Cardinal in the internal list.
112 	 * <br>
113 	 * @param pName String This string will be the name of the foot point.
114 	 * @param pF    ProtoN This object holds the cardinal that defines the kind of
115 	 *              numbers that are meaningful for this foot point
116 	 */
117 	public Foot(String pName, ProtoN pF) {
118 		setFootName(pName);
119 		cardinalList = new ArrayList<Cardinal>(1);
120 		cardinalList.add(pF.getCardinal());
121 	}
122 
123 	/**
124 	 * This method appends a Cardinal to the list of known cardinals for this foot.
125 	 * It will silently terminate IF the cardinal is already in the list.
126 	 * <br>
127 	 * The uniqueness test is done BY OBJECT and not by the cardinal's string name.
128 	 * That means visual inspection of the cardinal list could reveal entries that
129 	 * appear to be the same. They aren't, though. The cardinalList references
130 	 * objects and NOT their string names. To avoid this use CladosFCache.
131 	 * <br>
132 	 * @param fN Cardinal reference to append to this Foot
133 	 */
134 	public void appendCardinal(Cardinal fN) {
135 		if (cardinalList.contains(fN))
136 			return;
137 		cardinalList.ensureCapacity(cardinalList.size() + 1);
138 		cardinalList.add(fN);
139 	}
140 
141 	/**
142 	 * This method looks for the requested cardinal object in the Foot's tracking
143 	 * list.
144 	 * <br>
145 	 * @param pIn Cardinal This is a reference to the Cardinal to be found in the
146 	 *            Foot's cardinal list.
147 	 * @return int This method looks for the requested Cardinal in the Foot's list
148 	 *         and returns the index if it is found. If not, it returns -1.
149 	 */
150 	public int findCardinal(Cardinal pIn) {
151 		if (cardinalList.isEmpty())
152 			return -1;
153 		if (cardinalList.contains(pIn))
154 			return cardinalList.indexOf(pIn);
155 		return -1;
156 	}
157 
158 	/**
159 	 * Simple gettor for one of the cardinals associated with this Foot.
160 	 * <br>
161 	 * Note that an error condition (like indexOutOfBounds) will return null.
162 	 * <br>
163 	 * @param pIn Integer index of Cardinal in this Foot to retrieve.
164 	 * @return Cardinal found at the integer index... or null.
165 	 */
166 	public Cardinal getCardinal(int pIn) {
167 		if (pIn >= 0 & pIn < cardinalList.size())
168 			return cardinalList.get(pIn);
169 		return null;
170 	}
171 
172 	/**
173 	 * Simple gettor for the entire list of cardinals associated with this Foot.
174 	 * <br>
175 	 * @return ArrayList of Cardinal instances associated with this Foot.
176 	 */
177 	public ArrayList<Cardinal> getCardinals() {
178 		return cardinalList;
179 	}
180 
181 	/**
182 	 * Simple gettor of the Foot's name element
183 	 * <br>
184 	 * @return String name of the Foot
185 	 */
186 	public String getFootName() {
187 		return footName;
188 	}
189 
190 	/**
191 	 * This method removes a Cardinal from the list of known cardinals for this
192 	 * foot. It will silently return IF the cardinal wasn't already on the list.
193 	 * <br>
194 	 * The uniqueness test is done BY OBJECT and not by the cardinal's string name.
195 	 * That means visual inspection of the cardinal list could reveal entries that
196 	 * appear to be the same. They aren't, though. The cardinalList references
197 	 * objects and NOT their string names. To avoid this use CladosFCache.
198 	 * <br>
199 	 * @param pCard Cardinal reference to remove to this Foot
200 	 */
201 	public void removeCardinal(Cardinal pCard) {
202 		cardinalList.remove(pCard);
203 	}
204 
205 	/**
206 	 * Simple setter of the Foot's name element.
207 	 * <br>
208 	 * @param footName String name of the Foot to set here
209 	 */
210 	public void setFootName(String footName) {
211 		this.footName = footName;
212 	}
213 
214 	/**
215 	 * This is an exporter of internal details to XML. It exists to bypass certain
216 	 * security concerns related to Java serialization of objects.
217 	 * <br>
218 	 * @param pF Foot the be exported as XML.
219 	 * @param indent String indentation to assist with human readability of output
220 	 *               XML data
221 	 * @return String formatted as XML containing information about the Foot
222 	 */
223 	public final static String toXMLString(Foot pF, String indent) {
224 		if (indent == null)
225 			indent = "\t\t";
226 		StringBuilder rB = new StringBuilder(indent + "<Foot>\n");
227 		rB.append(indent).append("\t<Name>").append(pF.getFootName()).append("</Name>\n");
228 		// -----------------------------------------------------------------------
229 		rB.append(indent).append("\t<Cardinals number=\"").append(pF.cardinalList.size()).append("\" >\n");
230 		for (Cardinal point : pF.cardinalList)
231 			rB.append(indent).append(Cardinal.toXMLString(point, "\t\t"));
232 		rB.append(indent).append("\t</Cardinals>\n");
233 		// -----------------------------------------------------------------------
234 		rB.append(indent).append("</Foot>\n");
235 		return rB.toString();
236 	}
