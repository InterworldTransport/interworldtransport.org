25  package org.interworldtransport.cladosG;
26  
27  import org.interworldtransport.cladosF.FBuilder;
28  import org.interworldtransport.cladosF.FCache;
29  import org.interworldtransport.cladosF.ProtoN;
30  
31  /**
32   * Objects within the cladosG package have a number of attributes in common that
33   * relate to a point that describes where the geometry of the algebra is tangent
34   * to the manifold underneath it. In a flat space it is not really necessary to
35   * use a foot point, but a simple one is still used in the cladosG package to
36   * help avoid the trap of assuming transport of objects along the manifold
37   * doesn't cause them to re-orient and change into other objects.
38   * <br><br>
39   * At a minimum, a Foot is the name of an event and the type of numbers used 
40   * as wieghts for the tangent space. At this location it is
41   * assumed the geometry of the algebra is aligned with the tangent geometry of 
42   * the manifold at that point. This makes a Foot the root reference.
43   * <br><br>
44   * @version 2.0
45   * @author Dr Alfred W Differ
46   */
47  public final class Foot {
48  	/**
49  	 * This factory build method produces a new Foot with an empty cardinal list. It
50  	 * is intend to name an otherwise opaque constructor that creates a new Foot but
51  	 * re-uses a Cardinal.
52  	 * <br>
53  	 * @param pFootName String name for the new Foot
54  	 * @return Foot Factory method builds a new Foot re-using a Cardinal object.
55  	 */
56  	public static final Foot buildAsType(String pFootName) {
57  		return new Foot(pFootName);
58  	}
59  
60  	/**
61  	 * This String is the name the footPoint. Think of it as naming the tangent
62  	 * point between a flat algebra and the curvy manifold.
63  	 */
64  	private String footName;
65  
66  	/**
67  	 * Build the Foot from scratch.
68  	 * <br>
69  	 * @param pName String This string will be the name of the foot point.
70  	 */
71  	public Foot(String pName) {
72  		setFootName(pName);
73  		FBuilder.createCardinal(pName);
74  	}
75  
76  	/**
77  	 * Build the Foot from scratch then put the number's Cardinal in the internal list.
78  	 * <br>
79  	 * @param pName String This string will be the name of the foot point.
80  	 * @param pF    ProtoN This object holds the cardinal that defines the kind of
81  	 *              numbers that are meaningful for this foot point
82  	 */
83  	public Foot(String pName, ProtoN pF) {
84  		setFootName(pName);
85  		FCache.INSTANCE.appendCardinal(pF.getCardinal());
86  	}
87  
88  	/**
89  	 * Simple gettor of the Foot's name element
90  	 * <br>
91  	 * @return String name of the Foot
92  	 */
93  	public String getFootName() {
94  		return footName;
95  	}
96  
97  	/**
98  	 * Simple setter of the Foot's name element.
99  	 * <br>
100 	 * @param footName String name of the Foot to set here
101 	 */
102 	public void setFootName(String footName) {
103 		this.footName = footName;
104 	}
105 
106 	/**
107 	 * This is an exporter of internal details to XML. It exists to bypass certain
108 	 * security concerns related to Java serialization of objects.
109 	 * <br>
110 	 * @param pF Foot the be exported as XML.
111 	 * @param indent String indentation to assist with human readability of output
112 	 *               XML data
113 	 * @return String formatted as XML containing information about the Foot
114 	 */
115 	public final static String toXMLString(Foot pF, String indent) {
116 		if (indent == null)		indent = "\t\t";
117 		StringBuilder rB = new StringBuilder(indent + "<Foot name=\"");
118 		rB	.append(pF.getFootName())
119 			.append("\" />\n");
120 		return rB.toString();
121 	}
122 }
