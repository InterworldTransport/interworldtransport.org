25  package org.interworldtransport.cladosGExceptions;
26  
27  /**
28   * This class is designed to be the top of the CladosG exception family.  
29   * All instances of descendants are assumed to originate from geometric objects
30   * with unhandled problems, but the objects themselves might not have been 
31   * successfully constructed, or they might not the the actual source.
32   * The message about the exception is common to all in the family.
33   * <br><br>
34   * @version 2.0
35   * @author Dr Alfred W Differ
36   *
37   */
38  public class CladosException extends Exception 
39  {
40  /**
41   * The source message is the reason given by the originating Monad for the exception.
42   */
43      private String  SourceMessage;
44  /**
45   * This method is the main constructor of all Clados Exceptions.
46   * It needs only a source Monad and message.
47   * @param pMessage	String
48   */
49      public CladosException(String pMessage) {
50      	super();
51      	SourceMessage=pMessage;
52      }
53  /**
54   * This method delivers a the explanation from the Monad that originated the Exception
55   * 
56   * @return String
57   */
58      public String  getSourceMessage() {
59      	return SourceMessage;
60      }
61  }
